```python
from dataclasses import dataclass
from datetime import datetime
from typing import List, Optional
from enum import Enum

class StormCategory(Enum):
    TD = "Tropical Depression"
    TS = "Tropical Storm"
    CAT1 = "Category 1"
    CAT2 = "Category 2"
    CAT3 = "Category 3"
    CAT4 = "Category 4"
    CAT5 = "Category 5"

@dataclass
class ActiveStorm:
    """Active tropical system from NHC."""
    id: str  # e.g., "AL012026" (Atlantic #1 of 2026)
    name: str  # e.g., "Alberto"
    category: StormCategory
    lat: float
    lon: float
    max_wind_mph: int
    movement: str  # e.g., "NW at 12 mph"
    pressure_mb: int
    forecast_track: List[dict]  # Future positions
    last_updated: datetime

    @property
    def is_major(self) -> bool:
        """Cat 3+ is considered major hurricane."""
        return self.category in [StormCategory.CAT3, StormCategory.CAT4, StormCategory.CAT5]

    @property
    def threatens_gom(self) -> bool:
        """Check if storm is in or heading toward Gulf of Mexico."""
        # GOM bounding box: 18-31°N, 80-98°W
        in_gom = 18 <= self.lat <= 31 and -98 <= self.lon <= -80
        # Also check forecast track
        for point in self.forecast_track:
            if 18 <= point.get("lat", 0) <= 31 and -98 <= point.get("lon", 0) <= -80:
                return True
        return in_gom

class NOAAHurricaneTracker:
    """Tracks active Atlantic hurricanes from NHC."""

    NHC_RSS_URL = "https://www.nhc.noaa.gov/index-at.xml"

    async def get_active_storms(self) -> List[ActiveStorm]:
        """Fetch current active storms from NHC RSS."""
        import httpx
        import feedparser

        async with httpx.AsyncClient() as client:
            response = await client.get(self.NHC_RSS_URL, timeout=30)
            feed = feedparser.parse(response.text)

        storms = []
        for entry in feed.entries:
            # Parse NHC RSS entry format
            storm = self._parse_nhc_entry(entry)
            if storm:
                storms.append(storm)

        return storms

    def _parse_nhc_entry(self, entry) -> Optional[ActiveStorm]:
        """Parse NHC RSS entry into ActiveStorm."""
        # NHC entries have structured format
        # Title: "Tropical Storm Alberto Advisory Number 5"
        # Summary contains position, winds, pressure

            # Extract data from entry
```

### Task 2: Create impact assessment

```python
# src/liquidity/weather/impact.py

@dataclass
class OilProductionImpact:
    """Estimated impact on oil production from weather event."""
    storm: ActiveStorm
    affected_platforms: int  # Estimated
    production_at_risk_bpd: int  # Barrels per day
    evacuation_likely: bool
    refinery_risk: str  # "none", "low", "moderate", "high"

def assess_gom_impact(storm: ActiveStorm) -> OilProductionImpact:
    """Assess potential oil production impact from storm."""
    # GOM produces ~1.8 million bpd offshore
    # Cat 3+ typically causes ~50-80% evacuation

    if storm.category in [StormCategory.CAT4, StormCategory.CAT5]:
        return OilProductionImpact(
            storm=storm,
            affected_platforms=200,
            production_at_risk_bpd=1_500_000,
            evacuation_likely=True,
            refinery_risk="high",
        )
    elif storm.category == StormCategory.CAT3:
        return OilProductionImpact(
            storm=storm,
            affected_platforms=150,
            production_at_risk_bpd=1_000_000,
            evacuation_likely=True,
            refinery_risk="moderate",
        )
    elif storm.category in [StormCategory.CAT1, StormCategory.CAT2]:
        return OilProductionImpact(
            storm=storm,
            affected_platforms=50,
            production_at_risk_bpd=300_000,
            evacuation_likely=False,
            refinery_risk="low",
        )
    else:
        return OilProductionImpact(
            storm=storm,
            affected_platforms=0,
            production_at_risk_bpd=0,
            evacuation_likely=False,
            refinery_risk="none",
        )
```
