### 2. Discord Client (discord.py)
```python
from discord_webhook import DiscordWebhook, DiscordEmbed
from dataclasses import dataclass
import os

@dataclass
class DiscordConfig:
    webhook_url: str
    username: str = "Liquidity Monitor"
    avatar_url: str | None = None
    rate_limit_seconds: int = 60  # Min time between same alert type

class DiscordClient:
    def __init__(self, config: DiscordConfig | None = None):
        self.config = config or DiscordConfig(
            webhook_url=os.getenv('LIQUIDITY_DISCORD_WEBHOOK_URL', '')
        )
        self._last_sent: dict[str, datetime] = {}

    def send_embed(self, embed: DiscordEmbed, alert_type: str) -> bool:
        """Send embed with rate limiting."""
        if not self._can_send(alert_type):
            return False

        webhook = DiscordWebhook(
            url=self.config.webhook_url,
            username=self.config.username,
            avatar_url=self.config.avatar_url,
        )
        webhook.add_embed(embed)
        response = webhook.execute()

        if response.status_code == 200:
            self._last_sent[alert_type] = datetime.utcnow()
            return True
        return False

    def _can_send(self, alert_type: str) -> bool:
        """Check rate limit for alert type."""
        if alert_type not in self._last_sent:
            return True
        elapsed = (datetime.utcnow() - self._last_sent[alert_type]).total_seconds()
        return elapsed >= self.config.rate_limit_seconds
```

### 3. Message Formatter (formatter.py)
```python
from discord_webhook import DiscordEmbed

class AlertFormatter:
    """Format alerts into Discord embeds."""

    @staticmethod
    def regime_change(
        previous: str,
        current: str,
        prev_intensity: int,
        curr_intensity: int,
        confidence: str,
        metrics: dict,
    ) -> DiscordEmbed:
        """Format regime change alert."""
        color = 0x00ff88 if current == 'EXPANSION' else 0xff4444
        emoji = '🟢' if current == 'EXPANSION' else '🔴'

        embed = DiscordEmbed(
            title=f"{emoji} REGIME CHANGE: {current}",
            color=color,
        )
        embed.add_embed_field(
            name="Previous",
            value=f"{previous} (intensity {prev_intensity})",
            inline=True,
        )
        embed.add_embed_field(
            name="Current",
            value=f"{current} (intensity {curr_intensity})",
            inline=True,
        )
        embed.add_embed_field(
            name="Confidence",
            value=confidence,
            inline=True,
        )
        embed.add_embed_field(
            name="Key Metrics",
            value=f"• Net Liquidity: ${metrics['net_liquidity']:.1f}T\n"
                  f"• Global Liquidity: ${metrics['global_liquidity']:.1f}T\n"
                  f"• DXY: {metrics['dxy']:.2f}",
            inline=False,
        )
        embed.set_timestamp()
        return embed

    @staticmethod
    def stress_breach(
        indicator: str,
        value: float,
        threshold: float,
        severity: str,
    ) -> DiscordEmbed:
        """Format stress threshold breach alert."""
        color = 0xffaa00 if severity == 'elevated' else 0xff4444
        emoji = '⚠️' if severity == 'elevated' else '🚨'

        embed = DiscordEmbed(
            title=f"{emoji} STRESS ALERT: {indicator}",
            color=color,
        )
        embed.add_embed_field(name="Current", value=f"{value:.1f} bps", inline=True)
        embed.add_embed_field(name="Threshold", value=f"{threshold:.1f} bps", inline=True)
        embed.add_embed_field(name="Severity", value=severity.upper(), inline=True)
        embed.set_timestamp()
        return embed

    @staticmethod
    def dxy_move(
        current: float,
        change_pct: float,
        direction: str,
    ) -> DiscordEmbed:
        """Format significant DXY move alert."""
        color = 0x00aaff if direction == 'up' else 0xffaa00
        emoji = '📈' if direction == 'up' else '📉'

        embed = DiscordEmbed(
            title=f"{emoji} DXY MOVE: {change_pct:+.2f}%",
            color=color,
        )
        embed.add_embed_field(name="DXY", value=f"{current:.2f}", inline=True)
        embed.add_embed_field(name="Change", value=f"{change_pct:+.2f}%", inline=True)
        embed.add_embed_field(
            name="Implication",
            value="USD strengthening (risk-off)" if direction == 'up'
                  else "USD weakening (risk-on)",
            inline=False,
        )
        embed.set_timestamp()
        return embed

    @staticmethod
    def correlation_shift(
        asset: str,
        previous: float,
        current: float,
        change: float,
    ) -> DiscordEmbed:
        """Format correlation regime shift alert."""
        color = 0x9933ff

        embed = DiscordEmbed(
            title=f"🔄 CORRELATION SHIFT: {asset}",
            color=color,
        )
        embed.add_embed_field(name="Previous", value=f"{previous:.2f}", inline=True)
        embed.add_embed_field(name="Current", value=f"{current:.2f}", inline=True)
        embed.add_embed_field(name="Change", value=f"{change:+.2f}", inline=True)
        embed.set_timestamp()
        return embed
```

### 4. Alert Handlers (handlers.py)
```python
from liquidity.analyzers import RegimeClassifier, CorrelationEngine
from liquidity.collectors import FXCollector, StressIndicatorCollector

class AlertHandlers:
    def __init__(
        self,
        discord: DiscordClient,
        formatter: AlertFormatter,
    ):
        self.discord = discord
        self.formatter = formatter
        self._previous_regime: str | None = None
        self._previous_correlations: dict[str, float] = {}

    def check_regime_change(self, classifier: RegimeClassifier, data) -> bool:
        """Check and alert on regime change."""
        result = classifier.classify(data)
        current = result.direction.value

        if self._previous_regime and current != self._previous_regime:
            embed = self.formatter.regime_change(
                previous=self._previous_regime,
                current=current,
                prev_intensity=result.intensity,  # simplified
                curr_intensity=result.intensity,
                confidence=result.confidence.value,
                metrics=self._get_key_metrics(),
            )
            sent = self.discord.send_embed(embed, 'regime_change')
            self._previous_regime = current
            return sent

        self._previous_regime = current
        return False

    def check_stress_breach(self, collector: StressIndicatorCollector) -> bool:
        """Check and alert on stress threshold breach."""
        data = collector.collect_sync()

        thresholds = {
            'sofr_ois_spread': (10, 25),  # elevated, critical (bps)
            'repo_stress': (0.5, 0.8),
        }

        for indicator, (elevated, critical) in thresholds.items():
            value = data.get(indicator, 0)
            if value >= critical:
                embed = self.formatter.stress_breach(
                    indicator=indicator,
                    value=value,
                    threshold=critical,
                    severity='critical',
                )
                return self.discord.send_embed(embed, f'stress_{indicator}')
            elif value >= elevated:
                embed = self.formatter.stress_breach(
                    indicator=indicator,
                    value=value,
                    threshold=elevated,
                    severity='elevated',
                )
                return self.discord.send_embed(embed, f'stress_{indicator}')
        return False

    def check_dxy_move(self, collector: FXCollector) -> bool:
        """Check and alert on >1% DXY move."""
        data = collector.collect_sync()
        change_pct = data.get('dxy_change_1d', 0)

        if abs(change_pct) >= 1.0:
            direction = 'up' if change_pct > 0 else 'down'
            embed = self.formatter.dxy_move(
                current=data.get('dxy', 0),
                change_pct=change_pct,
                direction=direction,
            )
            return self.discord.send_embed(embed, 'dxy_move')
        return False

    def check_correlation_shift(self, engine: CorrelationEngine) -> bool:
        """Check and alert on >0.3 correlation change."""
        current_corrs = engine.get_correlations()

        for asset, corr in current_corrs.items():
            prev = self._previous_correlations.get(asset)
            if prev is not None:
                change = abs(corr - prev)
                if change >= 0.3:
                    embed = self.formatter.correlation_shift(
                        asset=asset,
                        previous=prev,
                        current=corr,
                        change=corr - prev,
                    )
                    sent = self.discord.send_embed(embed, f'corr_{asset}')
                    self._previous_correlations[asset] = corr
                    return sent
            self._previous_correlations[asset] = corr
        return False
```

### 5. Alert Scheduler (scheduler.py)
```python
import asyncio
from datetime import datetime

class AlertScheduler:
    """Run alert checks on schedule."""

    def __init__(
        self,
        handlers: AlertHandlers,
        check_interval_seconds: int = 300,  # 5 minutes
    ):
        self.handlers = handlers
        self.interval = check_interval_seconds
        self._running = False

    async def start(self):
        """Start the alert check loop."""
        self._running = True
        while self._running:
            await self._run_checks()
            await asyncio.sleep(self.interval)

    async def _run_checks(self):
        """Run all alert checks."""
        # These would fetch data and check thresholds
        self.handlers.check_regime_change(...)
        self.handlers.check_stress_breach(...)
        self.handlers.check_dxy_move(...)
        self.handlers.check_correlation_shift(...)

    def stop(self):
        self._running = False
```

```python
from pydantic import BaseModel

class AlertConfig(BaseModel):
    enabled: bool = True
    discord_webhook_url: str = ""

    # Thresholds
    dxy_move_threshold_pct: float = 1.0
    correlation_shift_threshold: float = 0.3

    # Stress thresholds (bps)

    rate_limit_stress: int = 300
    rate_limit_dxy: int = 3600
    rate_limit_correlation: int = 3600

    # Check interval
    check_interval_seconds: int = 300

def load_alert_config() -> AlertConfig:
    return AlertConfig(
        discord_webhook_url=os.getenv('LIQUIDITY_DISCORD_WEBHOOK_URL', ''),
```
