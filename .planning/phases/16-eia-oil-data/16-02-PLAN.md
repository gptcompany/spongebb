### 2. Create Cushing-Specific Methods

```python
async def collect_cushing(self, lookback_weeks: int = 52) -> pd.DataFrame:
    """Collect Cushing inventory data."""
    return await self.collect(
        series_ids=["cushing_inventory"],
        lookback_weeks=lookback_weeks
    )

def calculate_cushing_utilization(self, df: pd.DataFrame) -> pd.DataFrame:
    """Calculate storage utilization vs capacity (70.8 million barrels)."""
    CUSHING_CAPACITY = 70_800  # thousand barrels
    df["utilization_pct"] = (df["value"] / CUSHING_CAPACITY) * 100
    return df
```

### 3. Add Historical Percentiles

```python
def calculate_cushing_percentile(self, df: pd.DataFrame) -> pd.DataFrame:
    """Calculate where current level sits vs 52-week range."""
    df["percentile_52w"] = df["value"].rolling(52).apply(
        lambda x: (x.iloc[-1] - x.min()) / (x.max() - x.min()) * 100
    )
    return df
```
