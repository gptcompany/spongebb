### 2. Server Setup (server.py)
```python
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware

app = FastAPI(
    title="Global Liquidity Monitor API",
    version="1.0.0",
    description="Real-time liquidity regime classification"
)

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],  # Internal use
    allow_methods=["GET"],
    allow_headers=["*"],
)
```

### 3. Response Schemas (schemas.py)
```python
from pydantic import BaseModel
from datetime import datetime

class APIMetadata(BaseModel):
    timestamp: datetime
    source: str = "openbb_liquidity"
    version: str = "1.0.0"

class NetLiquidityResponse(BaseModel):
    value: float
    walcl: float
    tga: float
    rrp: float
    as_of_date: datetime
    metadata: APIMetadata

class GlobalLiquidityResponse(BaseModel):
    value: float
    components: dict[str, float]  # fed, ecb, boj, pboc
    as_of_date: datetime
    metadata: APIMetadata

class RegimeResponse(BaseModel):
    regime: str  # EXPANSION or CONTRACTION
    intensity: int  # 0-100
    confidence: str  # HIGH, MEDIUM, LOW
    as_of_date: datetime
    metadata: APIMetadata

class StealthQEResponse(BaseModel):
    score: float  # -100 to +100
    components: dict[str, float]
    interpretation: str  # "Strong Easing", "Neutral", "Tightening"
    as_of_date: datetime
    metadata: APIMetadata
```
