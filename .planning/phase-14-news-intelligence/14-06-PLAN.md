### Fallback Strategy

```python
async def get_statement(self, date: date) -> FOMCStatement:
    """Fetch statement with 3-tier fallback."""

    # Check local cache first (30 day TTL)
    cached = self._cache.get(date)
    if cached:
        return cached

    # Tier 1: Fed official website
    try:
        text = await self._scrape_fed_website(date)
        if text:
            return self._save_and_return(date, text, source="fed")
    except Exception as e:
        logger.warning(f"Fed website failed for {date}: {e}")

    # Tier 2: FedTools library
    try:
        text = await self._fetch_fedtools(date)
        if text:
            return self._save_and_return(date, text, source="fedtools")
    except Exception as e:
        logger.warning(f"FedTools failed for {date}: {e}")

    # Tier 3: GitHub archive
    try:
        text = await self._fetch_github_archive(date)
        if text:
            return self._save_and_return(date, text, source="github")
    except Exception as e:
        logger.warning(f"GitHub archive failed for {date}: {e}")

    raise StatementNotFoundError(f"All sources failed for {date}")
```

### Local Cache Strategy

```python
class StatementCache:
    """Local file cache for FOMC statements."""

    def __init__(self, cache_dir: Path = Path(".cache/fomc")):
        self._cache_dir = cache_dir
        self._cache_dir.mkdir(parents=True, exist_ok=True)
        self._ttl_days = 30  # Re-fetch after 30 days

    def get(self, date: date) -> FOMCStatement | None:
        path = self._cache_dir / f"{date.isoformat()}.json"
        if path.exists():
            data = json.loads(path.read_text())
            cached_at = datetime.fromisoformat(data["cached_at"])
            if datetime.utcnow() - cached_at < timedelta(days=self._ttl_days):
                return FOMCStatement(**data)
        return None

    def put(self, statement: FOMCStatement) -> None:
        path = self._cache_dir / f"{statement.date.isoformat()}.json"
        data = statement.model_dump()
        data["cached_at"] = datetime.utcnow().isoformat()
        path.write_text(json.dumps(data))
```

### Key Decisions

- **3-tier fallback**: Fed website → FedTools → GitHub archive
- **Local cache**: 30-day TTL, JSON files in .cache/fomc/
- **FedTools**: Python library wraps Fed API, adds reliability
- **Historical**: GitHub archives have 1996-present
- **Parsing**: BeautifulSoup for HTML → plain text
- **Format**: Clean text (no HTML tags, normalized whitespace)
