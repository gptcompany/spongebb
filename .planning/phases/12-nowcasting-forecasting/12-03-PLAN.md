# Plan 12-03: Regime Forecaster (Ensemble)

## Overview

Upgrade the regime classifier from binary threshold-based to a probabilistic ensemble combining HMM, Markov Switching, and LSTM for 3-state regime detection with 7-30 day forecasting capability.

## Goals

- Upgrade from 2 states (EXPANSION/CONTRACTION) to 3 states (EXPANSION/NEUTRAL/CONTRACTION)
- Provide smoothed regime probabilities (not just hard classification)
- Forecast regime 7/14/30 days ahead
- Ensemble accuracy > 75% on held-out test set
- Regime flip rate < 15% (avoid spurious changes)

## Approach

Build an ensemble of three complementary models:
1. **HMM (40% weight)**: Best for interpretability and regime persistence
2. **Markov Switching (30% weight)**: Parameter switching for enhanced accuracy
3. **LSTM (30% weight)**: Non-linear dependencies and forecasting

### Ensemble Weighting

```python
P(regime) = 0.4 * P_hmm + 0.3 * P_markov + 0.3 * P_lstm
```

### Task 3: Implement HMM Classifier
**File**: `src/liquidity/nowcasting/regime/hmm_classifier.py`

```python
from hmmlearn import hmm
import numpy as np
import pandas as pd
from dataclasses import dataclass
from typing import List, Tuple
from enum import IntEnum

class RegimeState(IntEnum):
    """3-state regime classification."""
    EXPANSION = 0
    NEUTRAL = 1
    CONTRACTION = 2

@dataclass
class RegimeProbabilities:
    """Regime probabilities at a point in time."""
    timestamp: pd.Timestamp
    expansion: float      # P(Expansion)
    neutral: float        # P(Neutral)
    contraction: float    # P(Contraction)
    most_likely: RegimeState
    confidence: float     # Max probability

    @property
    def as_array(self) -> np.ndarray:
        return np.array([self.expansion, self.neutral, self.contraction])

class HMMRegimeClassifier:
    """
    Hidden Markov Model for regime detection.

    3-state Gaussian HMM with:
    - States: EXPANSION, NEUTRAL, CONTRACTION
    - Emissions: Gaussian distributions on liquidity features
    - Transition matrix: High diagonal for regime persistence
    """

    def __init__(self, n_states: int = 3, n_iter: int = 100, random_state: int = 42):
        self.n_states = n_states
        self.model = hmm.GaussianHMM(
            n_components=n_states,
            covariance_type="full",
            n_iter=n_iter,
            random_state=random_state
        )
        self._is_fitted = False
        self._state_mapping = {}  # Maps learned states to RegimeState

    def fit(self, features: pd.DataFrame) -> "HMMRegimeClassifier":
        """
        Fit HMM on historical features.

        Args:
            features: DataFrame with columns [net_liq_return, global_liq_return, stealth_qe]
        """
        X = features.values
        self.model.fit(X)
        self._is_fitted = True

        # Map learned states to regime labels based on mean returns
        self._map_states_to_regimes(features)

        return self

    def predict_proba(self, features: pd.DataFrame) -> List[RegimeProbabilities]:
        """
        Get smoothed regime probabilities for each time step.

        Uses forward-backward algorithm for full-sample smoothing.
        """
        if not self._is_fitted:
            raise ValueError("Model not fitted. Call fit() first.")

        X = features.values
        posteriors = self.model.predict_proba(X)  # (n_samples, n_states)

        results = []
        for i, (ts, row) in enumerate(features.iterrows()):
            probs = posteriors[i]
            # Reorder according to state mapping
            ordered_probs = self._reorder_probs(probs)

            results.append(RegimeProbabilities(
                timestamp=ts,
                expansion=ordered_probs[0],
                neutral=ordered_probs[1],
                contraction=ordered_probs[2],
                most_likely=RegimeState(np.argmax(ordered_probs)),
                confidence=np.max(ordered_probs)
            ))

        return results

    def get_transition_matrix(self) -> pd.DataFrame:
        """
        Get transition probability matrix.

        High diagonal values indicate regime persistence.
        """
        transmat = self.model.transmat_
        labels = ['EXPANSION', 'NEUTRAL', 'CONTRACTION']
        return pd.DataFrame(transmat, index=labels, columns=labels)

    def _map_states_to_regimes(self, features: pd.DataFrame) -> None:
        """
        Map HMM states to regime labels based on mean feature values.

        EXPANSION: Highest net_liq_return
        CONTRACTION: Lowest net_liq_return
        NEUTRAL: Middle
        """
        states = self.model.predict(features.values)

        # Calculate mean of first feature (net_liq_return) per state
        means = []
        for state in range(self.n_states):
            mask = states == state
            mean_return = features.iloc[:, 0][mask].mean()
            means.append((state, mean_return))

        # Sort by mean return
        sorted_states = sorted(means, key=lambda x: x[1], reverse=True)

        # Map: highest = EXPANSION, lowest = CONTRACTION
        self._state_mapping = {
            sorted_states[0][0]: RegimeState.EXPANSION,
            sorted_states[1][0]: RegimeState.NEUTRAL,
            sorted_states[2][0]: RegimeState.CONTRACTION
        }

    def _reorder_probs(self, probs: np.ndarray) -> np.ndarray:
        """Reorder probabilities to match [EXPANSION, NEUTRAL, CONTRACTION]."""
        ordered = np.zeros(3)
        for hmm_state, regime in self._state_mapping.items():
            ordered[regime] = probs[hmm_state]
        return ordered
```

```python
from statsmodels.tsa.regime_switching.markov_regression import MarkovRegression
import numpy as np
import pandas as pd
from typing import List
from .hmm_classifier import RegimeProbabilities, RegimeState

class MarkovSwitchingClassifier:
    """
    Markov Switching Regression for regime detection.

    Parameters (mean, variance, AR coefficients) switch between regimes.
    More flexible than HMM but more computationally expensive.
    """

    def __init__(self, k_regimes: int = 3, order: int = 1):
        self.k_regimes = k_regimes
        self.order = order  # AR order
        self.model = None
        self.results = None
        self._state_mapping = {}

    def fit(self, endog: pd.Series, exog: pd.DataFrame = None) -> "MarkovSwitchingClassifier":
        """
        Fit Markov Switching model via MLE.

        Args:
            endog: Target variable (e.g., Net Liquidity returns)
            exog: Optional exogenous variables
        """
        self.model = MarkovRegression(
            endog=endog,
            k_regimes=self.k_regimes,
            order=self.order,
            switching_variance=True,
            exog=exog
        )
        self.results = self.model.fit(disp=False)

        # Map states to regimes based on regime-specific means
        self._map_states_to_regimes()

        return self

    def predict_proba(self, filtered: bool = False) -> List[RegimeProbabilities]:
        """
        Get regime probabilities.

        Args:
            filtered: If True, use filtered probs (real-time).
                     If False, use smoothed probs (full-sample).
        """
        if self.results is None:
            raise ValueError("Model not fitted. Call fit() first.")

        if filtered:
            probs = self.results.filtered_regime_probabilities
        else:
            probs = self.results.smoothed_regime_probabilities

        results = []
        for i, ts in enumerate(probs.index):
            p = probs.iloc[i].values
            ordered = self._reorder_probs(p)

            results.append(RegimeProbabilities(
                timestamp=ts,
                expansion=ordered[0],
                neutral=ordered[1],
                contraction=ordered[2],
                most_likely=RegimeState(np.argmax(ordered)),
                confidence=np.max(ordered)
            ))

        return results

    def _map_states_to_regimes(self) -> None:
        """Map states based on regime-specific means from params."""
        # Extract regime-specific means from parameters
        params = self.results.params
        # Regime means are typically first k_regimes parameters
        regime_means = params[:self.k_regimes]

        sorted_idx = np.argsort(regime_means)[::-1]  # Descending
        self._state_mapping = {
            sorted_idx[0]: RegimeState.EXPANSION,
            sorted_idx[1]: RegimeState.NEUTRAL,
            sorted_idx[2]: RegimeState.CONTRACTION
        }

    def _reorder_probs(self, probs: np.ndarray) -> np.ndarray:
        """Reorder to [EXPANSION, NEUTRAL, CONTRACTION]."""
        ordered = np.zeros(3)
        for ms_state, regime in self._state_mapping.items():
            ordered[regime] = probs[ms_state]
        return ordered
```

### Task 5: Implement LSTM Forecaster
**File**: `src/liquidity/nowcasting/regime/lstm_forecaster.py`

```python
import torch
import torch.nn as nn
import numpy as np
import pandas as pd
from typing import List, Tuple
from dataclasses import dataclass
from .hmm_classifier import RegimeState

@dataclass
class RegimeForecast:
    """Regime forecast for future dates."""
    horizon: int              # Days ahead
    probabilities: np.ndarray # (3,) for [EXP, NEU, CON]
    most_likely: RegimeState
    confidence: float

class LSTMRegimeClassifier(nn.Module):
    """
    LSTM neural network for regime classification and forecasting.

    Architecture:
    - Input: 30-day sequence of [net_liq, global_liq, stealth_qe]
    - LSTM layers with dropout
    - Output: 3-class probabilities
    """

    def __init__(
        self,
        input_size: int = 3,
        hidden_size: int = 64,
        num_layers: int = 2,
        num_classes: int = 3,
        dropout: float = 0.2
    ):
        super().__init__()

        self.lstm = nn.LSTM(
            input_size=input_size,
            hidden_size=hidden_size,
            num_layers=num_layers,
            dropout=dropout if num_layers > 1 else 0,
            batch_first=True
        )

        self.fc = nn.Sequential(
            nn.Linear(hidden_size, 32),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(32, num_classes)
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Forward pass.

        Args:
            x: (batch_size, seq_length, input_size)

        Returns:
            logits: (batch_size, num_classes)
        """
        lstm_out, (h_n, c_n) = self.lstm(x)
        last_hidden = h_n[-1]  # (batch_size, hidden_size)
        logits = self.fc(last_hidden)
        return logits


class LSTMRegimeForecaster:
    """
    Wrapper for LSTM regime forecasting.

    Handles data preparation, training, and inference.
    """

    def __init__(self, seq_length: int = 30, device: str = "cpu"):
        self.seq_length = seq_length
        self.device = torch.device(device)
        self.model = LSTMRegimeClassifier().to(self.device)
        self.scaler = None
        self._is_fitted = False

    def fit(
        self,
        features: pd.DataFrame,
        labels: pd.Series,
        epochs: int = 50,
        batch_size: int = 32,
        lr: float = 0.001
    ) -> "LSTMRegimeForecaster":
        """
        Train LSTM on historical data.

        Args:
            features: DataFrame with [net_liq, global_liq, stealth_qe]
            labels: Series with RegimeState labels (0, 1, 2)
        """
        # Prepare sequences
        X, y = self._create_sequences(features, labels)

        # Convert to tensors
        X_tensor = torch.FloatTensor(X).to(self.device)
        y_tensor = torch.LongTensor(y).to(self.device)

        # Training
        optimizer = torch.optim.Adam(self.model.parameters(), lr=lr)
        loss_fn = nn.CrossEntropyLoss()

        dataset = torch.utils.data.TensorDataset(X_tensor, y_tensor)
        loader = torch.utils.data.DataLoader(dataset, batch_size=batch_size, shuffle=True)

        self.model.train()
        for epoch in range(epochs):
            total_loss = 0
            for X_batch, y_batch in loader:
                optimizer.zero_grad()
                logits = self.model(X_batch)
                loss = loss_fn(logits, y_batch)
                loss.backward()
                optimizer.step()
                total_loss += loss.item()

        self._is_fitted = True
        return self

    def predict_proba(self, features: pd.DataFrame) -> np.ndarray:
        """
        Get current regime probabilities.

        Returns:
            probs: (3,) array of [P(EXP), P(NEU), P(CON)]
        """
        if not self._is_fitted:
            raise ValueError("Model not fitted. Call fit() first.")

        # Use last seq_length observations
        X = features.iloc[-self.seq_length:].values
        X_tensor = torch.FloatTensor(X).unsqueeze(0).to(self.device)

        self.model.eval()
        with torch.no_grad():
            logits = self.model(X_tensor)
            probs = torch.softmax(logits, dim=1).cpu().numpy()[0]

        return probs

    def forecast(self, features: pd.DataFrame, horizons: List[int] = [7, 14, 30]) -> List[RegimeForecast]:
        """
        Forecast regime for future horizons.

        Note: Simple approach - uses current prediction as forecast.
        More sophisticated: train separate models for each horizon.
        """
        current_probs = self.predict_proba(features)

        forecasts = []
        for h in horizons:
            # Apply decay to confidence for longer horizons
            decay = np.exp(-h / 30)
            adjusted_probs = current_probs * decay + (1 - decay) / 3
            adjusted_probs /= adjusted_probs.sum()

            forecasts.append(RegimeForecast(
                horizon=h,
                probabilities=adjusted_probs,
                most_likely=RegimeState(np.argmax(adjusted_probs)),
                confidence=np.max(adjusted_probs)
            ))

        return forecasts

    def _create_sequences(
        self,
        features: pd.DataFrame,
        labels: pd.Series
    ) -> Tuple[np.ndarray, np.ndarray]:
        """Create sequence samples for LSTM training."""
        X, y = [], []
        values = features.values

        for i in range(len(values) - self.seq_length):
            X.append(values[i:i + self.seq_length])
            y.append(labels.iloc[i + self.seq_length])

        return np.array(X), np.array(y)
```

```python
import numpy as np
import pandas as pd
from typing import List, Dict, Optional
from dataclasses import dataclass
from .hmm_classifier import HMMRegimeClassifier, RegimeProbabilities, RegimeState
from .markov_switching import MarkovSwitchingClassifier
from .lstm_forecaster import LSTMRegimeForecaster, RegimeForecast

@dataclass
class EnsembleResult:
    """Ensemble regime classification result."""
    timestamp: pd.Timestamp
    probabilities: np.ndarray    # [P(EXP), P(NEU), P(CON)]
    most_likely: RegimeState
    confidence: float
    component_probs: Dict[str, np.ndarray]  # Per-model breakdown
    forecasts: Optional[List[RegimeForecast]] = None

class RegimeEnsemble:
    """
    Ensemble of HMM, Markov Switching, and LSTM for regime detection.

    Weights:
    - HMM: 40% (interpretability, persistence)
    - Markov Switching: 30% (parameter switching)
    - LSTM: 30% (non-linearity, forecasting)
    """

    def __init__(
        self,
        weights: Dict[str, float] = None,
        hmm_params: Dict = None,
        ms_params: Dict = None,
        lstm_params: Dict = None
    ):
        self.weights = weights or {'hmm': 0.4, 'markov': 0.3, 'lstm': 0.3}

        self.hmm = HMMRegimeClassifier(**(hmm_params or {}))
        self.markov = MarkovSwitchingClassifier(**(ms_params or {}))
        self.lstm = LSTMRegimeForecaster(**(lstm_params or {}))

        self._is_fitted = False

    def fit(
        self,
        features: pd.DataFrame,
        labels: pd.Series = None,
        endog_col: str = 'net_liq_return'
    ) -> "RegimeEnsemble":
        """
        Fit all ensemble components.

        Args:
            features: DataFrame with liquidity features
            labels: Optional regime labels for LSTM (will generate if None)
            endog_col: Column to use as target for Markov Switching
        """
        # Fit HMM (unsupervised)
        self.hmm.fit(features)

        # Fit Markov Switching
        self.markov.fit(features[endog_col])

        # Generate labels from HMM if not provided
        if labels is None:
            hmm_results = self.hmm.predict_proba(features)
            labels = pd.Series([r.most_likely for r in hmm_results], index=features.index)

        # Fit LSTM (supervised)
        self.lstm.fit(features, labels)

        self._is_fitted = True
        return self

    def predict(self, features: pd.DataFrame, forecast_horizons: List[int] = [7, 14, 30]) -> EnsembleResult:
        """
        Get ensemble regime classification and forecast.

        Returns weighted average of component probabilities.
        """
        if not self._is_fitted:
            raise ValueError("Ensemble not fitted. Call fit() first.")

        # Get probabilities from each component
        hmm_results = self.hmm.predict_proba(features)
        hmm_probs = hmm_results[-1].as_array

        markov_results = self.markov.predict_proba(filtered=False)
        markov_probs = markov_results[-1].as_array if markov_results else np.array([0.33, 0.34, 0.33])

        lstm_probs = self.lstm.predict_proba(features)

        # Weighted ensemble
        ensemble_probs = (
            self.weights['hmm'] * hmm_probs +
            self.weights['markov'] * markov_probs +
            self.weights['lstm'] * lstm_probs
        )
        ensemble_probs /= ensemble_probs.sum()  # Normalize

        # Get forecasts from LSTM
        forecasts = self.lstm.forecast(features, forecast_horizons)

        return EnsembleResult(
            timestamp=features.index[-1],
            probabilities=ensemble_probs,
            most_likely=RegimeState(np.argmax(ensemble_probs)),
            confidence=np.max(ensemble_probs),
            component_probs={
                'hmm': hmm_probs,
                'markov': markov_probs,
                'lstm': lstm_probs
            },
            forecasts=forecasts
        )

    def get_transition_matrix(self) -> pd.DataFrame:
        """Get HMM transition matrix for regime persistence analysis."""
        return self.hmm.get_transition_matrix()
```
