### 1. HTML Export Module (dashboard/export.py)
```python
from pathlib import Path
from datetime import datetime
import plotly.io as pio

class HTMLExporter:
    """Export dashboard to standalone HTML file."""

    def __init__(self, output_dir: Path | None = None):
        self.output_dir = output_dir or Path('exports')
        self.output_dir.mkdir(exist_ok=True)

    def export_dashboard(
        self,
        figures: dict[str, go.Figure],
        title: str = "Global Liquidity Monitor",
    ) -> Path:
        """Export all figures to a single HTML file."""
        timestamp = datetime.utcnow().strftime('%Y%m%d_%H%M%S')
        output_path = self.output_dir / f'liquidity_dashboard_{timestamp}.html'

        html_parts = [self._create_header(title)]

        for name, fig in figures.items():
            html_parts.append(f'<h2>{name}</h2>')
            html_parts.append(pio.to_html(fig, full_html=False, include_plotlyjs=False))

        html_parts.append(self._create_footer())

        full_html = self._wrap_html(html_parts, title)

        with open(output_path, 'w') as f:
            f.write(full_html)

        return output_path

    def _create_header(self, title: str) -> str:
        return f'''
        <div class="header">
            <h1>{title}</h1>
            <p>Generated: {datetime.utcnow().strftime('%Y-%m-%d %H:%M UTC')}</p>
        </div>
        '''

    def _create_footer(self) -> str:
        return '''
        <div class="footer">
            <p>Global Liquidity Monitor - OpenBB SDK</p>
        </div>
        '''

    def _wrap_html(self, parts: list[str], title: str) -> str:
        return f'''
        <!DOCTYPE html>
        <html>
        <head>
            <title>{title}</title>
            <script src="https://cdn.plot.ly/plotly-latest.min.js"></script>
            <style>
                body {{
                    font-family: -apple-system, BlinkMacSystemFont, 'Segoe UI', Roboto, sans-serif;
                    background: #1a1a2e;
                    color: #eee;
                    margin: 0;
                    padding: 20px;
                }}
                .header {{
                    text-align: center;
                    margin-bottom: 30px;
                }}
                h1 {{ color: #00ff88; }}
                h2 {{ color: #aaa; border-bottom: 1px solid #333; padding-bottom: 10px; }}
                .footer {{
                    text-align: center;
                    margin-top: 30px;
                    padding-top: 20px;
                    border-top: 1px solid #333;
                    color: #666;
                }}
            </style>
        </head>
        <body>
            {''.join(parts)}
        </body>
        </html>
        '''

    def export_single_figure(self, fig: go.Figure, name: str) -> Path:
        """Export a single figure to HTML."""
        timestamp = datetime.utcnow().strftime('%Y%m%d_%H%M%S')
        output_path = self.output_dir / f'{name}_{timestamp}.html'

        pio.write_html(fig, output_path, include_plotlyjs=True)
        return output_path
```

```python
import dash_bootstrap_components as dbc
from dash import html, dcc

def create_quality_status_bar(quality_report: QualityReport) -> dbc.Row:
    """Create status bar with quality indicators."""
    # Overall score color
    if quality_report.overall_score >= 90:
        score_color = 'success'
    elif quality_report.overall_score >= 70:
        score_color = 'warning'
    else:
        score_color = 'danger'

    return dbc.Row([
        dbc.Col([
            dbc.Badge(
                f"Quality: {quality_report.overall_score:.0f}%",
                color=score_color,
                className='me-2',
            ),
        ], width='auto'),
        dbc.Col([
            html.Span(
                f"Last Update: {format_relative_time(quality_report.timestamp)}",
                className='text-muted',
            ),
        ], width='auto'),
        dbc.Col([
            create_freshness_indicators(quality_report.stale_sources),
        ], width='auto'),
    ], className='mb-3 align-items-center')

def create_freshness_indicators(stale_sources: list[str]) -> html.Div:
    """Show freshness status for each source."""
    if not stale_sources:
        return html.Span("All data fresh ✓", className='text-success')

    badges = [
        dbc.Badge(source, color='warning', className='me-1')
        for source in stale_sources[:3]
    ]
    if len(stale_sources) > 3:
        badges.append(html.Span(f"+{len(stale_sources)-3} more", className='text-muted'))

    return html.Div([
        html.Span("Stale: ", className='text-warning'),
        *badges,
    ])

def create_quality_detail_panel() -> dbc.Card:
    """Detailed quality metrics panel (collapsible)."""
    return dbc.Card([
        dbc.CardHeader([
            html.A(
                "Data Quality Details",
                id='quality-collapse-toggle',
                style={'cursor': 'pointer'},
            ),
        ]),
        dbc.Collapse([
            dbc.CardBody([
                dbc.Row([
                    dbc.Col([
                        create_quality_gauge('Freshness', 'freshness-gauge'),
                    ], width=4),
                    dbc.Col([
                        create_quality_gauge('Completeness', 'completeness-gauge'),
                    ], width=4),
                    dbc.Col([
                        create_quality_gauge('Validation', 'validation-gauge'),
                    ], width=4),
                ]),
                html.Hr(),
                html.Div(id='source-freshness-table'),
            ]),
        ], id='quality-collapse', is_open=False),
    ])

def create_quality_gauge(label: str, id: str) -> dcc.Graph:
    """Small gauge chart for quality metric."""
    return dcc.Graph(
        id=id,
        config={'displayModeBar': False},
        style={'height': '150px'},
    )

def create_source_freshness_table(last_updates: dict[str, datetime]) -> dbc.Table:
    """Table showing freshness per source."""
    rows = []
    for source, ts in sorted(last_updates.items()):
        age = format_relative_time(ts)
        status = '✓' if (datetime.utcnow() - ts).total_seconds() < 86400 else '⚠'
        rows.append(html.Tr([
            html.Td(source),
            html.Td(age),
            html.Td(status),
        ]))

    return dbc.Table([
        html.Thead(html.Tr([
            html.Th('Source'),
            html.Th('Last Update'),
            html.Th('Status'),
        ])),
        html.Tbody(rows),
    ], striped=True, bordered=True, hover=True, size='sm')
```

### 3. Sanity Bounds on Charts (dashboard/components/bounds.py)
```python
import plotly.graph_objects as go

class SanityBounds:
    """Add historical min/max bounds to charts."""

    # Historical bounds (from analysis)
    BOUNDS = {
        'net_liquidity': (4.5e12, 7.0e12),    # $4.5T - $7T
        'global_liquidity': (25e12, 35e12),    # $25T - $35T
        'dxy': (90, 115),                      # DXY range
        'vix': (10, 80),
        'move': (50, 200),
        'sofr': (0, 10),                       # %
        'gold': (1500, 2500),                  # $/oz
        'copper': (3, 5),                      # $/lb
    }

    @classmethod
    def add_bounds(cls, fig: go.Figure, metric: str) -> go.Figure:
        """Add horizontal bands for historical bounds."""
        if metric not in cls.BOUNDS:
            return fig

        low, high = cls.BOUNDS[metric]

        # Add shaded region for "normal" range
        fig.add_hrect(
            y0=low, y1=high,
            fillcolor='rgba(0, 255, 136, 0.1)',
            line_width=0,
            annotation_text='Historical Range',
            annotation_position='top left',
        )

        # Add dashed lines at bounds
        fig.add_hline(
            y=low,
            line_dash='dot',
            line_color='rgba(255, 255, 255, 0.3)',
        )
        fig.add_hline(
            y=high,
            line_dash='dot',
            line_color='rgba(255, 255, 255, 0.3)',
        )

        return fig

    @classmethod
    def is_outside_bounds(cls, metric: str, value: float) -> bool:
        """Check if value is outside historical bounds."""
        if metric not in cls.BOUNDS:
            return False
        low, high = cls.BOUNDS[metric]
        return value < low or value > high

    @classmethod
    def get_bound_status(cls, metric: str, value: float) -> str:
        """Get status relative to bounds."""
        if metric not in cls.BOUNDS:
            return 'unknown'
        low, high = cls.BOUNDS[metric]
        if value < low:
            return 'below_normal'
        elif value > high:
            return 'above_normal'
        return 'normal'
```

### 4. Export Callback (dashboard/callbacks.py)
```python
from dash import callback, Output, Input, State, ctx
from dash.exceptions import PreventUpdate
import json

@callback(
    Output('export-download', 'data'),
    Input('export-btn', 'n_clicks'),
    [State('net-liquidity-chart', 'figure'),
     State('global-liquidity-chart', 'figure'),
     State('regime-gauge', 'figure'),
     State('dxy-chart', 'figure'),
     State('correlation-heatmap', 'figure')],
    prevent_initial_call=True,
)
def export_to_html(n_clicks, *figures):
    """Export dashboard to HTML file."""
    if not n_clicks:
        raise PreventUpdate

    exporter = HTMLExporter()

    # Convert figure dicts back to Figure objects
    figure_dict = {
        'Net Liquidity': go.Figure(figures[0]),
        'Global Liquidity': go.Figure(figures[1]),
        'Regime': go.Figure(figures[2]),
        'DXY': go.Figure(figures[3]),
        'Correlations': go.Figure(figures[4]),
    }

    output_path = exporter.export_dashboard(figure_dict)

    # Return download
    return dcc.send_file(str(output_path))
```

```python
def create_gauge_figure(value: float, label: str) -> go.Figure:
    """Create a gauge figure for quality metric."""
    color = '#00ff88' if value >= 90 else '#ffaa00' if value >= 70 else '#ff4444'

    fig = go.Figure(go.Indicator(
        mode='gauge+number',
        value=value,
        title={'text': label},
        gauge={
            'axis': {'range': [0, 100]},
            'bar': {'color': color},
            'steps': [
                {'range': [0, 70], 'color': 'rgba(255, 68, 68, 0.3)'},
                {'range': [70, 90], 'color': 'rgba(255, 170, 0, 0.3)'},
                {'range': [90, 100], 'color': 'rgba(0, 255, 136, 0.3)'},
            ],
        },
    ))
    fig.update_layout(
        template='plotly_dark',
        margin=dict(l=20, r=20, t=40, b=20),
        height=150,
    )
    return fig
```
