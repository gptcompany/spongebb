### 5. Regime Panel (components/regime.py)
```python
def create_regime_panel():
    """Regime classification with color coding and intensity gauge."""
    return dbc.Card([
        dbc.CardHeader("Liquidity Regime"),
        dbc.CardBody([
            html.Div(id='regime-indicator', className='regime-display'),
            dcc.Graph(id='regime-gauge'),  # Intensity 0-100
            html.Div(id='regime-metrics'),
        ]),
    ])

def create_regime_indicator(regime: str, intensity: int, confidence: str):
    """Color-coded regime display."""
    color = '#00ff88' if regime == 'EXPANSION' else '#ff4444'
    return html.Div([
        html.H2(regime, style={'color': color}),
        html.P(f"Intensity: {intensity}/100"),
        html.P(f"Confidence: {confidence}"),
    ])
```

### 6. Main Layout (layout.py)
```python
from dash import html
import dash_bootstrap_components as dbc
from .components import header, liquidity, regime

def create_layout():
    return html.Div([
        header.create_header(),
        dbc.Container([
            # Status bar
            html.Div(id='status-bar', className='mb-3'),
            # Main panels
            liquidity.create_liquidity_panel(),
            html.Hr(),
            dbc.Row([
                dbc.Col([regime.create_regime_panel()], width=6),
                dbc.Col([html.Div(id='correlation-placeholder')], width=6),
            ]),
        ], fluid=True),
        # Interval for auto-refresh (5 minutes)
        dcc.Interval(id='refresh-interval', interval=5*60*1000, n_intervals=0),
    ])
```
