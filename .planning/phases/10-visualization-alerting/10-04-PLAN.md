- QA-02: System detects missing values and gaps in time series

- QA-05: Unit tests validate Hayes formula against known historical values

### 2. Freshness Detection (freshness.py)
```python
from dataclasses import dataclass
from datetime import datetime, timedelta
from enum import Enum

class FreshnessStatus(Enum):
    FRESH = "fresh"
    STALE = "stale"
    CRITICAL = "critical"

@dataclass
class FreshnessConfig:
    # Source -> max age in hours
    thresholds: dict[str, int] = field(default_factory=lambda: {
        'fed_balance_sheet': 48,    # Weekly update
        'ecb_balance_sheet': 48,
        'boj_balance_sheet': 48,
        'pboc_balance_sheet': 168,  # Monthly update
        'sofr': 24,                 # Daily
        'dxy': 24,
        'vix': 24,
        'move': 24,
        'treasury_yields': 24,
        'tic_data': 720,           # Monthly
    })

class FreshnessChecker:
    def __init__(self, config: FreshnessConfig | None = None):
        self.config = config or FreshnessConfig()

    def check(self, source: str, last_update: datetime) -> FreshnessStatus:
        """Check if data source is fresh, stale, or critical."""
        threshold_hours = self.config.thresholds.get(source, 24)
        age = datetime.utcnow() - last_update

        if age < timedelta(hours=threshold_hours):
            return FreshnessStatus.FRESH
        elif age < timedelta(hours=threshold_hours * 2):
            return FreshnessStatus.STALE
        return FreshnessStatus.CRITICAL

    def check_all(self, last_updates: dict[str, datetime]) -> dict[str, FreshnessStatus]:
        """Check freshness for all sources."""
        return {source: self.check(source, ts) for source, ts in last_updates.items()}

    def get_stale_sources(self, last_updates: dict[str, datetime]) -> list[str]:
        """Get list of stale or critical sources."""
        statuses = self.check_all(last_updates)
        return [s for s, status in statuses.items()
                if status != FreshnessStatus.FRESH]
```

### 3. Gap Detection (completeness.py)
```python
import pandas as pd
import numpy as np

@dataclass
class GapInfo:
    source: str
    start_date: date
    end_date: date
    gap_days: int
    severity: str  # minor (<3 days), major (3-7), critical (>7)

class CompletenessChecker:
    def __init__(self, max_gap_days: int = 3):
        self.max_gap_days = max_gap_days

    def find_gaps(self, df: pd.DataFrame, date_col: str = 'date') -> list[GapInfo]:
        """Find gaps in time series data."""
        if df.empty:
            return []

        df = df.sort_values(date_col)
        dates = pd.to_datetime(df[date_col])

        gaps = []
        for i in range(1, len(dates)):
            delta = (dates.iloc[i] - dates.iloc[i-1]).days
            if delta > 1:  # More than 1 day gap (excluding weekends)
                # Check if it's just a weekend
                if not self._is_weekend_gap(dates.iloc[i-1], dates.iloc[i]):
                    severity = self._classify_gap(delta)
                    gaps.append(GapInfo(
                        source=df.name if hasattr(df, 'name') else 'unknown',
                        start_date=dates.iloc[i-1].date(),
                        end_date=dates.iloc[i].date(),
                        gap_days=delta,
                        severity=severity,
                    ))
        return gaps

    def _is_weekend_gap(self, start: datetime, end: datetime) -> bool:
        """Check if gap is just a weekend."""
        if (end - start).days == 2:
            return start.weekday() == 4  # Friday to Sunday
        if (end - start).days == 3:
            return start.weekday() == 4  # Friday to Monday
        return False

    def _classify_gap(self, days: int) -> str:
        if days <= 3:
            return 'minor'
        elif days <= 7:
            return 'major'
        return 'critical'

    def check_missing_values(self, df: pd.DataFrame) -> dict[str, float]:
        """Check percentage of missing values per column."""
        return (df.isnull().sum() / len(df) * 100).to_dict()

    def completeness_score(self, df: pd.DataFrame) -> float:
        """Calculate overall completeness score (0-100)."""
        missing_pct = df.isnull().sum().sum() / (len(df) * len(df.columns)) * 100
        return max(0, 100 - missing_pct)
```

### 4. Cross-Source Validation (cross_validation.py)
```python
@dataclass
class ValidationResult:
    source_a: str
    source_b: str
    metric: str
    value_a: float
    value_b: float
    difference_pct: float
    status: str  # match, minor_diff, major_diff

class CrossValidator:
    def __init__(self, tolerance_pct: float = 1.0):
        self.tolerance = tolerance_pct

    def validate(
        self,
        source_a: str,
        source_b: str,
        metric: str,
        value_a: float,
        value_b: float,
    ) -> ValidationResult:
        """Compare values from two sources."""
        if value_a == 0 and value_b == 0:
            diff_pct = 0
        elif value_a == 0:
            diff_pct = 100
        else:
            diff_pct = abs(value_a - value_b) / abs(value_a) * 100

        if diff_pct <= self.tolerance:
            status = 'match'
        elif diff_pct <= self.tolerance * 5:
            status = 'minor_diff'
        else:
            status = 'major_diff'

        return ValidationResult(
            source_a=source_a,
            source_b=source_b,
            metric=metric,
            value_a=value_a,
            value_b=value_b,
            difference_pct=diff_pct,
            status=status,
        )

    def validate_fed_data(self, fred_value: float, direct_value: float) -> ValidationResult:
        """Cross-validate Fed balance sheet data."""
        return self.validate('FRED', 'FederalReserve', 'WALCL', fred_value, direct_value)

    def validate_sofr(self, nyfed_value: float, fred_value: float) -> ValidationResult:
        """Cross-validate SOFR rate."""
        return self.validate('NY_Fed', 'FRED', 'SOFR', nyfed_value, fred_value)
```

### 5. Anomaly Detection (anomalies.py)
```python
import numpy as np
from scipy import stats

@dataclass
class Anomaly:
    date: date
    metric: str
    value: float
    z_score: float
    anomaly_type: str  # spike, drop, outlier

class AnomalyDetector:
    def __init__(self, z_threshold: float = 3.0, lookback_days: int = 90):
        self.z_threshold = z_threshold
        self.lookback = lookback_days

    def detect(self, df: pd.DataFrame, value_col: str, date_col: str = 'date') -> list[Anomaly]:
        """Detect anomalies using z-score method."""
        anomalies = []

        for i in range(self.lookback, len(df)):
            window = df[value_col].iloc[i-self.lookback:i]
            current = df[value_col].iloc[i]
            current_date = df[date_col].iloc[i]

            mean = window.mean()
            std = window.std()

            if std == 0:
                continue

            z_score = (current - mean) / std

            if abs(z_score) > self.z_threshold:
                anomaly_type = 'spike' if z_score > 0 else 'drop'
                anomalies.append(Anomaly(
                    date=current_date,
                    metric=value_col,
                    value=current,
                    z_score=z_score,
                    anomaly_type=anomaly_type,
                ))

        return anomalies

    def detect_jumps(self, df: pd.DataFrame, value_col: str, jump_threshold_pct: float = 10) -> list[Anomaly]:
        """Detect sudden jumps in values."""
        anomalies = []
        values = df[value_col]
        dates = df['date']

        for i in range(1, len(values)):
            prev = values.iloc[i-1]
            curr = values.iloc[i]

            if prev == 0:
                continue

            change_pct = abs(curr - prev) / abs(prev) * 100

            if change_pct > jump_threshold_pct:
                anomalies.append(Anomaly(
                    date=dates.iloc[i],
                    metric=value_col,
                    value=curr,
                    z_score=change_pct / jump_threshold_pct,
                    anomaly_type='jump',
                ))

        return anomalies
```

### 7. Quality Score (quality_score.py)
```python
@dataclass
class QualityReport:
    overall_score: float  # 0-100
    freshness_score: float
    completeness_score: float
    validation_score: float
    anomaly_count: int
    stale_sources: list[str]
    critical_issues: list[str]
    timestamp: datetime

class QualityScorer:
    def __init__(
        self,
        freshness_checker: FreshnessChecker,
        completeness_checker: CompletenessChecker,
        cross_validator: CrossValidator,
        anomaly_detector: AnomalyDetector,
    ):
        self.freshness = freshness_checker
        self.completeness = completeness_checker
        self.validator = cross_validator
        self.anomaly = anomaly_detector

    def calculate_score(
        self,
        data: dict[str, pd.DataFrame],
        last_updates: dict[str, datetime],
    ) -> QualityReport:
        """Calculate comprehensive quality score."""
        # Freshness score (0-100)
        freshness_statuses = self.freshness.check_all(last_updates)
        fresh_count = sum(1 for s in freshness_statuses.values()
                        if s == FreshnessStatus.FRESH)
        freshness_score = fresh_count / len(freshness_statuses) * 100

        # Completeness score (0-100)
        completeness_scores = [
            self.completeness.completeness_score(df)
            for df in data.values()
        ]
        completeness_score = np.mean(completeness_scores)

        # Validation score (simplified)
        validation_score = 95.0  # Placeholder

        # Overall score (weighted average)
        overall_score = (
            freshness_score * 0.3 +
            completeness_score * 0.4 +
            validation_score * 0.3
        )

        return QualityReport(
            overall_score=overall_score,
            freshness_score=freshness_score,
            completeness_score=completeness_score,
            validation_score=validation_score,
            anomaly_count=0,  # Computed separately
            stale_sources=self.freshness.get_stale_sources(last_updates),
            critical_issues=[],
            timestamp=datetime.utcnow(),
        )
```
