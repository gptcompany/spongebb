## Objective

Implementare CVaR (Conditional Value at Risk) / Expected Shortfall come misura di tail risk superiore a VaR.

## Background

CVaR risponde alla domanda: "Qual è la perdita media quando superiamo il VaR?"

```
CVaR(α) = E[Loss | Loss > VaR(α)]
```

Vantaggi su VaR:
- Coerente (sub-additivo)
- Considera severità delle perdite, non solo frequenza
- Basel III / FRTB lo preferisce a VaR

```python
"""CVaR / Expected Shortfall calculator."""
from dataclasses import dataclass
from typing import Optional
import numpy as np
import pandas as pd
from scipy import stats

from .var.historical import HistoricalVaR
from .var.parametric import ParametricVaR, Distribution


@dataclass
class CVaRResult:
    """CVaR calculation result."""
    cvar_95: float          # CVaR at 95% confidence
    cvar_99: float          # CVaR at 99% confidence
    var_95: float           # Corresponding VaR
    var_99: float           # Corresponding VaR
    tail_observations: int  # Count of tail losses
    method: str            # 'historical' or 'parametric'
    as_of_date: Optional[pd.Timestamp] = None


class ExpectedShortfall:
    """Expected Shortfall (CVaR) calculator.

    Computes the average loss beyond VaR threshold.
    More conservative than VaR for tail risk management.
    """

    def __init__(self, window: int = 252):
        """Initialize calculator.

        Args:
            window: Rolling window for calculations
        """
        self.window = window

    def calculate_historical(
        self,
        returns: pd.Series,
        as_of: Optional[pd.Timestamp] = None
    ) -> CVaRResult:
        """Calculate historical CVaR (empirical).

        Args:
            returns: Series of returns
            as_of: Optional point-in-time date

        Returns:
            CVaRResult with historical CVaR
        """
        if as_of is not None:
            returns = returns[returns.index <= as_of]

        if len(returns) > self.window:
            returns = returns.iloc[-self.window:]

        clean_returns = returns.dropna()

        # VaR thresholds (percentiles of LOSSES)
        var_95_threshold = np.percentile(clean_returns, 5)  # 5th percentile
        var_99_threshold = np.percentile(clean_returns, 1)  # 1st percentile

        # CVaR = mean of returns below VaR threshold
        tail_95 = clean_returns[clean_returns <= var_95_threshold]
        tail_99 = clean_returns[clean_returns <= var_99_threshold]

        cvar_95 = -tail_95.mean() if len(tail_95) > 0 else -var_95_threshold
        cvar_99 = -tail_99.mean() if len(tail_99) > 0 else -var_99_threshold

        return CVaRResult(
            cvar_95=cvar_95,
            cvar_99=cvar_99,
            var_95=-var_95_threshold,
            var_99=-var_99_threshold,
            tail_observations=len(tail_95),
            method='historical',
            as_of_date=returns.index[-1] if len(returns) > 0 else as_of
        )

    def calculate_parametric(
        self,
        returns: pd.Series,
        distribution: Distribution = Distribution.NORMAL,
        as_of: Optional[pd.Timestamp] = None
    ) -> CVaRResult:
        """Calculate parametric CVaR.

        For Normal distribution:
        CVaR(α) = μ + σ × φ(z_α) / (1-α)

        Args:
            returns: Series of returns
            distribution: NORMAL or T_STUDENT
            as_of: Optional point-in-time date

        Returns:
            CVaRResult with parametric CVaR
        """
        if as_of is not None:
            returns = returns[returns.index <= as_of]

        if len(returns) > self.window:
            returns = returns.iloc[-self.window:]

        clean_returns = returns.dropna()
        mu = clean_returns.mean()
        sigma = clean_returns.std()

        if distribution == Distribution.NORMAL:
            # Normal CVaR formula
            z_95 = stats.norm.ppf(0.05)
            z_99 = stats.norm.ppf(0.01)

            phi_95 = stats.norm.pdf(z_95)
            phi_99 = stats.norm.pdf(z_99)

            cvar_95 = -(mu + sigma * phi_95 / 0.05)
            cvar_99 = -(mu + sigma * phi_99 / 0.01)

            var_95 = -(mu + sigma * z_95)
            var_99 = -(mu + sigma * z_99)

        else:  # T_STUDENT
            # Fit t-distribution
            df, loc, scale = stats.t.fit(clean_returns)

            t_95 = stats.t.ppf(0.05, df)
            t_99 = stats.t.ppf(0.01, df)

            # t-distribution CVaR formula
            pdf_95 = stats.t.pdf(t_95, df)
            pdf_99 = stats.t.pdf(t_99, df)

            # CVaR for t-distribution
            factor = (df + t_95**2) / (df - 1)
            cvar_95 = -(loc + scale * pdf_95 / 0.05 * factor)

            factor = (df + t_99**2) / (df - 1)
            cvar_99 = -(loc + scale * pdf_99 / 0.01 * factor)

            var_95 = -(loc + scale * t_95)
            var_99 = -(loc + scale * t_99)

        return CVaRResult(
            cvar_95=cvar_95,
            cvar_99=cvar_99,
            var_95=var_95,
            var_99=var_99,
            tail_observations=int(len(clean_returns) * 0.05),
            method=f'parametric-{distribution.value}',
            as_of_date=returns.index[-1] if len(returns) > 0 else as_of
        )

    def calculate_rolling(
        self,
        returns: pd.Series,
        method: str = 'historical'
    ) -> pd.DataFrame:
        """Calculate rolling CVaR time series.

        Args:
            returns: Series of returns
            method: 'historical' or 'parametric'

        Returns:
            DataFrame with cvar_95, cvar_99 columns
        """
        results = []

        for i in range(self.window, len(returns) + 1):
            window_returns = returns.iloc[i - self.window:i]

            if method == 'historical':
                result = self.calculate_historical(window_returns)
            else:
                result = self.calculate_parametric(window_returns)

            results.append({
                'date': returns.index[i - 1],
                'cvar_95': result.cvar_95,
                'cvar_99': result.cvar_99,
                'var_95': result.var_95,
                'var_99': result.var_99
            })

        return pd.DataFrame(results).set_index('date')

    def compare_var_cvar(
        self,
        returns: pd.Series
    ) -> pd.DataFrame:
        """Compare VaR and CVaR across methods.

        Args:
            returns: Series of returns

        Returns:
            DataFrame comparing metrics
        """
        historical = self.calculate_historical(returns)
        param_normal = self.calculate_parametric(returns, Distribution.NORMAL)
        param_t = self.calculate_parametric(returns, Distribution.T_STUDENT)

        comparison = pd.DataFrame({
            'Method': ['Historical', 'Parametric-Normal', 'Parametric-t'],
            'VaR_95': [historical.var_95, param_normal.var_95, param_t.var_95],
            'CVaR_95': [historical.cvar_95, param_normal.cvar_95, param_t.cvar_95],
            'VaR_99': [historical.var_99, param_normal.var_99, param_t.var_99],
            'CVaR_99': [historical.cvar_99, param_normal.cvar_99, param_t.cvar_99],
            'CVaR/VaR_99': [
                historical.cvar_99 / historical.var_99,
                param_normal.cvar_99 / param_normal.var_99,
                param_t.cvar_99 / param_t.var_99
            ]
        })

        return comparison.set_index('Method')
```

```python
"""Tests for CVaR / Expected Shortfall."""
import numpy as np
import pandas as pd
import pytest
from scipy import stats
from liquidity.risk.cvar import ExpectedShortfall, CVaRResult
from liquidity.risk.var.parametric import Distribution


class TestExpectedShortfall:
    """Test Expected Shortfall calculations."""

    @pytest.fixture
    def sample_returns(self) -> pd.Series:
        """Generate sample returns."""
        np.random.seed(42)
        dates = pd.date_range('2020-01-01', periods=500, freq='B')
        return pd.Series(
            np.random.normal(0.0005, 0.015, 500),
            index=dates
        )

    def test_cvar_greater_than_var(self, sample_returns):
        """CVaR should always be >= VaR."""
        es = ExpectedShortfall(window=252)
        result = es.calculate_historical(sample_returns)

        assert result.cvar_95 >= result.var_95
        assert result.cvar_99 >= result.var_99

    def test_cvar_99_greater_than_cvar_95(self, sample_returns):
        """CVaR 99% should be >= CVaR 95%."""
        es = ExpectedShortfall(window=252)
        result = es.calculate_historical(sample_returns)

        assert result.cvar_99 >= result.cvar_95

    def test_parametric_normal(self, sample_returns):
        """Parametric normal CVaR should work."""
        es = ExpectedShortfall(window=252)
        result = es.calculate_parametric(sample_returns, Distribution.NORMAL)

        assert result.cvar_95 > 0
        assert result.method == 'parametric-normal'

    def test_parametric_t(self, sample_returns):
        """Parametric t-dist CVaR should be higher due to fat tails."""
        es = ExpectedShortfall(window=252)
        normal = es.calculate_parametric(sample_returns, Distribution.NORMAL)
        t_dist = es.calculate_parametric(sample_returns, Distribution.T_STUDENT)

        # t-dist typically shows higher tail risk
        assert t_dist.method == 'parametric-t-student'

    def test_rolling_cvar(self, sample_returns):
        """Rolling CVaR should produce time series."""
        es = ExpectedShortfall(window=100)
        rolling = es.calculate_rolling(sample_returns)

        assert len(rolling) > 0
        assert 'cvar_95' in rolling.columns
        assert 'cvar_99' in rolling.columns

        """Comparison table should have all methods."""
        es = ExpectedShortfall(window=252)
        comparison = es.compare_var_cvar(sample_returns)

        assert len(comparison) == 3
        assert 'VaR_95' in comparison.columns
        assert 'CVaR_95' in comparison.columns
        assert 'CVaR/VaR_99' in comparison.columns

    def test_cvar_ratio_reasonable(self, sample_returns):
        """CVaR/VaR ratio should be > 1 and reasonable."""
        es = ExpectedShortfall(window=252)
        comparison = es.compare_var_cvar(sample_returns)

        # CVaR/VaR typically 1.1-1.5 for normal-ish data
        for ratio in comparison['CVaR/VaR_99']:
            assert 1.0 < ratio < 2.0
```
