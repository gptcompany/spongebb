### Low Priority (weekly summary)
- Market sentiment: "bullish", "bearish", "rally", "selloff"

### Task 1: Create keyword matcher module

```python
# src/liquidity/news/oil_alerts.py

from dataclasses import dataclass
from enum import Enum
from typing import List, Dict, Set
import re

class AlertPriority(Enum):
    HIGH = "high"      # Immediate Discord alert
    MEDIUM = "medium"  # Daily digest
    LOW = "low"        # Weekly summary

@dataclass
class KeywordMatch:
    """Match result for a news item."""
    keyword: str
    category: str
    priority: AlertPriority
    context: str  # Surrounding text

# Keyword definitions
SUPPLY_KEYWORDS: Dict[str, Dict] = {
    # High priority - immediate alerts
    "sanctions": {"category": "sanctions", "priority": AlertPriority.HIGH},
    "embargo": {"category": "sanctions", "priority": AlertPriority.HIGH},
    "force majeure": {"category": "outage", "priority": AlertPriority.HIGH},
    "pipeline explosion": {"category": "outage", "priority": AlertPriority.HIGH},
    "refinery fire": {"category": "outage", "priority": AlertPriority.HIGH},
    "production halt": {"category": "outage", "priority": AlertPriority.HIGH},
    "strait of hormuz": {"category": "geopolitical", "priority": AlertPriority.HIGH},
    "blockade": {"category": "geopolitical", "priority": AlertPriority.HIGH},

    # Medium priority - daily digest
    "production cut": {"category": "opec", "priority": AlertPriority.MEDIUM},
    "output reduction": {"category": "opec", "priority": AlertPriority.MEDIUM},
    "quota": {"category": "opec", "priority": AlertPriority.MEDIUM},
    "compliance": {"category": "opec", "priority": AlertPriority.MEDIUM},
    "opec+": {"category": "opec", "priority": AlertPriority.MEDIUM},
    "supply disruption": {"category": "supply", "priority": AlertPriority.MEDIUM},
    "inventory draw": {"category": "supply", "priority": AlertPriority.MEDIUM},

    # Country-specific (medium)
    "iran oil": {"category": "country", "priority": AlertPriority.MEDIUM},
    "russia oil": {"category": "country", "priority": AlertPriority.MEDIUM},
    "libya oil": {"category": "country", "priority": AlertPriority.MEDIUM},
    "venezuela oil": {"category": "country", "priority": AlertPriority.MEDIUM},
    "nigeria oil": {"category": "country", "priority": AlertPriority.MEDIUM},

    # Low priority - weekly summary
    "demand growth": {"category": "demand", "priority": AlertPriority.LOW},
    "demand destruction": {"category": "demand", "priority": AlertPriority.LOW},
}

class SupplyDisruptionMatcher:
    """Matches news items against supply disruption keywords."""

    def __init__(self, keywords: Dict = None):
        self.keywords = keywords or SUPPLY_KEYWORDS
        # Compile regex patterns for efficiency
        self._patterns = {
            kw: re.compile(rf"\b{re.escape(kw)}\b", re.IGNORECASE)
            for kw in self.keywords
        }

    def match(self, text: str) -> List[KeywordMatch]:
        """Find all keyword matches in text."""
        matches = []
        text_lower = text.lower()

        for keyword, config in self.keywords.items():
            pattern = self._patterns[keyword]
            for match in pattern.finditer(text):
                # Extract context (50 chars before/after)
                start = max(0, match.start() - 50)
                end = min(len(text), match.end() + 50)
                context = text[start:end]

                matches.append(KeywordMatch(
                    keyword=keyword,
                    category=config["category"],
                    priority=config["priority"],
                    context=f"...{context}...",
                ))

        return matches

    def get_highest_priority(self, matches: List[KeywordMatch]) -> AlertPriority:
        """Get highest priority from matches."""
        if not matches:
            return AlertPriority.LOW
        priorities = [m.priority for m in matches]
        if AlertPriority.HIGH in priorities:
            return AlertPriority.HIGH
        if AlertPriority.MEDIUM in priorities:
            return AlertPriority.MEDIUM
        return AlertPriority.LOW
```

### Task 2: Integrate with news poller

```python
# In oil_feeds.py, extend OilNewsPoller:

class OilNewsPoller(NewsPoller):
    def __init__(self, ...):
        super().__init__(...)
        self.matcher = SupplyDisruptionMatcher()

    async def poll_with_alerts(self) -> Tuple[List[NewsItem], List[KeywordMatch]]:
        """Poll feeds and check for supply disruption keywords."""
        items = await self.poll_all()

        all_matches = []
        for item in items:
            text = f"{item.title} {item.summary}"
            matches = self.matcher.match(text)
            if matches:
                all_matches.extend(matches)

        return items, all_matches
```

async def send_supply_alert(matches: List[KeywordMatch], item: NewsItem):
    """Send Discord alert for supply disruption."""
    priority = SupplyDisruptionMatcher().get_highest_priority(matches)

    if priority == AlertPriority.HIGH:
        emoji = "🚨"
        color = 0xFF0000  # Red
    elif priority == AlertPriority.MEDIUM:
        emoji = "⚠️"
        color = 0xFFA500  # Orange
    else:
        return  # Don't send low priority immediately

    embed = {
        "title": f"{emoji} Oil Supply Alert",
        "description": item.title,
        "color": color,
        "fields": [
            {"name": "Source", "value": item.source.value, "inline": True},
            {"name": "Keywords", "value": ", ".join(m.keyword for m in matches), "inline": True},
        ],
        "url": str(item.link),
    }

    await send_discord_webhook(embed)

# tests/unit/news/test_oil_alerts.py

def test_keyword_match_sanctions():
    matcher = SupplyDisruptionMatcher()
    text = "US announces new sanctions on Iranian oil exports"
    matches = matcher.match(text)

    assert len(matches) >= 1
    assert any(m.keyword == "sanctions" for m in matches)
    assert any(m.priority == AlertPriority.HIGH for m in matches)

def test_keyword_match_opec():
    matcher = SupplyDisruptionMatcher()
    text = "OPEC+ agrees to production cut of 1 million bpd"
    matches = matcher.match(text)

    assert any(m.keyword == "production cut" for m in matches)
    assert any(m.category == "opec" for m in matches)

def test_no_false_positives():
    matcher = SupplyDisruptionMatcher()
    text = "Oil prices rose 2% on strong demand data"
    matches = matcher.match(text)

    # Should not match high priority keywords
    high_priority = [m for m in matches if m.priority == AlertPriority.HIGH]
    assert len(high_priority) == 0
