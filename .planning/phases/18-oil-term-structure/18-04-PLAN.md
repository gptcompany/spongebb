# src/liquidity/dashboard/components/oil_term_structure.py
"""Oil term structure dashboard component."""

from datetime import datetime, timedelta
from typing import Any

import dash_bootstrap_components as dbc
import pandas as pd
import plotly.graph_objects as go
from dash import dcc, html

from liquidity.analyzers.term_structure import (
    CurveShape,
    TermStructureAnalyzer,
    TermStructureSignal,
    RollYieldMetrics,
)


def create_oil_term_structure_panel(
    signal: TermStructureSignal | None = None,
    roll_yield: RollYieldMetrics | None = None,
    price_history: pd.DataFrame | None = None,
) -> dbc.Card:
    """
    Create oil term structure dashboard panel.

    Args:
        signal: Current term structure signal
        roll_yield: Current roll yield metrics
        price_history: Historical price data for chart

    Returns:
        Dash Bootstrap Card component
    """
    return dbc.Card([
        dbc.CardHeader([
            html.H5("Oil Term Structure", className="mb-0"),
            html.Small("Contango/Backwardation Analysis", className="text-muted"),
        ]),
        dbc.CardBody([
            dbc.Tabs([
                dbc.Tab(
                    _create_signal_tab(signal, roll_yield),
                    label="Signal",
                    tab_id="signal",
                ),
                dbc.Tab(
                    _create_price_tab(price_history),
                    label="Price",
                    tab_id="price",
                ),
                dbc.Tab(
                    _create_roll_yield_tab(roll_yield),
                    label="Roll Yield",
                    tab_id="roll",
                ),
            ], id="term-structure-tabs", active_tab="signal"),
        ]),
    ], className="h-100")


def _create_signal_tab(
    signal: TermStructureSignal | None,
    roll_yield: RollYieldMetrics | None,
) -> html.Div:
    """Create signal gauge and metrics."""
    if signal is None:
        return html.Div("No signal data available", className="text-muted p-3")

    return html.Div([
        # Signal gauge
        dbc.Row([
            dbc.Col([
                dcc.Graph(
                    figure=create_curve_gauge(signal),
                    config={"displayModeBar": False},
                    style={"height": "200px"},
                ),
            ], width=6),
            dbc.Col([
                _create_signal_metrics(signal, roll_yield),
            ], width=6),
        ]),
    ])


def create_curve_gauge(signal: TermStructureSignal) -> go.Figure:
    """
    Create gauge chart for curve shape.

    Left = Backwardation (bullish), Right = Contango (bearish)
    """
    # Map curve shape to gauge value
    shape_map = {
        CurveShape.BACKWARDATION: -signal.intensity,
        CurveShape.FLAT: 0,
        CurveShape.CONTANGO: signal.intensity,
    }
    value = shape_map.get(signal.curve_shape, 0)

    fig = go.Figure(go.Indicator(
        mode="gauge+number",
        value=value,
        title={"text": signal.curve_shape.value},
        gauge={
            "axis": {"range": [-100, 100], "tickwidth": 1},
            "bar": {"color": "darkblue"},
            "steps": [
                {"range": [-100, -50], "color": "darkgreen"},  # Strong backwardation
                {"range": [-50, 0], "color": "lightgreen"},    # Mild backwardation
                {"range": [0, 50], "color": "lightyellow"},    # Mild contango
                {"range": [50, 100], "color": "salmon"},       # Strong contango
            ],
            "threshold": {
                "line": {"color": "red", "width": 4},
                "thickness": 0.75,
                "value": value,
            },
        },
    ))

    fig.update_layout(
        margin=dict(l=20, r=20, t=40, b=20),
        paper_bgcolor="rgba(0,0,0,0)",
        font={"color": "#333"},
    )

    return fig


def _create_signal_metrics(
    signal: TermStructureSignal,
    roll_yield: RollYieldMetrics | None,
) -> html.Div:
    """Create metrics cards."""
    metrics = [
        ("Curve Shape", signal.curve_shape.value, _get_shape_badge(signal.curve_shape)),
        ("Intensity", f"{signal.intensity:.0f}/100", None),
        ("Confidence", f"{signal.confidence * 100:.0f}%", None),
        ("5D Momentum", f"{signal.momentum_5d:+.2f}%", None),
        ("20D Momentum", f"{signal.momentum_20d:+.2f}%", None),
    ]

    if roll_yield:
        metrics.append(("Roll Yield (Ann.)", f"{roll_yield.annual_yield:+.1f}%", None))
        metrics.append(("Regime Days", str(roll_yield.days_in_current_regime), None))

    return html.Div([
        dbc.Row([
            dbc.Col([
                html.Div([
                    html.Small(label, className="text-muted d-block"),
                    html.Span(value, className="fs-5 fw-bold"),
                    badge if badge else None,
                ], className="mb-2")
            ], width=6)
            for label, value, badge in metrics
        ]),
    ])


def _get_shape_badge(shape: CurveShape) -> dbc.Badge:
    """Get colored badge for curve shape."""
    colors = {
        CurveShape.BACKWARDATION: "success",
        CurveShape.FLAT: "secondary",
        CurveShape.CONTANGO: "warning",
    }
    return dbc.Badge(
        shape.value,
        color=colors.get(shape, "secondary"),
        className="ms-2",
    )


def _create_price_tab(price_history: pd.DataFrame | None) -> html.Div:
    """Create price chart with momentum bands."""
    if price_history is None or price_history.empty:
        return html.Div("No price data available", className="text-muted p-3")

    return html.Div([
        dcc.Graph(
            figure=create_price_chart(price_history),
            config={"displayModeBar": True},
            style={"height": "300px"},
        ),
    ])


def create_price_chart(df: pd.DataFrame) -> go.Figure:
    """Create price chart with momentum overlay."""
    # Filter to WTI front
    wti = df[df["series_id"] == "wti_front"].sort_values("timestamp")

    if wti.empty:
        return go.Figure()

    fig = go.Figure()

    # Price line
    fig.add_trace(go.Scatter(
        x=wti["timestamp"],
        y=wti["value"],
        mode="lines",
        name="WTI Front",
        line=dict(color="#1f77b4", width=2),
    ))

    # Add 20-day moving average
    wti_ma = wti["value"].rolling(20).mean()
    fig.add_trace(go.Scatter(
        x=wti["timestamp"],
        y=wti_ma,
        mode="lines",
        name="20D MA",
        line=dict(color="#ff7f0e", width=1, dash="dash"),
    ))

    fig.update_layout(
        title="WTI Crude Oil Price",
        xaxis_title="Date",
        yaxis_title="$/barrel",
        legend=dict(orientation="h", yanchor="bottom", y=1.02),
        margin=dict(l=40, r=20, t=60, b=40),
        hovermode="x unified",
    )

    return fig


def _create_roll_yield_tab(roll_yield: RollYieldMetrics | None) -> html.Div:
    """Create roll yield visualization."""
    if roll_yield is None:
        return html.Div("No roll yield data available", className="text-muted p-3")

    return html.Div([
        dbc.Row([
            dbc.Col([
                dcc.Graph(
                    figure=create_roll_yield_bars(roll_yield),
                    config={"displayModeBar": False},
                    style={"height": "250px"},
                ),
            ], width=8),
            dbc.Col([
                _create_roll_yield_summary(roll_yield),
            ], width=4),
        ]),
    ])


def create_roll_yield_bars(roll_yield: RollYieldMetrics) -> go.Figure:
    """Create bar chart for roll yields at different horizons."""
    horizons = ["1 Month", "3 Month", "12 Month"]
    values = [
        roll_yield.monthly_yield,
        roll_yield.quarterly_yield,
        roll_yield.annual_yield,
    ]

    colors = ["green" if v > 0 else "red" for v in values]

    fig = go.Figure(go.Bar(
        x=horizons,
        y=values,
        marker_color=colors,
        text=[f"{v:+.1f}%" for v in values],
        textposition="outside",
    ))

    fig.update_layout(
        title="Roll Yield by Horizon",
        yaxis_title="Annualized %",
        showlegend=False,
        margin=dict(l=40, r=20, t=60, b=40),
    )

    # Add zero line
    fig.add_hline(y=0, line_dash="dash", line_color="gray")

    return fig


def _create_roll_yield_summary(roll_yield: RollYieldMetrics) -> html.Div:
    """Create roll yield summary text."""
    trend_icons = {
        "IMPROVING": "📈",
        "STABLE": "➡️",
        "DETERIORATING": "📉",
    }

    interpretation = (
        "Positive roll yield (backwardation) - commodity longs benefit"
        if roll_yield.annual_yield > 0
        else "Negative roll yield (contango) - commodity longs pay cost"
    )

    return html.Div([
        html.H6("Summary"),
        html.P([
            html.Strong("Trend: "),
            trend_icons.get(roll_yield.yield_trend, "❓"),
            f" {roll_yield.yield_trend}",
        ]),
        html.P([
            html.Strong("Regime Duration: "),
            f"{roll_yield.days_in_current_regime} days",
        ]),
        html.Hr(),
        html.Small(interpretation, className="text-muted"),
    ], className="p-2")


# Callback helper for live updates
def update_term_structure_panel(
    collector: Any,
    analyzer: TermStructureAnalyzer,
) -> dict:
    """
    Helper to fetch data and update panel.

    Returns dict with signal, roll_yield, and price_history.
    """
    import asyncio

    # Fetch data
    loop = asyncio.new_event_loop()
    try:
        price_data = loop.run_until_complete(collector.collect_with_momentum())
    finally:
        loop.close()

    if price_data.empty:
        return {"signal": None, "roll_yield": None, "price_history": None}

    # Analyze
    signal = analyzer.analyze(price_data)
    roll_yield = analyzer.calculate_roll_yield(price_data)

    return {
        "signal": signal,
        "roll_yield": roll_yield,
        "price_history": price_data,
    }

class TestCurveGauge:
    """Test gauge chart creation."""

    def test_returns_figure(self, sample_signal):
        fig = create_curve_gauge(sample_signal)
        assert isinstance(fig, go.Figure)

    def test_gauge_in_range(self, sample_signal):
        fig = create_curve_gauge(sample_signal)
        # Figure should have indicator trace
        assert len(fig.data) == 1
        assert fig.data[0].type == "indicator"

    @pytest.mark.parametrize("shape,expected_sign", [
        (CurveShape.BACKWARDATION, -1),
        (CurveShape.CONTANGO, 1),
        (CurveShape.FLAT, 0),
    ])
    def test_gauge_direction_by_shape(self, shape, expected_sign):
        signal = TermStructureSignal(
            timestamp=datetime.now(),
            curve_shape=shape,
            intensity=50,
            roll_yield_proxy=0,
            momentum_5d=0,
            momentum_20d=0,
            confidence=0.5,
        )
        fig = create_curve_gauge(signal)
        value = fig.data[0].value

        if expected_sign == 0:
            assert value == 0
        else:
            assert (value > 0) == (expected_sign > 0)
