# Plan 21-01: Supply-Demand Balance Calculator

## Goal

Calcolare il bilancio supply-demand settimanale per il mercato oil USA.

## Formula

```
Weekly Balance = Supply - Demand

Supply Components:
- Domestic Production (WCRFPUS2)
- Crude Imports (WCRIMUS2)
- SPR Releases (when applicable)

Demand Components:
- Refinery Inputs (WGIRSTUS1)
- Crude Exports (WCREXUS2)

Balance Interpretation:
- Positive: Build (surplus, bearish)
- Negative: Draw (deficit, bullish)
```

# src/liquidity/oil/supply_demand.py

from dataclasses import dataclass
from datetime import datetime
from typing import Optional
import pandas as pd

from liquidity.collectors.eia import EIACollector

@dataclass
class SupplyDemandBalance:
    """Weekly oil supply-demand balance."""
    date: datetime
    production: float  # thousand b/d
    imports: float  # thousand b/d
    exports: float  # thousand b/d
    refinery_inputs: float  # thousand b/d
    total_supply: float  # thousand b/d
    total_demand: float  # thousand b/d
    balance: float  # thousand b/d (supply - demand)
    balance_barrels: float  # million barrels/week
    signal: str  # "build", "draw", "flat"

class SupplyDemandCalculator:
    """Calculates weekly oil supply-demand balance."""

    BARRELS_PER_WEEK = 7  # days

    def __init__(self, eia_collector: EIACollector = None):
        self.eia = eia_collector or EIACollector()

    async def calculate_balance(
        self,
        start_date: datetime = None,
        end_date: datetime = None,
    ) -> pd.DataFrame:
        """Calculate weekly supply-demand balance.

        Returns DataFrame with:
        - production, imports, exports, refinery_inputs
        - total_supply, total_demand, balance
        - signal: "build", "draw", "flat"
        """
        # Fetch data from EIA
        production = await self.eia.collect_production(start_date, end_date)
        stocks = await self.eia.collect(start_date, end_date)

        # Merge and calculate
        df = self._merge_data(production, stocks)
        df = self._calculate_balance(df)
        df = self._classify_signal(df)

        return df

    def _merge_data(self, production: pd.DataFrame, stocks: pd.DataFrame) -> pd.DataFrame:
        """Merge production and stocks data."""
        # Implementation
        pass

    def _calculate_balance(self, df: pd.DataFrame) -> pd.DataFrame:
        """Calculate supply, demand, balance."""
        df["total_supply"] = df["production"] + df["imports"]
        df["total_demand"] = df["refinery_inputs"] + df["exports"]
        df["balance"] = df["total_supply"] - df["total_demand"]
        df["balance_barrels"] = df["balance"] * self.BARRELS_PER_WEEK / 1000  # million bbl
        return df

    def _classify_signal(self, df: pd.DataFrame) -> pd.DataFrame:
        """Classify balance signal."""
        def signal(balance: float) -> str:
            if balance > 100:  # >100 kb/d build
                return "build"
            elif balance < -100:  # >100 kb/d draw
                return "draw"
            return "flat"

        df["signal"] = df["balance"].apply(signal)
        return df

    async def get_current_balance(self) -> SupplyDemandBalance:
        """Get latest weekly balance."""
        df = await self.calculate_balance()
        latest = df.iloc[-1]

        return SupplyDemandBalance(
            date=latest.name,
            production=latest["production"],
            imports=latest["imports"],
            exports=latest["exports"],
            refinery_inputs=latest["refinery_inputs"],
            total_supply=latest["total_supply"],
            total_demand=latest["total_demand"],
            balance=latest["balance"],
            balance_barrels=latest["balance_barrels"],
            signal=latest["signal"],
        )

def test_balance_calculation():
    """Test supply-demand balance formula."""
