@dataclass
class RollYieldMetrics:
    """Detailed roll yield analysis."""
    monthly_yield: float  # 1-month annualized
    quarterly_yield: float  # 3-month annualized
    annual_yield: float  # 12-month estimate
    yield_trend: Literal["IMPROVING", "STABLE", "DETERIORATING"]
    days_in_current_regime: int


class TermStructureAnalyzer:
    # ... existing code ...

    def calculate_roll_yield(
        self, price_data: pd.DataFrame
    ) -> RollYieldMetrics:
        """
        Calculate detailed roll yield metrics.

        Roll yield = gain/loss from rolling futures contracts.
        Backwardation: positive (buy cheap, sell dear)
        Contango: negative (buy dear, sell cheap)
        """
        momentum_5d = self._get_latest_momentum(price_data, "wti_front_momentum_5d")
        momentum_20d = self._get_latest_momentum(price_data, "wti_front_momentum_20d")

        # Estimate yields at different horizons
        monthly = momentum_5d * (252 / 5)  # Annualize 5-day
        quarterly = momentum_20d * (252 / 20)  # Annualize 20-day
        annual = (monthly + quarterly) / 2  # Blended

        # Determine trend
        if momentum_5d > momentum_20d + 0.5:
            trend = "IMPROVING"
        elif momentum_5d < momentum_20d - 0.5:
            trend = "DETERIORATING"
        else:
            trend = "STABLE"

        # Calculate regime duration
        days = self._count_regime_days(price_data)

        return RollYieldMetrics(
            monthly_yield=round(monthly, 2),
            quarterly_yield=round(quarterly, 2),
            annual_yield=round(annual, 2),
            yield_trend=trend,
            days_in_current_regime=days,
        )

    def _count_regime_days(self, df: pd.DataFrame) -> int:
        """Count days in current regime."""
        if df.empty:
            return 0

        # Get momentum history
        mom = df[df["series_id"] == "wti_front_momentum_5d"].sort_values("timestamp")
        if len(mom) < 2:
            return 1

        # Count consecutive days with same sign
        current_sign = np.sign(mom["value"].iloc[-1])
        count = 0

        for val in mom["value"].iloc[::-1]:
            if np.sign(val) == current_sign:
                count += 1
            else:
                break

        return count

# src/liquidity/alerts/oil_term_structure_alerts.py
"""Oil term structure alerts."""

from dataclasses import dataclass
from datetime import datetime
from typing import Any

from liquidity.alerts.base import BaseAlertEngine
from liquidity.analyzers.term_structure import (
    CurveShape,
    TermStructureAnalyzer,
    TermStructureSignal,
    RollYieldMetrics,
)


@dataclass
class TermStructureAlert:
    """Term structure alert data."""
    timestamp: datetime
    alert_type: str
    curve_shape: str
    intensity: float
    message: str
    severity: str  # INFO, WARNING, CRITICAL


class TermStructureAlertEngine(BaseAlertEngine):
    """Generate alerts for oil term structure changes."""

    # Thresholds
    INTENSITY_WARNING = 70
    INTENSITY_CRITICAL = 90
    ROLL_YIELD_THRESHOLD = 20  # % annualized
    REGIME_CHANGE_LOOKBACK = 5  # days

    def __init__(
        self,
        analyzer: TermStructureAnalyzer | None = None,
        discord_webhook: str | None = None,
    ):
        super().__init__(discord_webhook=discord_webhook)
        self.analyzer = analyzer or TermStructureAnalyzer()
        self._last_shape: CurveShape | None = None

    def check_alerts(
        self,
        signal: TermStructureSignal,
        roll_yield: RollYieldMetrics | None = None,
    ) -> list[TermStructureAlert]:
        """
        Check for alert conditions.

        Returns list of triggered alerts.
        """
        alerts = []

        # 1. Curve shape change
        if self._last_shape and signal.curve_shape != self._last_shape:
            alerts.append(self._create_regime_change_alert(signal))

        # 2. High intensity
        if signal.intensity >= self.INTENSITY_CRITICAL:
            alerts.append(self._create_intensity_alert(signal, "CRITICAL"))
        elif signal.intensity >= self.INTENSITY_WARNING:
            alerts.append(self._create_intensity_alert(signal, "WARNING"))

        # 3. Extreme roll yield
        if roll_yield and abs(roll_yield.annual_yield) > self.ROLL_YIELD_THRESHOLD:
            alerts.append(self._create_roll_yield_alert(signal, roll_yield))

        # Update state
        self._last_shape = signal.curve_shape

        return alerts

    def _create_regime_change_alert(
        self, signal: TermStructureSignal
    ) -> TermStructureAlert:
        """Create alert for regime change."""
        old = self._last_shape.value if self._last_shape else "UNKNOWN"
        new = signal.curve_shape.value

        return TermStructureAlert(
            timestamp=signal.timestamp,
            alert_type="REGIME_CHANGE",
            curve_shape=new,
            intensity=signal.intensity,
            message=f"Oil term structure shifted: {old} → {new}",
            severity="WARNING",
        )

    def _create_intensity_alert(
        self, signal: TermStructureSignal, severity: str
    ) -> TermStructureAlert:
        """Create alert for high intensity."""
        shape = signal.curve_shape.value
        direction = "tight supply" if shape == "BACKWARDATION" else "oversupply"

        return TermStructureAlert(
            timestamp=signal.timestamp,
            alert_type="HIGH_INTENSITY",
            curve_shape=shape,
            intensity=signal.intensity,
            message=f"Strong {shape.lower()} signal ({signal.intensity:.0f}/100): {direction}",
            severity=severity,
        )

    def _create_roll_yield_alert(
        self, signal: TermStructureSignal, roll_yield: RollYieldMetrics
    ) -> TermStructureAlert:
        """Create alert for extreme roll yield."""
        direction = "positive" if roll_yield.annual_yield > 0 else "negative"

        return TermStructureAlert(
            timestamp=signal.timestamp,
            alert_type="EXTREME_ROLL_YIELD",
            curve_shape=signal.curve_shape.value,
            intensity=signal.intensity,
            message=f"Extreme roll yield: {roll_yield.annual_yield:+.1f}% annualized ({direction})",
            severity="WARNING",
        )

    def format_discord_message(self, alert: TermStructureAlert) -> dict[str, Any]:
        """Format alert for Discord webhook."""
        color = {
            "INFO": 0x3498DB,  # Blue
            "WARNING": 0xF39C12,  # Orange
            "CRITICAL": 0xE74C3C,  # Red
        }.get(alert.severity, 0x95A5A6)

        emoji = {
            "REGIME_CHANGE": "🔄",
            "HIGH_INTENSITY": "⚠️",
            "EXTREME_ROLL_YIELD": "📊",
        }.get(alert.alert_type, "📢")

        return {
            "embeds": [{
                "title": f"{emoji} Oil Term Structure Alert",
                "description": alert.message,
                "color": color,
                "fields": [
                    {"name": "Curve Shape", "value": alert.curve_shape, "inline": True},
                    {"name": "Intensity", "value": f"{alert.intensity:.0f}/100", "inline": True},
                    {"name": "Severity", "value": alert.severity, "inline": True},
                ],
                "timestamp": alert.timestamp.isoformat(),
            }]
        }

    async def send_alert(self, alert: TermStructureAlert) -> bool:
        """Send alert to Discord."""
        if not self.discord_webhook:
            return False

        message = self.format_discord_message(alert)
        return await self._send_discord_alert(message)

class TestRegimeChangeAlert:
    """Test regime change detection."""

    def test_no_alert_on_first_signal(self, engine, backwardation_signal):
        alerts = engine.check_alerts(backwardation_signal)
        # No regime change on first check
        regime_alerts = [a for a in alerts if a.alert_type == "REGIME_CHANGE"]
        assert len(regime_alerts) == 0

    def test_alert_on_regime_change(self, engine, backwardation_signal, contango_signal):
        # First signal
        engine.check_alerts(backwardation_signal)

        # Second signal with different shape
        alerts = engine.check_alerts(contango_signal)

        regime_alerts = [a for a in alerts if a.alert_type == "REGIME_CHANGE"]
        assert len(regime_alerts) == 1
        assert "BACKWARDATION → CONTANGO" in regime_alerts[0].message
