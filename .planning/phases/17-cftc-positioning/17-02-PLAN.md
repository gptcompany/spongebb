    """Analyze COT positioning data for trading signals."""

    # Percentile thresholds for extremes
    EXTREME_HIGH = 90  # Top 10%
    EXTREME_LOW = 10   # Bottom 10%

    # Rolling window for percentile calculation
    LOOKBACK_WEEKS = 52  # 1 year

    def __init__(self, lookback_weeks: int = 52):
        self.lookback_weeks = lookback_weeks

    def calculate_percentiles(
        self,
        df: pd.DataFrame,
        series_id: str,
    ) -> pd.Series:
        """Calculate rolling percentile rank for a series.

        Uses 'weak' method for percentileofscore to avoid deprecation warning
        and ensure consistent behavior. Excludes current value from comparison
        set to avoid look-ahead bias.
        """
        values = df[df["series_id"] == series_id].sort_values("timestamp")["value"]

        def _percentile_rank(window: pd.Series) -> float:
            """Calculate percentile of last value vs prior values."""
            if len(window) < 2:
                return 50.0  # Default to median if not enough history
            historical = window.iloc[:-1].values  # All values except current
            current = window.iloc[-1]
            return stats.percentileofscore(historical, current, kind='weak')

        return values.rolling(self.lookback_weeks, min_periods=10).apply(_percentile_rank)

### 2. Implement Ratio Calculations

```python
def calculate_ratios(self, row: dict) -> dict:
    """Calculate positioning ratios from raw data."""
    comm_long = row.get("comm_long", 0)
    comm_short = row.get("comm_short", 0)
    spec_long = row.get("spec_long", 0)
    spec_short = row.get("spec_short", 0)
    oi = row.get("open_interest", 1)  # Avoid div by zero

    comm_net = comm_long - comm_short
    spec_net = spec_long - spec_short

    return {
        "comm_net": comm_net,
        "spec_net": spec_net,
        "comm_spec_ratio": comm_net / spec_net if spec_net != 0 else 0,
        "spec_long_short_ratio": spec_long / spec_short if spec_short > 0 else float("inf"),
        "comm_pct_of_oi": (comm_long + comm_short) / oi * 100,
        "spec_pct_of_oi": (spec_long + spec_short) / oi * 100,
    }
```

### 3. Implement Percentile Ranking

```python
def calculate_percentile_ranks(
    self,
    df: pd.DataFrame,
    commodity: str,
) -> pd.DataFrame:
    """Add percentile ranks to positioning data."""
    comm_series = f"cot_{commodity.lower()}_comm_net"
    spec_series = f"cot_{commodity.lower()}_spec_net"

    # Get historical values
    comm_values = df[df["series_id"] == comm_series].set_index("timestamp")["value"]
    spec_values = df[df["series_id"] == spec_series].set_index("timestamp")["value"]

    # Calculate rolling percentiles
    comm_pctl = comm_values.rolling(self.lookback_weeks, min_periods=10).apply(
        lambda x: stats.percentileofscore(x[:-1], x.iloc[-1]) if len(x) > 1 else 50
    )
    spec_pctl = spec_values.rolling(self.lookback_weeks, min_periods=10).apply(
        lambda x: stats.percentileofscore(x[:-1], x.iloc[-1]) if len(x) > 1 else 50
    )

    # Create percentile series for output
    result = []
    for ts, pctl in comm_pctl.items():
        result.append({
            "timestamp": ts,
            "series_id": f"cot_{commodity.lower()}_comm_pctl",
            "source": "calculated",
            "value": pctl,
            "unit": "percentile",
        })
    for ts, pctl in spec_pctl.items():
        result.append({
            "timestamp": ts,
            "series_id": f"cot_{commodity.lower()}_spec_pctl",
            "source": "calculated",
            "value": pctl,
            "unit": "percentile",
        })

    return pd.DataFrame(result)
```

### 4. Implement Extremes Detection

```python
def detect_extremes(
    self,
    df: pd.DataFrame,
    commodities: list[str] | None = None,
) -> pd.DataFrame:
    """Detect extreme positioning conditions."""
    if commodities is None:
        commodities = ["WTI", "GOLD", "COPPER", "SILVER", "NATGAS"]

    extremes = []
    for commodity in commodities:
        spec_pctl_series = f"cot_{commodity.lower()}_spec_pctl"
        comm_pctl_series = f"cot_{commodity.lower()}_comm_pctl"

        spec_pctl = df[df["series_id"] == spec_pctl_series]
        comm_pctl = df[df["series_id"] == comm_pctl_series]

        if spec_pctl.empty or comm_pctl.empty:
            continue

        latest_spec = spec_pctl.iloc[-1]
        latest_comm = comm_pctl.iloc[-1]

        extreme_type = None
        if latest_spec["value"] >= self.EXTREME_HIGH:
            extreme_type = "SPEC_EXTREME_LONG"
        elif latest_spec["value"] <= self.EXTREME_LOW:
            extreme_type = "SPEC_EXTREME_SHORT"
        elif latest_comm["value"] >= self.EXTREME_HIGH:
            extreme_type = "COMM_EXTREME_LONG"
        elif latest_comm["value"] <= self.EXTREME_LOW:
            extreme_type = "COMM_EXTREME_SHORT"

        if extreme_type:
            extremes.append({
                "timestamp": latest_spec["timestamp"],
                "commodity": commodity,
                "extreme_type": extreme_type,
                "spec_percentile": latest_spec["value"],
                "comm_percentile": latest_comm["value"],
            })

    return pd.DataFrame(extremes)
```
