# Plan 12-04: Correlation Trend Predictor

## Overview

Build a correlation trend predictor to forecast whether cross-asset correlations (BTC, SPX, GOLD, DXY vs Net Liquidity) are strengthening or weakening, enabling proactive portfolio adjustments.

## Goals

- Predict correlation direction (strengthening/weakening) for 7/14/30 day horizons
- Directional accuracy > 60%
- Detect correlation breakdowns before they occur
- Alert on significant correlation regime changes

## Approach

Use rolling beta forecasting combined with trend detection:
1. Compute rolling correlations (30d, 90d windows)
2. Apply EWMA smoothing to reduce noise
3. Fit AR(1) model to predict correlation trend
4. Classify direction: strengthening, stable, weakening

### Correlation Trend Model

```
corr_{t+h} = α + β * corr_t + γ * Δcorr_t + ε

Where:
- corr_t: Current rolling correlation
- Δcorr_t: Change in correlation (momentum)
- h: Forecast horizon (7, 14, 30 days)
```

### Task 2: Implement Correlation Features
**File**: `src/liquidity/nowcasting/correlation/features.py`

```python
import numpy as np
import pandas as pd
from typing import Dict, List, Tuple
from dataclasses import dataclass
from enum import Enum

class TrendDirection(Enum):
    """Correlation trend direction."""
    STRENGTHENING = "strengthening"
    STABLE = "stable"
    WEAKENING = "weakening"

@dataclass
class CorrelationFeatures:
    """Features for correlation prediction."""
    asset: str
    current_corr_30d: float
    current_corr_90d: float
    corr_momentum: float           # Δcorr (30d change)
    corr_acceleration: float       # ΔΔcorr (change in momentum)
    ewma_corr: float               # EWMA smoothed
    zscore: float                  # How unusual is current corr

class CorrelationFeatureBuilder:
    """
    Build features for correlation trend prediction.
    """

    def __init__(self, ewma_span: int = 20, zscore_window: int = 252):
        self.ewma_span = ewma_span
        self.zscore_window = zscore_window

    def compute_features(
        self,
        asset_returns: pd.Series,
        liquidity_returns: pd.Series,
        asset_name: str
    ) -> pd.DataFrame:
        """
        Compute correlation features over time.

        Args:
            asset_returns: Daily returns of asset (BTC, SPX, etc.)
            liquidity_returns: Daily returns of Net Liquidity
            asset_name: Name for labeling

        Returns:
            DataFrame with correlation features at each timestamp
        """
        # Rolling correlations
        corr_30d = asset_returns.rolling(30).corr(liquidity_returns)
        corr_90d = asset_returns.rolling(90).corr(liquidity_returns)

        # Momentum features
        corr_momentum = corr_30d.diff(periods=30)
        corr_acceleration = corr_momentum.diff(periods=10)

        # EWMA smoothing
        ewma_corr = corr_30d.ewm(span=self.ewma_span).mean()

        # Z-score (how unusual is current correlation)
        corr_mean = corr_30d.rolling(self.zscore_window).mean()
        corr_std = corr_30d.rolling(self.zscore_window).std()
        zscore = (corr_30d - corr_mean) / (corr_std + 1e-8)

        return pd.DataFrame({
            f'{asset_name}_corr_30d': corr_30d,
            f'{asset_name}_corr_90d': corr_90d,
            f'{asset_name}_momentum': corr_momentum,
            f'{asset_name}_acceleration': corr_acceleration,
            f'{asset_name}_ewma': ewma_corr,
            f'{asset_name}_zscore': zscore
        })

    def get_current_features(
        self,
        asset_returns: pd.Series,
        liquidity_returns: pd.Series,
        asset_name: str
    ) -> CorrelationFeatures:
        """Get features for latest observation."""
        features_df = self.compute_features(asset_returns, liquidity_returns, asset_name)
        latest = features_df.iloc[-1]

        return CorrelationFeatures(
            asset=asset_name,
            current_corr_30d=latest[f'{asset_name}_corr_30d'],
            current_corr_90d=latest[f'{asset_name}_corr_90d'],
            corr_momentum=latest[f'{asset_name}_momentum'],
            corr_acceleration=latest[f'{asset_name}_acceleration'],
            ewma_corr=latest[f'{asset_name}_ewma'],
            zscore=latest[f'{asset_name}_zscore']
        )
```

### Task 3: Implement Trend Predictor
**File**: `src/liquidity/nowcasting/correlation/trend_predictor.py`

```python
from sklearn.linear_model import Ridge
from sklearn.preprocessing import StandardScaler
import numpy as np
import pandas as pd
from typing import List, Dict, Optional
from dataclasses import dataclass
from .features import CorrelationFeatureBuilder, TrendDirection, CorrelationFeatures

@dataclass
class CorrelationForecast:
    """Correlation trend forecast for an asset."""
    asset: str
    horizon: int                    # Days ahead
    predicted_corr: float           # Point estimate
    direction: TrendDirection       # Strengthening/stable/weakening
    confidence: float               # Prediction confidence
    current_corr: float             # For reference
    change: float                   # predicted - current

@dataclass
class CorrelationTrendReport:
    """Full correlation trend report for all assets."""
    timestamp: pd.Timestamp
    forecasts: Dict[str, List[CorrelationForecast]]  # asset -> [7d, 14d, 30d]
    alerts: List[str]               # Significant changes detected

class CorrelationTrendPredictor:
    """
    Predict correlation trends for BTC, SPX, GOLD, DXY vs Net Liquidity.

    Uses AR(1) + momentum model for trend prediction:
    corr_{t+h} = α + β * corr_t + γ * Δcorr_t
    """

    ASSETS = ['BTC', 'SPX', 'GOLD', 'DXY', 'TLT', 'HYG', 'COPPER']
    HORIZONS = [7, 14, 30]

    # Thresholds for direction classification
    STRENGTHEN_THRESHOLD = 0.05    # +5% correlation change
    WEAKEN_THRESHOLD = -0.05       # -5% correlation change
    ALERT_THRESHOLD = 0.15         # 15% change triggers alert

    def __init__(self, alpha: float = 1.0):
        self.alpha = alpha
        self.feature_builder = CorrelationFeatureBuilder()
        self.models: Dict[str, Dict[int, Ridge]] = {}  # asset -> horizon -> model
        self.scalers: Dict[str, StandardScaler] = {}
        self._is_fitted = False

    def fit(
        self,
        asset_returns: Dict[str, pd.Series],
        liquidity_returns: pd.Series
    ) -> "CorrelationTrendPredictor":
        """
        Fit prediction models for all assets and horizons.

        Args:
            asset_returns: Dict of asset name -> daily returns
            liquidity_returns: Net Liquidity daily returns
        """
        for asset in self.ASSETS:
            if asset not in asset_returns:
                continue

            self.models[asset] = {}

            # Build features
            features_df = self.feature_builder.compute_features(
                asset_returns[asset],
                liquidity_returns,
                asset
            )

            # Create targets for each horizon
            corr_col = f'{asset}_corr_30d'

            for horizon in self.HORIZONS:
                # Target: correlation h days ahead
                y = features_df[corr_col].shift(-horizon)

                # Features: current values
                X = features_df[[
                    f'{asset}_corr_30d',
                    f'{asset}_momentum',
                    f'{asset}_acceleration',
                    f'{asset}_zscore'
                ]]

                # Drop NaN
                mask = ~(X.isna().any(axis=1) | y.isna())
                X_clean = X[mask]
                y_clean = y[mask]

                # Scale and fit
                scaler = StandardScaler()
                X_scaled = scaler.fit_transform(X_clean)

                model = Ridge(alpha=self.alpha)
                model.fit(X_scaled, y_clean)

                self.models[asset][horizon] = model

            self.scalers[asset] = scaler

        self._is_fitted = True
        return self

    def predict(
        self,
        asset_returns: Dict[str, pd.Series],
        liquidity_returns: pd.Series
    ) -> CorrelationTrendReport:
        """
        Generate correlation trend forecasts for all assets.
        """
        if not self._is_fitted:
            raise ValueError("Model not fitted. Call fit() first.")

        forecasts = {}
        alerts = []

        for asset in self.ASSETS:
            if asset not in asset_returns or asset not in self.models:
                continue

            asset_forecasts = []

            # Get current features
            current_features = self.feature_builder.get_current_features(
                asset_returns[asset],
                liquidity_returns,
                asset
            )

            # Predict for each horizon
            X = np.array([[
                current_features.current_corr_30d,
                current_features.corr_momentum,
                current_features.corr_acceleration,
                current_features.zscore
            ]])
            X_scaled = self.scalers[asset].transform(X)

            for horizon in self.HORIZONS:
                predicted_corr = self.models[asset][horizon].predict(X_scaled)[0]
                change = predicted_corr - current_features.current_corr_30d

                # Classify direction
                if change > self.STRENGTHEN_THRESHOLD:
                    direction = TrendDirection.STRENGTHENING
                elif change < self.WEAKEN_THRESHOLD:
                    direction = TrendDirection.WEAKENING
                else:
                    direction = TrendDirection.STABLE

                # Check for alert
                if abs(change) > self.ALERT_THRESHOLD:
                    alerts.append(
                        f"{asset}: Correlation expected to {'increase' if change > 0 else 'decrease'} "
                        f"by {abs(change):.2f} over {horizon}d"
                    )

                asset_forecasts.append(CorrelationForecast(
                    asset=asset,
                    horizon=horizon,
                    predicted_corr=predicted_corr,
                    direction=direction,
                    confidence=self._estimate_confidence(X_scaled, horizon),
                    current_corr=current_features.current_corr_30d,
                    change=change
                ))

            forecasts[asset] = asset_forecasts

        return CorrelationTrendReport(
            timestamp=pd.Timestamp.now(),
            forecasts=forecasts,
            alerts=alerts
        )

    def _estimate_confidence(self, X: np.ndarray, horizon: int) -> float:
        """
        Estimate prediction confidence.

        Lower confidence for longer horizons and unusual observations.
        """
        # Base confidence decreases with horizon
        base = np.exp(-horizon / 30)

        # Reduce confidence for outlier observations (high z-score)
        zscore = X[0, 3]  # Last feature is zscore
        zscore_penalty = np.exp(-abs(zscore) / 3)

        return min(base * zscore_penalty, 0.95)

    def get_correlation_breakdown_risk(
        self,
        asset_returns: Dict[str, pd.Series],
        liquidity_returns: pd.Series
    ) -> Dict[str, float]:
        """
        Calculate risk of correlation breakdown for each asset.

        High risk = correlation likely to flip sign or collapse to near-zero.
        """
        risks = {}

        for asset in self.ASSETS:
            if asset not in asset_returns:
                continue

            features = self.feature_builder.get_current_features(
                asset_returns[asset],
                liquidity_returns,
                asset
            )

            # Risk factors:
            # 1. Current correlation near zero (unstable)
            zero_risk = np.exp(-abs(features.current_corr_30d) * 3)

            # 2. High acceleration (rapid change)
            accel_risk = min(abs(features.corr_acceleration) * 5, 1.0)

            # 3. Divergence between 30d and 90d (mean reversion)
            divergence = abs(features.current_corr_30d - features.current_corr_90d)
            divergence_risk = min(divergence * 2, 1.0)

            # Combined risk
            risks[asset] = (zero_risk + accel_risk + divergence_risk) / 3

        return risks
```

```python
"""
Backtest correlation trend predictor directional accuracy.

Usage: uv run python scripts/backtest_correlation.py
"""

async def main():
    # Load 5 years of data
    asset_returns = await load_asset_returns()
    liquidity_returns = await load_liquidity_returns()

    # Walk-forward validation
    train_window = 252 * 2  # 2 years training

    results = []
    for test_date in pd.date_range('2022-01-01', '2025-12-31', freq='W'):
        train_end = test_date - pd.Timedelta(days=1)

        # Train predictor
        predictor = CorrelationTrendPredictor()
        predictor.fit(
            {a: asset_returns[a].loc[:train_end] for a in asset_returns},
            liquidity_returns.loc[:train_end]
        )

        # Predict
        report = predictor.predict(
            {a: asset_returns[a].loc[:test_date] for a in asset_returns},
            liquidity_returns.loc[:test_date]
        )

        # Evaluate vs actual
        for asset, forecasts in report.forecasts.items():
            for f in forecasts:
                actual_date = test_date + pd.Timedelta(days=f.horizon)
                if actual_date in liquidity_returns.index:
                    actual_corr = (
                        asset_returns[asset].loc[:actual_date]
                        .rolling(30).corr(liquidity_returns.loc[:actual_date])
                        .iloc[-1]
                    )
                    actual_change = actual_corr - f.current_corr
                    predicted_direction = 1 if f.change > 0 else -1
                    actual_direction = 1 if actual_change > 0 else -1

                    results.append({
                        'date': test_date,
                        'asset': asset,
                        'horizon': f.horizon,
                        'correct': predicted_direction == actual_direction
                    })

    df = pd.DataFrame(results)
    accuracy = df.groupby(['asset', 'horizon'])['correct'].mean()
    print(accuracy)
```
