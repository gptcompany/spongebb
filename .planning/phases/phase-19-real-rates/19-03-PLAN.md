# Plan 19-03: Oil-Real Rates Correlation Engine

## Goal

Analizzare la correlazione tra oil prices e real rates per identificare regime shifts.

## Rationale

**Correlazione storica oil-real rates: ~-0.4 to -0.6**

Meccanismi:
1. **USD channel**: Real rates ↑ → USD ↑ → Oil ↓ (priced in USD)
2. **Growth channel**: Real rates ↑ → Growth expectations ↓ → Oil demand ↓
3. **Inflation channel**: Oil ↑ → Inflation ↑ → Real rates ↓ (feedback loop)

Questo spiega perché:
- 2022: Real rates surge → Oil correction
- 2020: Real rates collapse → Oil recovery

### Task 2: Create oil_real_rates module

```python
# src/liquidity/analyzers/oil_real_rates.py

@dataclass
class OilRealRatesCorrelation:
    """Oil vs Real Rates correlation analysis."""
    timestamp: datetime
    corr_30d: float
    corr_90d: float
    corr_ewma: float  # halflife=30
    p_value_30d: float
    p_value_90d: float
    regime: str  # "normal", "breakdown", "surge"

class OilRealRatesAnalyzer:
    """Analyze oil-real rates relationship."""

    NORMAL_CORR_RANGE = (-0.7, -0.3)  # Expected negative correlation

    def __init__(self, settings: Settings | None = None):
        self.settings = settings or get_settings()
        self._fred = FREDCollector(settings=self.settings)
        self._corr_engine = CorrelationEngine(settings=self.settings)

    async def compute_correlation(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
        window_30d: int = 30,
        window_90d: int = 90,
    ) -> pd.DataFrame:
        """Compute rolling oil-real rates correlation.

        Returns DataFrame with:
        - oil_price: WTI price
        - real_10y: 10Y TIPS yield
        - corr_30d: 30-day rolling correlation
        - corr_90d: 90-day rolling correlation
        - regime: "normal", "breakdown", "surge"
        """
        # Fetch oil data
        oil_df = await self._fetch_oil_data(start_date, end_date)

        # Fetch real rates
        rates_df = await self._fred.fetch_series("DFII10", start_date, end_date)

        # Align and merge
        df = pd.merge(
            oil_df, rates_df,
            left_index=True, right_index=True,
            how="inner"
        )

        # Calculate returns (correlation on returns, not levels)
        df["oil_ret"] = df["oil_price"].pct_change()
        df["rates_ret"] = df["real_10y"].diff()  # rates in levels, use diff

        # Rolling correlations
        df["corr_30d"] = df["oil_ret"].rolling(window_30d).corr(df["rates_ret"])
        df["corr_90d"] = df["oil_ret"].rolling(window_90d).corr(df["rates_ret"])

        # Classify regime
        df["regime"] = df["corr_30d"].apply(self._classify_regime)

        return df

    def _classify_regime(self, corr: float) -> str:
        """Classify correlation regime."""
        if pd.isna(corr):
            return "unknown"
        if corr < self.NORMAL_CORR_RANGE[0]:
            return "surge"  # Unusually strong negative correlation
        if corr > self.NORMAL_CORR_RANGE[1]:
            return "breakdown"  # Correlation breakdown (positive or weak)
        return "normal"

    async def get_current_state(self) -> OilRealRatesCorrelation:
        """Get current correlation state."""
        df = await self.compute_correlation()
        latest = df.dropna().iloc[-1]

        # Calculate p-values
        window_30 = df["oil_ret"].iloc[-30:].dropna()
        window_90 = df["oil_ret"].iloc[-90:].dropna()

        _, p_30 = stats.pearsonr(
            window_30,
            df["rates_ret"].iloc[-30:].dropna()
        )
        _, p_90 = stats.pearsonr(
            window_90,
            df["rates_ret"].iloc[-90:].dropna()
        )

        return OilRealRatesCorrelation(
            timestamp=latest.name,
            corr_30d=latest["corr_30d"],
            corr_90d=latest["corr_90d"],
            corr_ewma=df["corr_30d"].ewm(halflife=30).mean().iloc[-1],
            p_value_30d=p_30,
            p_value_90d=p_90,
            regime=latest["regime"],
        )
```

async def test_oil_real_correlation():
