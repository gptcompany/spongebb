## Objective

Create a collector for EUR/USD cross-currency basis swap data as a post-LIBOR stress indicator. Negative basis indicates USD funding stress globally.

## Background

Cross-currency basis swaps show the premium/discount for borrowing USD vs other currencies. A widening negative basis (e.g., EUR/USD < -30bps) indicates global USD funding stress, a key signal for liquidity crises.

```python
# src/liquidity/collectors/xccy_basis.py
"""Cross-currency basis swap collector.

EUR/USD basis as post-LIBOR stress indicator.
"""

import logging
from datetime import UTC, datetime, timedelta
from typing import Any

import httpx
import pandas as pd

from liquidity.collectors.base import BaseCollector, CollectorFetchError
from liquidity.collectors.registry import registry
from liquidity.config import Settings, get_settings

logger = logging.getLogger(__name__)

# ECB SDW API for EUR/USD basis
ECB_SDW_BASE = "https://data-api.ecb.europa.eu/service/data"

# Stress thresholds (basis points)
STRESS_THRESHOLDS = {
    "normal": 0,       # > 0 bps
    "mild": -10,       # -10 to 0 bps
    "moderate": -30,   # -30 to -10 bps
    "severe": -50,     # < -30 bps
}


class XCcyBasisCollector(BaseCollector[pd.DataFrame]):
    """Collector for cross-currency basis swap data.

    Provides EUR/USD cross-currency basis as a stress indicator.
    Negative basis indicates USD funding premium (stress).

    Data Sources (in order of preference):
    1. ECB SDW (Statistical Data Warehouse)
    2. FRED proxy (if available)
    3. CME index (may require subscription)
    """

    def __init__(
        self,
        name: str = "xccy_basis",
        settings: Settings | None = None,
        **kwargs: Any,
    ) -> None:
        super().__init__(name=name, settings=settings, **kwargs)
        self._settings = settings or get_settings()
        self._client: httpx.AsyncClient | None = None

    async def _get_client(self) -> httpx.AsyncClient:
        if self._client is None or self._client.is_closed:
            self._client = httpx.AsyncClient(
                timeout=30.0,
                headers={"Accept": "application/json"},
            )
        return self._client

    async def collect(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
        tenor: str = "3M",
    ) -> pd.DataFrame:
        """Collect cross-currency basis data.

        Args:
            start_date: Start date (default: 365 days ago)
            end_date: End date (default: today)
            tenor: Swap tenor ("3M", "1Y", "5Y")

        Returns:
            DataFrame with columns: timestamp, series_id, source, value, unit
        """
        if start_date is None:
            start_date = datetime.now(UTC) - timedelta(days=365)
        if end_date is None:
            end_date = datetime.now(UTC)

        async def _fetch() -> pd.DataFrame:
            # Try ECB first
            try:
                return await self._fetch_ecb(start_date, end_date, tenor)
            except Exception as e:
                logger.warning("ECB SDW failed: %s, trying fallback", e)

            # Try FRED fallback
            try:
                return await self._fetch_fred_proxy(start_date, end_date)
            except Exception as e:
                logger.warning("FRED proxy failed: %s", e)

            # Return empty with cached value placeholder
            logger.error("All xccy basis sources failed")
            return self._empty_df()

        try:
            return await self.fetch_with_retry(_fetch)
        except Exception as e:
            logger.error("XCcy basis fetch failed: %s", e)
            raise CollectorFetchError(f"XCcy basis fetch failed: {e}") from e

    async def _fetch_ecb(
        self,
        start_date: datetime,
        end_date: datetime,
        tenor: str,
    ) -> pd.DataFrame:
        """Fetch from ECB Statistical Data Warehouse."""
        client = await self._get_client()

        # ECB SDMX flow for money market rates
        # This is a simplified approach - full SDMX parsing would be more robust
        flow = "FM"
        key = f"M.U2.EUR.4F.BB.US_FX_EUR_M_{tenor}.HSTA"

        url = f"{ECB_SDW_BASE}/{flow}/{key}"
        params = {
            "startPeriod": start_date.strftime("%Y-%m"),
            "endPeriod": end_date.strftime("%Y-%m"),
            "format": "jsondata",
        }

        try:
            response = await client.get(url, params=params)
            response.raise_for_status()
            data = response.json()
        except httpx.HTTPStatusError as e:
            logger.warning("ECB SDW request failed: %s", e)
            raise

        # Parse ECB SDMX JSON response
        observations = data.get("dataSets", [{}])[0].get("series", {})

        rows = []
        for series_key, series_data in observations.items():
            obs = series_data.get("observations", {})
            for time_idx, values in obs.items():
                # Get time dimension
                time_dims = data.get("structure", {}).get("dimensions", {}).get("observation", [])
                time_values = time_dims[0].get("values", []) if time_dims else []

                if int(time_idx) < len(time_values):
                    period = time_values[int(time_idx)].get("id")
                    value = values[0] if values else None

                    if value is not None:
                        rows.append({
                            "timestamp": pd.to_datetime(period),
                            "series_id": f"XCCY_EURUSD_{tenor}",
                            "source": "ecb_sdw",
                            "value": float(value),  # basis points
                            "unit": "bps",
                        })

        df = pd.DataFrame(rows)
        df = df.sort_values("timestamp").reset_index(drop=True)

        logger.info("Fetched %d ECB xccy basis records", len(df))
        return df

    async def _fetch_fred_proxy(
        self,
        start_date: datetime,
        end_date: datetime,
    ) -> pd.DataFrame:
        """Fetch proxy from FRED (if available series exists).

        Note: As of 2026, no direct EUR/USD basis series exists in FRED.
        This is a placeholder for future availability.
        """
        # No direct FRED series for EUR/USD basis currently
        raise CollectorFetchError("No FRED proxy available for xccy basis")

    def _empty_df(self) -> pd.DataFrame:
        return pd.DataFrame(
            columns=["timestamp", "series_id", "source", "value", "unit"]
        )

    @staticmethod
    def classify_stress(basis_bps: float) -> str:
        """Classify stress level from basis value.

        Args:
            basis_bps: Basis in basis points (negative = stress)

        Returns:
            Stress level: "normal", "mild", "moderate", "severe"
        """
        if basis_bps > STRESS_THRESHOLDS["normal"]:
            return "normal"
        elif basis_bps > STRESS_THRESHOLDS["mild"]:
            return "mild"
        elif basis_bps > STRESS_THRESHOLDS["moderate"]:
            return "moderate"
        else:
            return "severe"

    async def close(self) -> None:
        if self._client is not None and not self._client.is_closed:
            await self._client.aclose()


# Register collector
registry.register("xccy_basis", XCcyBasisCollector)
```

```python
# tests/integration/collectors/test_xccy_basis_integration.py
"""Integration tests for cross-currency basis collector."""

import pytest
from datetime import UTC, datetime, timedelta

from liquidity.collectors.xccy_basis import XCcyBasisCollector


@pytest.mark.integration
async def test_collect_ecb_real_api():
    """Test fetching from ECB SDW API.

    Note: ECB API may have data availability issues. Test is informational.
    """
    collector = XCcyBasisCollector()

    try:
        start = datetime.now(UTC) - timedelta(days=365)
        df = await collector.collect(start_date=start)

        # May be empty if ECB API unavailable
        assert isinstance(df, pd.DataFrame)

        if len(df) > 0:
            # Basis typically -100 to +50 bps
            assert df["value"].min() > -200
            assert df["value"].max() < 100

            # Check stress classification
            latest = df["value"].iloc[-1]
            stress = XCcyBasisCollector.classify_stress(latest)
            assert stress in ["normal", "mild", "moderate", "severe"]

    finally:
        await collector.close()


@pytest.mark.integration
async def test_stress_levels():
    """Test stress level interpretation."""
    # These thresholds are based on historical crisis data

    # Normal conditions (positive basis rare but possible)
    assert XCcyBasisCollector.classify_stress(10) == "normal"

    # GFC 2008: basis reached -100bps
    assert XCcyBasisCollector.classify_stress(-80) == "severe"

    # COVID March 2020: basis reached -50bps
    assert XCcyBasisCollector.classify_stress(-45) == "severe"

    # Typical stress: -20 to -30bps
    assert XCcyBasisCollector.classify_stress(-25) == "moderate"
```
