# src/liquidity/collectors/oil_term_structure.py
"""Oil term structure collector via yfinance."""

from datetime import datetime, timedelta
from typing import Any

import pandas as pd
import yfinance as yf

from liquidity.collectors.base import BaseCollector
from liquidity.collectors.registry import registry
from liquidity.config import get_settings

# Series mapping
SERIES_MAP = {
    "wti_front": "CL=F",
    "brent_front": "BZ=F",
}

UNIT_MAP = {
    "wti_front": "usd_per_barrel",
    "brent_front": "usd_per_barrel",
}


class OilTermStructureCollector(BaseCollector[pd.DataFrame]):
    """Collect WTI/Brent futures prices for term structure analysis."""

    def __init__(
        self,
        name: str = "oil_term_structure",
        settings: Any | None = None,
    ) -> None:
        super().__init__(name=name, settings=settings or get_settings())

    async def collect(
        self,
        symbols: list[str] | None = None,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect futures price data."""
        symbols = symbols or list(SERIES_MAP.keys())
        end_date = end_date or datetime.now()
        start_date = start_date or (end_date - timedelta(days=365 * 2))

        results = []
        for series_id in symbols:
            if series_id not in SERIES_MAP:
                continue

            ticker = SERIES_MAP[series_id]
            df = await self._fetch_ticker(ticker, start_date, end_date)

            if df.empty:
                continue

            # Normalize to standard schema
            normalized = pd.DataFrame({
                "timestamp": df.index,
                "series_id": series_id,
                "source": "yfinance",
                "value": df["Adj Close"].values,
                "unit": UNIT_MAP[series_id],
            })
            results.append(normalized)

        if not results:
            return pd.DataFrame()

        return pd.concat(results, ignore_index=True)

    async def _fetch_ticker(
        self,
        ticker: str,
        start_date: datetime,
        end_date: datetime,
    ) -> pd.DataFrame:
        """Fetch ticker data from yfinance."""
        try:
            data = yf.download(
                ticker,
                start=start_date,
                end=end_date,
                progress=False,
            )
            return data
        except Exception as e:
            self.logger.warning(f"Failed to fetch {ticker}: {e}")
            return pd.DataFrame()

    async def collect_wti(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Convenience method for WTI only."""
        return await self.collect(["wti_front"], start_date, end_date)

    async def collect_with_momentum(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect prices with momentum calculations."""
        df = await self.collect(["wti_front"], start_date, end_date)

        if df.empty:
            return df

        # Pivot and calculate momentum
        pivot = df.pivot(index="timestamp", columns="series_id", values="value")

        momentum_5d = pivot.pct_change(5) * 100
        momentum_20d = pivot.pct_change(20) * 100

        # Add momentum series
        for col in pivot.columns:
            mom_5d_df = pd.DataFrame({
                "timestamp": momentum_5d.index,
                "series_id": f"{col}_momentum_5d",
                "source": "calculated",
                "value": momentum_5d[col].values,
                "unit": "percent",
            })
            mom_20d_df = pd.DataFrame({
                "timestamp": momentum_20d.index,
                "series_id": f"{col}_momentum_20d",
                "source": "calculated",
                "value": momentum_20d[col].values,
                "unit": "percent",
            })
            df = pd.concat([df, mom_5d_df, mom_20d_df], ignore_index=True)

        return df


# Register collector
registry.register("oil_term_structure", OilTermStructureCollector)

"""Unit tests for OilTermStructureCollector."""

from datetime import datetime, timedelta

        assert SERIES_MAP["brent_front"] == "BZ=F"

    def test_unit_map_covers_all_series(self):

    """Test collector initialization."""

    def test_default_name(self):
        collector = OilTermStructureCollector()
        assert collector.name == "oil_term_structure"

        collector = OilTermStructureCollector(name="custom")
        assert collector.name == "custom"


class TestCollect:
    """Test collect method."""

    @pytest.fixture
    def mock_yf_data(self):
        dates = pd.date_range("2024-01-01", periods=10, freq="D")
        return pd.DataFrame(
            {"Adj Close": [70 + i for i in range(10)]},
            index=dates,
        )

    @pytest.mark.asyncio
    async def test_collect_returns_dataframe(self, mock_yf_data):
        collector = OilTermStructureCollector()

        with patch("yfinance.download", return_value=mock_yf_data):
            result = await collector.collect(["wti_front"])

        assert isinstance(result, pd.DataFrame)
        assert not result.empty

    @pytest.mark.asyncio
    async def test_collect_has_required_columns(self, mock_yf_data):
        collector = OilTermStructureCollector()

        with patch("yfinance.download", return_value=mock_yf_data):
            result = await collector.collect(["wti_front"])

        required_cols = ["timestamp", "series_id", "source", "value", "unit"]
        for col in required_cols:
            assert col in result.columns

    @pytest.mark.asyncio
    async def test_collect_correct_source(self, mock_yf_data):
        collector = OilTermStructureCollector()

        with patch("yfinance.download", return_value=mock_yf_data):
            result = await collector.collect(["wti_front"])

        assert (result["source"] == "yfinance").all()

    @pytest.mark.asyncio
    async def test_collect_empty_on_failure(self):
        collector = OilTermStructureCollector()

        with patch("yfinance.download", side_effect=Exception("API error")):
            result = await collector.collect(["wti_front"])

        assert result.empty

# tests/integration/collectors/test_oil_term_structure_integration.py
"""Integration tests for OilTermStructureCollector with live data."""

from datetime import datetime, timedelta

import pytest

from liquidity.collectors.oil_term_structure import OilTermStructureCollector


@pytest.mark.integration
class TestLiveDataFetch:
    """Test with real yfinance API."""

    @pytest.mark.asyncio
    async def test_collect_wti_live(self):
        """Fetch actual WTI data."""
        collector = OilTermStructureCollector()

        end_date = datetime.now()
        start_date = end_date - timedelta(days=30)

        result = await collector.collect(
            ["wti_front"],
            start_date=start_date,
            end_date=end_date,
        )

        # Should have data
        assert not result.empty

        # Price should be reasonable ($20-$200/barrel)
        prices = result[result["series_id"] == "wti_front"]["value"]
        assert prices.min() > 20
        assert prices.max() < 200

    @pytest.mark.asyncio
    async def test_collect_with_momentum_live(self):
        """Fetch with momentum calculations."""
        collector = OilTermStructureCollector()

        result = await collector.collect_with_momentum()

        # Should have momentum series
        series_ids = result["series_id"].unique()
        assert "wti_front" in series_ids
        assert "wti_front_momentum_5d" in series_ids

    @pytest.mark.asyncio
    async def test_data_freshness(self):
        """Verify data is recent (within 5 days)."""
        collector = OilTermStructureCollector()

        result = await collector.collect_wti()

        if not result.empty:
            latest = result["timestamp"].max()
            days_old = (datetime.now() - pd.Timestamp(latest).to_pydatetime()).days
            assert days_old < 5, f"Data is {days_old} days old"
