```python
from dash import html, dcc
import plotly.graph_objects as go

class PositioningPanel:
    """Dashboard panel for COT positioning visualization."""

    COMMODITIES = ["WTI", "GOLD", "COPPER", "SILVER", "NATGAS"]

    def __init__(self):
        self.layout = self._create_layout()

    def _create_layout(self) -> html.Div:
        """Create panel layout."""
        return html.Div([
            html.H3("📊 COT Positioning", className="panel-title"),

            # Heatmap section
            html.Div([
                html.H4("Current Positioning (52-Week Percentile)"),
                dcc.Graph(id="positioning-heatmap"),
            ], className="panel-section"),

            # Time series section
            html.Div([
                html.H4("Historical Net Positions"),
                dcc.Dropdown(
                    id="positioning-commodity-select",
                    options=[{"label": c, "value": c} for c in self.COMMODITIES],
                    value="WTI",
                    className="dropdown"
                ),
                dcc.Graph(id="positioning-timeseries"),
            ], className="panel-section"),

            # Extremes table
            html.Div([
                html.H4("⚠️ Current Extremes"),
                html.Div(id="positioning-extremes-table"),
            ], className="panel-section"),
        ], className="positioning-panel")
```

### 2. Create Heatmap Figure

```python
def create_positioning_heatmap(
    data: pd.DataFrame,
) -> go.Figure:
    """Create heatmap showing current percentile ranks."""
    commodities = ["WTI", "GOLD", "COPPER", "SILVER", "NATGAS"]
    categories = ["Speculator", "Commercial", "Swap Dealer"]

    # Build matrix
    z = []
    for commodity in commodities:
        row = []
        for cat, suffix in [("Speculator", "spec"), ("Commercial", "comm"), ("Swap Dealer", "swap")]:
            series_id = f"cot_{commodity.lower()}_{suffix}_pctl"
            val = data[data["series_id"] == series_id]["value"].iloc[-1] if not data[data["series_id"] == series_id].empty else 50
            row.append(val)
        z.append(row)

    fig = go.Figure(data=go.Heatmap(
        z=z,
        x=categories,
        y=commodities,
        colorscale=[
            [0.0, "#FF4136"],    # Red (extreme short)
            [0.1, "#FF851B"],    # Orange
            [0.3, "#FFDC00"],    # Yellow
            [0.5, "#AAAAAA"],    # Gray (neutral)
            [0.7, "#2ECC40"],    # Light green
            [0.9, "#01FF70"],    # Lime
            [1.0, "#3D9970"],    # Dark green (extreme long)
        ],
        text=[[f"{v:.0f}%" for v in row] for row in z],
        texttemplate="%{text}",
        textfont={"size": 14, "color": "white"},
        hovertemplate="<b>%{y}</b><br>%{x}: %{z:.1f}th percentile<extra></extra>",
        zmin=0,
        zmax=100,
        colorbar=dict(
            title="Percentile",
            tickvals=[0, 25, 50, 75, 100],
            ticktext=["0%", "25%", "50%", "75%", "100%"],
        ),
    ))

    fig.update_layout(
        title="COT Positioning Heatmap (52-Week Percentile)",
        xaxis_title="Trader Category",
        yaxis_title="Commodity",
        height=350,
        margin=dict(l=80, r=50, t=50, b=50),
    )

    return fig
```

### 3. Create Time Series Figure

```python
def create_positioning_timeseries(
    data: pd.DataFrame,
    commodity: str,
    weeks: int = 52,
) -> go.Figure:
    """Create time series of net positions."""
    fig = go.Figure()

    # Get series for this commodity
    comm_net = data[data["series_id"] == f"cot_{commodity.lower()}_comm_net"]
    spec_net = data[data["series_id"] == f"cot_{commodity.lower()}_spec_net"]

    if not comm_net.empty:
        fig.add_trace(go.Scatter(
            x=comm_net["timestamp"],
            y=comm_net["value"],
            name="Commercial Net",
            line=dict(color="#3D9970", width=2),
            fill="tozeroy",
            fillcolor="rgba(61, 153, 112, 0.2)",
        ))

    if not spec_net.empty:
        fig.add_trace(go.Scatter(
            x=spec_net["timestamp"],
            y=spec_net["value"],
            name="Speculator Net",
            line=dict(color="#FF4136", width=2),
            fill="tozeroy",
            fillcolor="rgba(255, 65, 54, 0.2)",
        ))

    # Add zero line
    fig.add_hline(y=0, line_dash="dash", line_color="gray")

    fig.update_layout(
        title=f"{commodity} Net Positioning (Contracts)",
        xaxis_title="Date",
        yaxis_title="Net Contracts",
        height=400,
        legend=dict(orientation="h", yanchor="bottom", y=1.02),
        hovermode="x unified",
    )

    return fig
```

### 4. Create Extremes Table

```python
def create_extremes_table(
    extremes: pd.DataFrame,
) -> html.Table:
    """Create table showing current extreme conditions."""
    if extremes.empty:
        return html.P("No extreme positioning detected", className="no-extremes")

    severity_colors = {
        "critical": "#FF4136",
        "warning": "#FF851B",
    }

    rows = []
    for _, row in extremes.iterrows():
        severity = "critical" if row["spec_percentile"] >= 95 or row["spec_percentile"] <= 5 else "warning"
        rows.append(html.Tr([
            html.Td(row["commodity"]),
            html.Td(row["extreme_type"]),
            html.Td(f"{row['spec_percentile']:.1f}%"),
            html.Td(f"{row['comm_percentile']:.1f}%"),
            html.Td("⚠️" if severity == "warning" else "🚨"),
        ], style={"backgroundColor": severity_colors.get(severity, "white")}))

    return html.Table([
        html.Thead(html.Tr([
            html.Th("Commodity"),
            html.Th("Alert Type"),
            html.Th("Spec %ile"),
            html.Th("Comm %ile"),
            html.Th(""),
        ])),
        html.Tbody(rows),
    ], className="extremes-table")
```

### 5. Register Callbacks

**File:** `src/liquidity/dashboard/callbacks/positioning_callbacks.py`

```python
from dash import Input, Output, callback
from liquidity.collectors import CFTCCOTCollector
from liquidity.analysis.positioning import PositioningAnalyzer


@callback(
    Output("positioning-heatmap", "figure"),
    Input("interval-component", "n_intervals"),
)
async def update_heatmap(_):
    """Update positioning heatmap."""
    collector = CFTCCOTCollector()
    data = await collector.collect(weeks=52)

    analyzer = PositioningAnalyzer()
    with_percentiles = analyzer.add_percentiles(data)

    return create_positioning_heatmap(with_percentiles)


@callback(
    Output("positioning-timeseries", "figure"),
    Input("positioning-commodity-select", "value"),
    Input("interval-component", "n_intervals"),
)
async def update_timeseries(commodity, _):
    """Update positioning time series."""
    collector = CFTCCOTCollector()
    data = await collector.collect(commodities=[commodity], weeks=104)

    return create_positioning_timeseries(data, commodity)


@callback(
    Output("positioning-extremes-table", "children"),
    Input("interval-component", "n_intervals"),
)
async def update_extremes(_):
    """Update extremes table."""
    collector = CFTCCOTCollector()
    data = await collector.collect(weeks=52)

    analyzer = PositioningAnalyzer()
    with_percentiles = analyzer.add_percentiles(data)
    extremes = analyzer.detect_extremes(with_percentiles)

    return create_extremes_table(extremes)
```
