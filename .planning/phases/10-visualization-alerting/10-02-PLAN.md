```python
def get_stress_color(value: float, thresholds: tuple) -> str:
    """Return color based on threshold breach."""
    low, high = thresholds
    if value < low:
        return '#00ff88'  # Green - normal
    elif value < high:
        return '#ffaa00'  # Yellow - elevated
    return '#ff4444'  # Red - critical
```

```python
    fig = go.Figure(data=go.Heatmap(
```

### 6. Calendar Overlay (components/calendar.py)
```python
def create_calendar_strip():
    """Horizontal strip showing upcoming events."""
    return html.Div([
        html.Span("Upcoming: ", className='calendar-label'),
        html.Div(id='calendar-events', className='calendar-strip'),
    ])

def create_calendar_events(events: list[CalendarEvent]) -> list:
    """Render event badges with impact color coding."""
    badges = []
    for event in events[:5]:  # Show next 5
        color = {
            'high': 'danger',
            'medium': 'warning',
            'low': 'secondary'
        }[event.impact.value]
        badges.append(dbc.Badge(
            f"{event.title} ({event.event_date.strftime('%b %d')})",
            color=color,
            className='me-2',
        ))
    return badges

def add_calendar_overlay(fig: go.Figure, events: list[CalendarEvent]) -> go.Figure:
    """Add vertical lines for high-impact events on time series."""
    for event in events:
        if event.impact.value == 'high':
            fig.add_vline(
                x=event.event_date,
                line_dash='dash',
                annotation_text=event.title,
                annotation_position='top',
            )
    return fig
```
