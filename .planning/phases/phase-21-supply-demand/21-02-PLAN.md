# Plan 21-02: Inventory Forecast

## Goal

Previsione inventory con comparazioni YoY e seasonal adjustment.

## Metrics

1. **YoY Change**: Current inventory vs same week last year
2. **5-Year Average**: Current vs 5-year seasonal average
3. **Days of Supply**: Inventory / Daily consumption rate

## Tasks

### Task 1: Create inventory_forecast.py

```python
# src/liquidity/oil/inventory_forecast.py

from dataclasses import dataclass
from datetime import datetime, timedelta
from typing import Optional
import pandas as pd
import numpy as np

from liquidity.collectors.eia import EIACollector

@dataclass
class InventoryForecast:
    """Inventory analysis and forecast."""
    date: datetime
    current_stocks: float  # million barrels
    yoy_change: float  # million barrels vs last year
    yoy_change_pct: float  # % change vs last year
    vs_5yr_avg: float  # million barrels vs 5-year avg
    vs_5yr_avg_pct: float  # % vs 5-year avg
    days_of_supply: float  # days of demand coverage
    weekly_trend: str  # "building", "drawing", "stable"
    forecast_4wk: float  # 4-week ahead forecast (million bbl)

class InventoryForecaster:
    """Forecasts oil inventory using seasonal patterns."""

    def __init__(self, eia_collector: EIACollector = None):
        self.eia = eia_collector or EIACollector()

    async def analyze_inventory(
        self,
        lookback_years: int = 5,
    ) -> pd.DataFrame:
        """Analyze current inventory vs historical patterns.

        Returns DataFrame with:
        - current_stocks
        - yoy_change, yoy_change_pct
        - vs_5yr_avg, vs_5yr_avg_pct
        - days_of_supply
        - weekly_trend
        """
        # Fetch 5+ years of data
        start_date = datetime.now() - timedelta(days=365 * (lookback_years + 1))
        stocks = await self.eia.collect(start_date=start_date)

        # Calculate metrics
        df = self._calculate_yoy(stocks)
        df = self._calculate_5yr_avg(df, lookback_years)
        df = self._calculate_days_of_supply(df)
        df = self._calculate_trend(df)

        return df

    def _calculate_yoy(self, df: pd.DataFrame) -> pd.DataFrame:
        """Calculate year-over-year change."""
        # Shift by 52 weeks for YoY comparison
        df["stocks_yoy"] = df["crude_stocks"].shift(52)
        df["yoy_change"] = df["crude_stocks"] - df["stocks_yoy"]
        df["yoy_change_pct"] = (df["yoy_change"] / df["stocks_yoy"]) * 100
        return df

    def _calculate_5yr_avg(self, df: pd.DataFrame, years: int) -> pd.DataFrame:
        """Calculate 5-year seasonal average."""
        # Group by week of year
        df["week_of_year"] = df.index.isocalendar().week

        # Calculate 5-year average per week
        five_yr_avg = (
            df.groupby("week_of_year")["crude_stocks"]
            .rolling(window=years, min_periods=3)
            .mean()
            .reset_index(level=0, drop=True)
        )

        df["five_yr_avg"] = five_yr_avg
        df["vs_5yr_avg"] = df["crude_stocks"] - df["five_yr_avg"]
        df["vs_5yr_avg_pct"] = (df["vs_5yr_avg"] / df["five_yr_avg"]) * 100

        return df

    def _calculate_days_of_supply(self, df: pd.DataFrame) -> pd.DataFrame:
        """Calculate days of supply based on refinery inputs."""
        # Days of supply = Stocks / (Refinery inputs per day)
        # Refinery inputs are in thousand b/d
        if "refinery_inputs" in df.columns:
            df["days_of_supply"] = (df["crude_stocks"] * 1000) / df["refinery_inputs"]
        else:
            # Estimate using average refinery run rate (~16.5 million b/d)
            df["days_of_supply"] = df["crude_stocks"] / 16.5

        return df

    def _calculate_trend(self, df: pd.DataFrame) -> pd.DataFrame:
        """Calculate weekly trend (4-week moving average)."""
        df["stocks_ma4"] = df["crude_stocks"].rolling(4).mean()
        df["stocks_ma4_change"] = df["stocks_ma4"].diff()

        def classify_trend(change: float) -> str:
            if pd.isna(change):
                return "unknown"
            if change > 1:  # >1 million bbl/week build
                return "building"
            elif change < -1:  # >1 million bbl/week draw
                return "drawing"
            return "stable"

        df["weekly_trend"] = df["stocks_ma4_change"].apply(classify_trend)
        return df

    async def forecast_4wk(self) -> float:
        """Forecast inventory 4 weeks ahead using seasonal pattern."""
        df = await self.analyze_inventory()
        current = df["crude_stocks"].iloc[-1]
        trend = df["stocks_ma4_change"].iloc[-1]

        # Simple linear forecast
        forecast = current + (trend * 4)
        return forecast

    async def get_current_analysis(self) -> InventoryForecast:
        """Get current inventory analysis."""
        df = await self.analyze_inventory()
        latest = df.iloc[-1]
        forecast = await self.forecast_4wk()

        return InventoryForecast(
            date=latest.name,
            current_stocks=latest["crude_stocks"],
            yoy_change=latest["yoy_change"],
            yoy_change_pct=latest["yoy_change_pct"],
            vs_5yr_avg=latest["vs_5yr_avg"],
            vs_5yr_avg_pct=latest["vs_5yr_avg_pct"],
            days_of_supply=latest["days_of_supply"],
            weekly_trend=latest["weekly_trend"],
            forecast_4wk=forecast,
        )
```

# tests/unit/oil/test_inventory_forecast.py

def test_yoy_calculation():
    """Test YoY change calculation."""
    # 450 million bbl now vs 430 last year = +20 million
    assert calculate_yoy(450, 430) == 20

def test_days_of_supply():
    """Test days of supply calculation."""
    # 430 million bbl / 16.5 million b/d = ~26 days
    assert abs(calculate_days_of_supply(430, 16.5) - 26) < 1

def test_trend_classification():
    """Test trend classification."""
    assert classify_trend(2.5) == "building"
    assert classify_trend(-3.0) == "drawing"
    assert classify_trend(0.5) == "stable"
