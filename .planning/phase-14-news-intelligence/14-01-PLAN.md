### Key Decisions

- Use `feedparser>=6.0` (RFC-compliant, battle-tested)
- Async HTTP via `httpx` (already in project)
- Poll interval: 60 seconds minimum per-feed (was 30s - risk of ban)
- Jitter: ±10s random offset to avoid thundering herd
- Cache last seen GUID per source to avoid duplicates
- Content hash deduplication: SHA256(title+date) with 24h TTL
- Graceful degradation: if feed fails, exponential backoff (1→2→4→8 min)
- Rate limiting: per-feed state tracks last fetch time

### Rate Limiting Strategy

```python
class RateLimiter:
    """Per-feed rate limiting with exponential backoff."""

    def __init__(self, min_interval: int = 60, max_backoff: int = 480):
        self._last_fetch: dict[str, datetime] = {}
        self._backoff: dict[str, int] = {}  # seconds
        self._min_interval = min_interval
        self._max_backoff = max_backoff

    async def wait_if_needed(self, feed_id: str) -> None:
        """Wait if rate limit not satisfied."""
        if feed_id in self._last_fetch:
            elapsed = (datetime.utcnow() - self._last_fetch[feed_id]).total_seconds()
            wait_time = self._get_interval(feed_id) - elapsed
            if wait_time > 0:
                jitter = random.uniform(-10, 10)
                await asyncio.sleep(wait_time + jitter)

    def record_success(self, feed_id: str) -> None:
        self._last_fetch[feed_id] = datetime.utcnow()
        self._backoff[feed_id] = 0

    def record_failure(self, feed_id: str) -> None:
        current = self._backoff.get(feed_id, 60)
        self._backoff[feed_id] = min(current * 2, self._max_backoff)

    def _get_interval(self, feed_id: str) -> int:
        return max(self._min_interval, self._backoff.get(feed_id, 0))
```

### Deduplication Strategy

```python
import hashlib

class DeduplicationCache:
    """Content-based deduplication with TTL."""

    def __init__(self, ttl_hours: int = 24):
        self._seen: dict[str, datetime] = {}
        self._ttl = timedelta(hours=ttl_hours)

    def is_duplicate(self, item: NewsItem) -> bool:
        """Check if item was seen within TTL."""
        content_hash = self._hash(item)
        if content_hash in self._seen:
            if datetime.utcnow() - self._seen[content_hash] < self._ttl:
                return True
        return False

    def mark_seen(self, item: NewsItem) -> None:
        self._seen[self._hash(item)] = datetime.utcnow()

    def _hash(self, item: NewsItem) -> str:
        content = f"{item.title}|{item.published.isoformat()}"
        return hashlib.sha256(content.encode()).hexdigest()[:16]
```
