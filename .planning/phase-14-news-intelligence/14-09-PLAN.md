### Architecture

```python
class FOMCStatementWatcher:
    """Real-time FOMC statement monitoring with auto-diff.

    Polls Fed RSS every 30 seconds. When new statement detected:
    1. Fetch full statement text
    2. Diff against previous statement
    3. Analyze sentiment shift
    4. Send Discord alert with summary
    """

    def __init__(
        self,
        scraper: FOMCStatementScraper,
        diff_engine: StatementDiffEngine,
        discord_client: DiscordClient,
        poll_interval: int = 30,
    ):
        self._scraper = scraper
        self._diff_engine = diff_engine
        self._discord = discord_client
        self._poll_interval = poll_interval
        self._last_statement_date: date | None = None

    async def start(self) -> None:
        """Start continuous monitoring loop."""
        while True:
            try:
                await self._check_for_new_statement()
            except Exception as e:
                logger.exception("Error in FOMC watcher: %s", e)
            await asyncio.sleep(self._poll_interval)

    async def _check_for_new_statement(self) -> None:
        """Check for new statement and alert if found."""
        latest = await self._scraper.get_latest()

        if self._last_statement_date is None:
            self._last_statement_date = latest.date
            return  # First run, just record

        if latest.date > self._last_statement_date:
            # New statement detected!
            await self._process_new_statement(latest)
            self._last_statement_date = latest.date

    async def _process_new_statement(self, new: FOMCStatement) -> None:
        """Process new statement: diff, analyze, alert."""
        # Fetch previous statement
        previous = await self._scraper.get_by_date(self._last_statement_date)

        # Generate diff
        diff = self._diff_engine.diff(previous.raw_text, new.raw_text)

        # Create and send alert
        embed = self._create_alert_embed(new, diff)
        await self._discord.send_embed_async(embed, "fomc_statement")

    def _create_alert_embed(self, statement: FOMCStatement, diff: StatementDiff) -> DiscordEmbed:
        """Create Discord embed for new FOMC statement."""
        color = {
            "hawkish": 0xFF0000,  # Red
            "neutral": 0xFFFF00,  # Yellow
            "dovish": 0x00FF00,   # Green
        }[diff.change_score.direction]

        return DiscordEmbed(
            title=f"🏛️ New FOMC Statement Released",
            description=f"**Date:** {statement.date}\n**Shift:** {diff.change_score.direction.upper()} ({diff.change_score.magnitude:+.2f})",
            color=color,
            url=str(statement.url),
            fields=[
                {"name": "Key Changes", "value": ", ".join(diff.change_score.key_changes[:5])},
                {"name": "Unchanged Ratio", "value": f"{diff.unchanged_ratio:.0%}"},
            ],
        )
```

### Latency Budget

| Step | Target | Notes |
|------|--------|-------|
| RSS poll | 30s | Configurable interval |
| Detect new entry | <1s | GUID comparison |
| Fetch statement | 2-5s | HTTP request |
| Diff generation | <1s | CPU-bound |
| Sentiment analysis | 1-3s | GPU inference |
| Discord webhook | 0.5-1s | HTTP POST |
| **Total** | <40s | Well under 60s target |

### Key Decisions

- **Poll interval**: 30 seconds (balance latency vs load)
- **RSS source**: Fed press releases feed (fastest updates)
- **Alert rate limit**: 5 minutes (prevent spam on corrections)
- **Fallback**: If RSS fails, check Fed website directly
