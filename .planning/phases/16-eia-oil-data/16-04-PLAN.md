def create_eia_panel() -> dbc.Card:
    """Create EIA oil data dashboard panel."""
    return dbc.Card([
        dbc.CardHeader("EIA Weekly Petroleum"),
        dbc.CardBody([
            dbc.Tabs([
                dbc.Tab(label="Cushing", children=[
                    dcc.Graph(id="cushing-inventory-chart"),
                    html.Div(id="cushing-utilization-badge"),

            ])
        ])
    ])
