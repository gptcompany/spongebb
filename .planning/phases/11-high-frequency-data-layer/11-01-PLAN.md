## Objective

Create a collector for Daily Treasury Statement (DTS) data from the US Treasury FiscalData API to replace weekly TGA data with daily updates.

## Background

Currently Net Liquidity uses FRED's WDTGAL (weekly TGA). The FiscalData API provides same-day TGA closing balance, reducing data lag from 7 days to <1 day.

```python
# src/liquidity/collectors/tga_daily.py
"""TGA Daily collector from US Treasury FiscalData API."""

import logging
from datetime import UTC, datetime, timedelta
from typing import Any

import httpx
import pandas as pd

from liquidity.collectors.base import BaseCollector, CollectorFetchError
from liquidity.collectors.registry import registry
from liquidity.config import Settings, get_settings

logger = logging.getLogger(__name__)

# API Configuration
FISCALDATA_BASE_URL = "https://api.fiscaldata.treasury.gov/services/api/fiscal_service"
ENDPOINT = "v1/accounting/dts/operating_cash_balance"


class TGADailyCollector(BaseCollector[pd.DataFrame]):
    """Collector for daily TGA data from US Treasury FiscalData API.

    Fetches Treasury General Account closing balance from the Daily Treasury
    Statement (DTS). Updates daily by 4PM ET.

    API Docs: https://fiscaldata.treasury.gov/api-documentation/
    """

    def __init__(
        self,
        name: str = "tga_daily",
        settings: Settings | None = None,
        **kwargs: Any,
    ) -> None:
        super().__init__(name=name, settings=settings, **kwargs)
        self._settings = settings or get_settings()
        self._client: httpx.AsyncClient | None = None

    async def _get_client(self) -> httpx.AsyncClient:
        """Get or create HTTP client."""
        if self._client is None or self._client.is_closed:
            self._client = httpx.AsyncClient(
                timeout=30.0,
                headers={"Accept": "application/json"},
            )
        return self._client

    async def collect(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect TGA daily data.

        Args:
            start_date: Start date (default: 90 days ago)
            end_date: End date (default: today)

        Returns:
            DataFrame with columns: timestamp, series_id, source, value, unit
        """
        if start_date is None:
            start_date = datetime.now(UTC) - timedelta(days=90)
        if end_date is None:
            end_date = datetime.now(UTC)

        async def _fetch() -> pd.DataFrame:
            return await self._fetch_tga(start_date, end_date)

        try:
            return await self.fetch_with_retry(_fetch)
        except Exception as e:
            logger.error("TGA daily fetch failed: %s", e)
            raise CollectorFetchError(f"TGA daily fetch failed: {e}") from e

    async def _fetch_tga(
        self,
        start_date: datetime,
        end_date: datetime,
    ) -> pd.DataFrame:
        """Fetch TGA data from FiscalData API."""
        client = await self._get_client()

        params = {
            "fields": "record_date,account_type,open_today_bal,open_month_bal",
            "filter": f"account_type:eq:Treasury General Account (TGA),"
                     f"record_date:gte:{start_date.strftime('%Y-%m-%d')},"
                     f"record_date:lte:{end_date.strftime('%Y-%m-%d')}",
            "sort": "-record_date",
            "page[size]": 500,
            "format": "json",
        }

        url = f"{FISCALDATA_BASE_URL}/{ENDPOINT}"
        response = await client.get(url, params=params)
        response.raise_for_status()

        data = response.json()
        records = data.get("data", [])

        if not records:
            logger.warning("No TGA data returned for date range")
            return pd.DataFrame(
                columns=["timestamp", "series_id", "source", "value", "unit"]
            )

        # Parse records
        rows = []
        for record in records:
            # Use open_today_bal as the closing balance
            # (close_today_bal is null after April 2022)
            value = record.get("open_today_bal")
            if value is not None:
                rows.append({
                    "timestamp": pd.to_datetime(record["record_date"]),
                    "series_id": "TGA_DAILY",
                    "source": "fiscaldata",
                    "value": float(value),  # millions USD
                    "unit": "millions_usd",
                })

        df = pd.DataFrame(rows)
        df = df.sort_values("timestamp").reset_index(drop=True)

        logger.info(
            "Fetched %d TGA daily records: %s to %s",
            len(df),
            df["timestamp"].min() if len(df) > 0 else "N/A",
            df["timestamp"].max() if len(df) > 0 else "N/A",
        )

        return df

    async def close(self) -> None:
        """Close the HTTP client."""
        if self._client is not None and not self._client.is_closed:
            await self._client.aclose()


# Register collector
registry.register("tga_daily", TGADailyCollector)
```

```python
"""Integration tests for TGA Daily collector with real API."""

import pytest
from datetime import UTC, datetime, timedelta

from liquidity.collectors.tga_daily import TGADailyCollector


@pytest.mark.integration
async def test_collect_real_api():
    """Test fetching real TGA data from FiscalData API."""
    collector = TGADailyCollector()

    try:
        # Fetch last 30 days
        start = datetime.now(UTC) - timedelta(days=30)
        df = await collector.collect(start_date=start)

        # Should have data (weekdays only, ~20 records)
        assert len(df) > 10, "Expected at least 10 records for 30 days"

        # Check data structure
        assert "timestamp" in df.columns
        assert "value" in df.columns
        assert df["series_id"].iloc[0] == "TGA_DAILY"

        # TGA should be in reasonable range (500B - 1.5T = 500000 - 1500000 million)
        latest_value = df["value"].iloc[-1]
        assert 100000 < latest_value < 2000000, f"TGA value {latest_value} out of range"

    finally:
        await collector.close()


@pytest.mark.integration
async def test_collect_date_range():
    """Test fetching specific date range."""
    collector = TGADailyCollector()

    try:
        start = datetime(2026, 1, 1, tzinfo=UTC)
        end = datetime(2026, 1, 31, tzinfo=UTC)

        df = await collector.collect(start_date=start, end_date=end)

        assert len(df) > 0
        assert df["timestamp"].min() >= start
        assert df["timestamp"].max() <= end

    finally:
        await collector.close()
```
