```python
# In src/liquidity/analyzers/regime_classifier.py, add:

from liquidity.oil.regime import OilRegimeClassifier, OilRegime

class CombinedRegime(Enum):
    VERY_BULLISH = "very_bullish"
    BULLISH = "bullish"
    NEUTRAL = "neutral"
    BEARISH = "bearish"
    VERY_BEARISH = "very_bearish"

@dataclass
class CombinedRegimeState:
    """Combined liquidity and oil regime state."""
    timestamp: datetime
    liquidity_regime: str  # EXPANSION/NEUTRAL/CONTRACTION
    oil_regime: OilRegime
    combined_regime: CombinedRegime
    confidence: float
    commodity_signal: str  # "long", "short", "neutral"
    drivers: list[str]

class CombinedRegimeAnalyzer:
    """Combines liquidity and oil regime for macro signals."""

    # Regime combination matrix
    REGIME_MATRIX = {
        ("EXPANSION", OilRegime.TIGHT): CombinedRegime.VERY_BULLISH,
        ("EXPANSION", OilRegime.BALANCED): CombinedRegime.BULLISH,
        ("EXPANSION", OilRegime.LOOSE): CombinedRegime.NEUTRAL,
        ("NEUTRAL", OilRegime.TIGHT): CombinedRegime.BULLISH,
        ("NEUTRAL", OilRegime.BALANCED): CombinedRegime.NEUTRAL,
        ("NEUTRAL", OilRegime.LOOSE): CombinedRegime.BEARISH,
        ("CONTRACTION", OilRegime.TIGHT): CombinedRegime.NEUTRAL,
        ("CONTRACTION", OilRegime.BALANCED): CombinedRegime.BEARISH,
        ("CONTRACTION", OilRegime.LOOSE): CombinedRegime.VERY_BEARISH,
    }

    def __init__(
        self,
        liquidity_classifier: RegimeClassifier = None,
        oil_classifier: OilRegimeClassifier = None,
    ):
        self.liquidity = liquidity_classifier or RegimeClassifier()
        self.oil = oil_classifier or OilRegimeClassifier()

    async def get_combined_regime(self) -> CombinedRegimeState:
        """Get combined liquidity-oil regime."""
        # Get individual regimes
        liquidity_state = await self.liquidity.classify_current()
        oil_state = await self.oil.classify()

        # Combine regimes
        combined = self._combine_regimes(
            liquidity_state.regime,
            oil_state.regime,
        )

        # Calculate confidence (average of both)
        confidence = (liquidity_state.confidence + oil_state.confidence) / 2

        # Generate signal
        signal = self._regime_to_signal(combined)

        # Collect drivers
        drivers = [
            f"Liquidity: {liquidity_state.regime}",
            f"Oil: {oil_state.regime.value}",
            *oil_state.drivers,
        ]

        return CombinedRegimeState(
            timestamp=datetime.now(),
            liquidity_regime=liquidity_state.regime,
            oil_regime=oil_state.regime,
            combined_regime=combined,
            confidence=confidence,
            commodity_signal=signal,
            drivers=drivers,
        )

    def _combine_regimes(
        self, liquidity: str, oil: OilRegime
    ) -> CombinedRegime:
        """Combine liquidity and oil regimes."""
        return self.REGIME_MATRIX.get(
            (liquidity, oil),
            CombinedRegime.NEUTRAL,
        )

    def _regime_to_signal(self, regime: CombinedRegime) -> str:
        """Convert combined regime to trading signal."""
        if regime in [CombinedRegime.VERY_BULLISH, CombinedRegime.BULLISH]:
            return "long"
        elif regime in [CombinedRegime.VERY_BEARISH, CombinedRegime.BEARISH]:
            return "short"
        return "neutral"
```

```python
def test_regime_matrix():
    analyzer = CombinedRegimeAnalyzer()

    # EXPANSION + TIGHT = VERY_BULLISH
    result = analyzer._combine_regimes("EXPANSION", OilRegime.TIGHT)
    assert result == CombinedRegime.VERY_BULLISH

    # CONTRACTION + LOOSE = VERY_BEARISH
    result = analyzer._combine_regimes("CONTRACTION", OilRegime.LOOSE)
    assert result == CombinedRegime.VERY_BEARISH

def test_signal_generation():
    analyzer = CombinedRegimeAnalyzer()

    assert analyzer._regime_to_signal(CombinedRegime.VERY_BULLISH) == "long"
    assert analyzer._regime_to_signal(CombinedRegime.NEUTRAL) == "neutral"
    assert analyzer._regime_to_signal(CombinedRegime.VERY_BEARISH) == "short"
```
