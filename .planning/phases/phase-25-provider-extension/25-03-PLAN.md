### T2: Provider discovery test
**File:** `tests/unit/test_provider/test_provider_discovery.py`

```python
def test_provider_object_exists():
    """liquidity_provider is a valid Provider with correct fetcher_dict."""
    from liquidity.openbb_ext.provider import liquidity_provider
    assert liquidity_provider.name == "liquidity"
    assert "LiquidityNetLiquidity" in liquidity_provider.fetcher_dict
    assert "LiquidityGlobalLiquidity" in liquidity_provider.fetcher_dict
    assert "LiquidityStealthQE" in liquidity_provider.fetcher_dict

def test_provider_no_credentials_required():
    """Provider requires no API keys."""
    from liquidity.openbb_ext.provider import liquidity_provider
    assert liquidity_provider.credentials == []
```

### T3: Integration test for `obb.liquidity.*` namespace
**File:** `tests/integration/test_openbb_provider.py`

```python
@pytest.mark.integration
def test_openbb_discovers_liquidity_provider():
    """After editable install, OpenBB SDK should see liquidity provider."""
    import subprocess
    result = subprocess.run(
        ["python", "-c", "from openbb import obb; print(hasattr(obb, 'liquidity'))"],
        capture_output=True, text=True, timeout=30,
    )
    assert result.returncode == 0
    assert "True" in result.stdout
```

Note: Integration test requires `uv pip install -e .` beforehand. Mark with `@pytest.mark.integration`.
