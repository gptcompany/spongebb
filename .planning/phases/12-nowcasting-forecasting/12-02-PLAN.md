# Plan 12-02: PBoC Balance Sheet Estimator

## Overview

Build a MIDAS regression model to estimate PBoC total assets 2-3 weeks before the official monthly release, using high-frequency Chinese money market indicators.

## Goals

- Estimate PBoC balance sheet 15-20 days before official release
- Reduce effective data lag from 30-45 days to ~10 days
- MAPE < 5% vs official monthly releases
- Provide uncertainty bounds

## Approach

Use MIDAS (Mixed-Data Sampling) regression with polynomial distributed lags to preserve high-frequency information from daily SHIBOR and weekly DR007.

### MIDAS Model

```
PBoC_monthly = α + Σ β_i * w(i) * SHIBOR_{t-i} + Σ γ_j * DR007_{t-j} + δ * CNY_CNH_spread + ε

Where:
- w(i) = exp(-(i²)/30)  # Almon polynomial weighting
- i = 1..30 (daily lags)
- j = 1..4 (weekly lags)
```

### Task 2: Implement Feature Engineering
**File**: `src/liquidity/nowcasting/midas/features.py`

```python
import numpy as np
import pandas as pd
from typing import List, Tuple

class MIDASFeatures:
    """
    Feature engineering for MIDAS regression.

    Key insight: Don't average HF data to LF. Instead, use polynomial
    distributed lags to preserve daily information in monthly model.
    """

    @staticmethod
    def almon_weights(n_lags: int, decay: float = 30.0) -> np.ndarray:
        """
        Almon polynomial weighting scheme.

        Gives more weight to recent observations, decaying exponentially.
        """
        lags = np.arange(1, n_lags + 1)
        weights = np.exp(-(lags ** 2) / decay)
        return weights / weights.sum()  # Normalize

    @staticmethod
    def create_midas_features(
        daily_series: pd.Series,
        weekly_series: pd.Series,
        n_daily_lags: int = 30,
        n_weekly_lags: int = 4
    ) -> Tuple[pd.DataFrame, List[str]]:
        """
        Create MIDAS feature matrix.

        Returns:
            features: DataFrame with weighted lag features
            feature_names: List of feature column names
        """
        features = []
        names = []

        # Daily lags with Almon weighting
        daily_weights = MIDASFeatures.almon_weights(n_daily_lags)
        for i in range(1, n_daily_lags + 1):
            feat = daily_series.shift(i) * daily_weights[i - 1]
            features.append(feat)
            names.append(f"daily_lag_{i}")

        # Weekly lags (converted to daily index)
        for j in range(1, n_weekly_lags + 1):
            lag_days = j * 5  # Approximate weekly to daily
            feat = weekly_series.shift(lag_days)
            features.append(feat)
            names.append(f"weekly_lag_{j}")

        return pd.concat(features, axis=1), names
```

```python
from sklearn.linear_model import Ridge
from sklearn.preprocessing import StandardScaler
from sklearn.model_selection import TimeSeriesSplit
import numpy as np
import pandas as pd
from dataclasses import dataclass
from typing import Optional, Tuple

@dataclass
class PBoCEstimate:
    """PBoC balance sheet estimate."""
    timestamp: pd.Timestamp
    estimate: float              # Point estimate (trillion CNY)
    std: float                   # Standard error
    ci_lower: float              # 95% CI lower
    ci_upper: float              # 95% CI upper
    days_ahead: int              # Days before official release
    confidence: float            # Model confidence (0-1)

class PBoCEstimator:
    """
    MIDAS regression estimator for PBoC balance sheet.

    Uses SHIBOR (daily), DR007 (weekly), and CNY-CNH spread to
    estimate monthly PBoC total assets before official release.
    """

    def __init__(self, alpha: float = 10.0):
        self.alpha = alpha  # Ridge regularization
        self.model = Ridge(alpha=alpha)
        self.scaler = StandardScaler()
        self.feature_builder = MIDASFeatures()
        self._is_fitted = False

    def fit(
        self,
        shibor_daily: pd.Series,
        dr007_weekly: pd.Series,
        cny_cnh_spread: pd.Series,
        pboc_monthly: pd.Series
    ) -> "PBoCEstimator":
        """
        Fit MIDAS regression on historical data.

        Args:
            shibor_daily: Daily SHIBOR overnight rate
            dr007_weekly: Weekly DR007 repo rate
            cny_cnh_spread: CNY-CNH spot spread
            pboc_monthly: Official PBoC total assets (target)
        """
        # Create features
        X, self.feature_names = self._create_features(
            shibor_daily, dr007_weekly, cny_cnh_spread
        )

        # Align with monthly target (resample to month-end)
        X_monthly = X.resample('ME').last()
        y = pboc_monthly.reindex(X_monthly.index)

        # Drop NaN
        mask = ~(X_monthly.isna().any(axis=1) | y.isna())
        X_clean = X_monthly[mask]
        y_clean = y[mask]

        # Scale features
        X_scaled = self.scaler.fit_transform(X_clean)

        # Fit Ridge regression
        self.model.fit(X_scaled, y_clean)
        self._is_fitted = True

        return self

    def estimate(
        self,
        shibor_daily: pd.Series,
        dr007_weekly: pd.Series,
        cny_cnh_spread: pd.Series
    ) -> PBoCEstimate:
        """
        Generate PBoC estimate using latest available data.

        Returns estimate for current month before official release.
        """
        if not self._is_fitted:
            raise ValueError("Model not fitted. Call fit() first.")

        # Create features from latest data
        X, _ = self._create_features(shibor_daily, dr007_weekly, cny_cnh_spread)
        X_latest = X.iloc[[-1]]  # Most recent

        # Scale and predict
        X_scaled = self.scaler.transform(X_latest)
        estimate = self.model.predict(X_scaled)[0]

        # Bootstrap for uncertainty (simplified)
        std = self._estimate_uncertainty(X_scaled)

        # Days until official release (typically 15th of next month)
        today = pd.Timestamp.now()
        next_release = (today + pd.offsets.MonthEnd(1) + pd.Timedelta(days=15))
        days_ahead = (next_release - today).days

        return PBoCEstimate(
            timestamp=today,
            estimate=estimate,
            std=std,
            ci_lower=estimate - 1.96 * std,
            ci_upper=estimate + 1.96 * std,
            days_ahead=days_ahead,
            confidence=self._calculate_confidence(X_scaled)
        )

    def _create_features(
        self,
        shibor: pd.Series,
        dr007: pd.Series,
        spread: pd.Series
    ) -> Tuple[pd.DataFrame, list]:
        """Create MIDAS feature matrix."""
        # Align all series to common daily index
        combined = pd.DataFrame({
            'shibor': shibor,
            'dr007': dr007,
            'spread': spread
        }).ffill()

        return self.feature_builder.create_midas_features(
            combined['shibor'],
            combined['dr007'],
            n_daily_lags=30,
            n_weekly_lags=4
        )

    def _estimate_uncertainty(self, X: np.ndarray) -> float:
        """Estimate prediction uncertainty via residual std."""
        # Simplified: use training residual std
        # Full implementation would use bootstrap
        return 0.5  # Placeholder: 0.5 trillion CNY
```

### Task 4: Cross-Validation and Hyperparameter Tuning
**File**: `src/liquidity/nowcasting/midas/pboc_estimator.py` (extend)

```python
def tune_alpha(self, X: pd.DataFrame, y: pd.Series) -> float:
    """
    Tune Ridge alpha via time-series cross-validation.

    Uses expanding window CV to respect temporal order.
    """
    tscv = TimeSeriesSplit(n_splits=5)
    alphas = [0.1, 1.0, 10.0, 100.0, 1000.0]

    best_alpha = 10.0
    best_score = -np.inf

    for alpha in alphas:
        model = Ridge(alpha=alpha)
        scores = []

        for train_idx, val_idx in tscv.split(X):
            X_train, X_val = X.iloc[train_idx], X.iloc[val_idx]
            y_train, y_val = y.iloc[train_idx], y.iloc[val_idx]

            model.fit(X_train, y_train)
            score = model.score(X_val, y_val)
            scores.append(score)

        mean_score = np.mean(scores)
        if mean_score > best_score:
            best_score = mean_score
            best_alpha = alpha

    return best_alpha
```

```python
"""
Backtest PBoC estimator on 2020-2025 data.

Usage: uv run python scripts/backtest_pboc.py
"""

async def main():
    # Load historical data
    shibor = await load_shibor_history()
    dr007 = await load_dr007_history()
    pboc_official = await load_pboc_history()

    # Walk-forward backtest
    results = []
    for test_date in pd.date_range('2022-01-01', '2025-12-31', freq='ME'):
        train_end = test_date - pd.Timedelta(days=45)

        estimator = PBoCEstimator()
        estimator.fit(
            shibor.loc[:train_end],
            dr007.loc[:train_end],
            pd.Series(),  # Spread
            pboc_official.loc[:train_end]
        )

        estimate = estimator.estimate(
            shibor.loc[:test_date],
            dr007.loc[:test_date],
            pd.Series()
        )

        official = pboc_official.loc[test_date]
        error_pct = abs(estimate.estimate - official) / official * 100

        results.append({
            'date': test_date,
            'estimate': estimate.estimate,
            'official': official,
            'error_pct': error_pct
        })

    df = pd.DataFrame(results)
    print(f"MAPE: {df['error_pct'].mean():.2f}%")
    print(f"Max error: {df['error_pct'].max():.2f}%")
```
