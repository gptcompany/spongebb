### Key Decisions

- **difflib.SequenceMatcher**: Industry standard, word-level
- **Score aggregation**: Sum of weighted keywords in changes
- **Threshold**: ±0.2 for hawkish/dovish classification
- **Key changes**: Top 5 most impactful word additions/deletions
- **Semantic layer**: Detect policy-relevant phrase changes (not just words)

### Semantic Phrase Detection

Word-level diff alone is insufficient. "Patient" → "data-dependent" is a major policy shift
but appears as simple add/delete at word level. We add semantic phrase detection:

```python
# Policy-critical phrases to track
POLICY_PHRASES = {
    # Forward guidance phrases
    "patient": "dovish",
    "data-dependent": "neutral",
    "gradual": "dovish",
    "measured": "neutral",
    "appropriate": "neutral",
    "as warranted": "conditional",

    # Rate guidance
    "considerable time": "dovish",
    "extended period": "dovish",
    "some time": "dovish",
    "near-term": "hawkish",

    # Economic assessment
    "solid": "hawkish",
    "moderate": "neutral",
    "modest": "dovish",
    "subdued": "dovish",
    "robust": "hawkish",
}

class SemanticDiffLayer:
    """Detect policy-relevant phrase changes."""

    def analyze_phrase_shifts(self, old: str, new: str) -> list[PhraseShift]:
        """Find policy phrases that changed between statements."""
        old_phrases = self._extract_policy_phrases(old)
        new_phrases = self._extract_policy_phrases(new)

        shifts = []
        # Detect additions
        for phrase in new_phrases - old_phrases:
            shifts.append(PhraseShift(
                phrase=phrase,
                change="added",
                policy_signal=POLICY_PHRASES.get(phrase, "unknown"),
            ))

        # Detect removals
        for phrase in old_phrases - new_phrases:
            shifts.append(PhraseShift(
                phrase=phrase,
                change="removed",
                policy_signal=POLICY_PHRASES.get(phrase, "unknown"),
            ))

        return shifts

    def _extract_policy_phrases(self, text: str) -> set[str]:
        """Extract all policy phrases from text."""
        text_lower = text.lower()
        return {p for p in POLICY_PHRASES if p in text_lower}

@dataclass
class PhraseShift:
    phrase: str
    change: Literal["added", "removed"]
    policy_signal: str
    significance: float = 0.0  # Computed based on historical impact
```
