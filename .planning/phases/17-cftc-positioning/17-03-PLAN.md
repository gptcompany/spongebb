```python
from enum import Enum
from dataclasses import dataclass
from datetime import datetime


class PositioningAlertType(Enum):
    """Types of positioning alerts."""
    SPEC_EXTREME_LONG = "spec_extreme_long"       # Speculators crowded long
    SPEC_EXTREME_SHORT = "spec_extreme_short"     # Speculators crowded short
    COMM_EXTREME_LONG = "comm_extreme_long"       # Commercials bullish
    COMM_EXTREME_SHORT = "comm_extreme_short"     # Commercials bearish
    COMM_SPEC_DIVERGENCE = "comm_spec_divergence" # Smart vs dumb money disagree


@dataclass
class PositioningAlert:
    """Container for a positioning alert."""
    alert_type: PositioningAlertType
    commodity: str
    timestamp: datetime
    spec_percentile: float
    comm_percentile: float
    message: str
    severity: str  # "warning" or "critical"

    def to_discord_embed(self) -> dict:
        """Format alert for Discord webhook."""
        color_map = {
            "warning": 0xFFA500,   # Orange
            "critical": 0xFF0000,  # Red
        }

        emoji_map = {
            PositioningAlertType.SPEC_EXTREME_LONG: "📈",
            PositioningAlertType.SPEC_EXTREME_SHORT: "📉",
            PositioningAlertType.COMM_EXTREME_LONG: "🔵",
            PositioningAlertType.COMM_EXTREME_SHORT: "🔴",
            PositioningAlertType.COMM_SPEC_DIVERGENCE: "⚡",
        }

        return {
            "title": f"{emoji_map.get(self.alert_type, '📊')} {self.commodity} Positioning Alert",
            "description": self.message,
            "color": color_map.get(self.severity, 0x808080),
            "fields": [
                {"name": "Speculator Percentile", "value": f"{self.spec_percentile:.1f}%", "inline": True},
                {"name": "Commercial Percentile", "value": f"{self.comm_percentile:.1f}%", "inline": True},
                {"name": "Alert Type", "value": self.alert_type.value, "inline": True},
            ],
            "timestamp": self.timestamp.isoformat(),
        }
```

### 2. Create Positioning Alert Engine

```python
class PositioningAlertEngine:
    """Engine for generating positioning alerts."""

    THRESHOLDS = {
        "extreme_high": 90,
        "extreme_low": 10,
        "critical_high": 95,
        "critical_low": 5,
    }

    def __init__(self, discord_webhook_url: str | None = None):
        self.webhook_url = discord_webhook_url
        self._last_alerts: dict[str, datetime] = {}  # Dedup by commodity

    def check_extremes(
        self,
        commodity: str,
        spec_percentile: float,
        comm_percentile: float,
        timestamp: datetime,
    ) -> list[PositioningAlert]:
        """Check for extreme positioning and generate alerts."""
        alerts = []

        # Speculator extremes
        if spec_percentile >= self.THRESHOLDS["critical_high"]:
            alerts.append(self._create_alert(
                PositioningAlertType.SPEC_EXTREME_LONG,
                commodity, timestamp, spec_percentile, comm_percentile,
                f"Speculators at {spec_percentile:.1f}th percentile - CROWDED LONG",
                "critical"
            ))
        elif spec_percentile >= self.THRESHOLDS["extreme_high"]:
            alerts.append(self._create_alert(
                PositioningAlertType.SPEC_EXTREME_LONG,
                commodity, timestamp, spec_percentile, comm_percentile,
                f"Speculators at {spec_percentile:.1f}th percentile - elevated long",
                "warning"
            ))
        elif spec_percentile <= self.THRESHOLDS["critical_low"]:
            alerts.append(self._create_alert(
                PositioningAlertType.SPEC_EXTREME_SHORT,
                commodity, timestamp, spec_percentile, comm_percentile,
                f"Speculators at {spec_percentile:.1f}th percentile - CROWDED SHORT",
                "critical"
            ))
        elif spec_percentile <= self.THRESHOLDS["extreme_low"]:
            alerts.append(self._create_alert(
                PositioningAlertType.SPEC_EXTREME_SHORT,
                commodity, timestamp, spec_percentile, comm_percentile,
                f"Speculators at {spec_percentile:.1f}th percentile - elevated short",
                "warning"
            ))

        # Commercial extremes (less frequent, higher signal)
        if comm_percentile >= self.THRESHOLDS["extreme_high"]:
            alerts.append(self._create_alert(
                PositioningAlertType.COMM_EXTREME_LONG,
                commodity, timestamp, spec_percentile, comm_percentile,
                f"Commercials at {comm_percentile:.1f}th percentile - SMART MONEY BULLISH",
                "critical"
            ))
        elif comm_percentile <= self.THRESHOLDS["extreme_low"]:
            alerts.append(self._create_alert(
                PositioningAlertType.COMM_EXTREME_SHORT,
                commodity, timestamp, spec_percentile, comm_percentile,
                f"Commercials at {comm_percentile:.1f}th percentile - SMART MONEY BEARISH",
                "critical"
            ))

        # Divergence (specs bullish, comms bearish or vice versa)
        if spec_percentile >= 70 and comm_percentile <= 30:
            alerts.append(self._create_alert(
                PositioningAlertType.COMM_SPEC_DIVERGENCE,
                commodity, timestamp, spec_percentile, comm_percentile,
                "Smart money vs speculators DIVERGING - commercials bearish while specs bullish",
                "warning"
            ))
        elif spec_percentile <= 30 and comm_percentile >= 70:
            alerts.append(self._create_alert(
                PositioningAlertType.COMM_SPEC_DIVERGENCE,
                commodity, timestamp, spec_percentile, comm_percentile,
                "Smart money vs speculators DIVERGING - commercials bullish while specs bearish",
                "warning"
            ))

        return alerts

    async def send_alerts(self, alerts: list[PositioningAlert]) -> None:
        """Send alerts to Discord webhook."""
        if not self.webhook_url or not alerts:
            return

        embeds = [alert.to_discord_embed() for alert in alerts]

        async with httpx.AsyncClient() as client:
            await client.post(
                self.webhook_url,
                json={"embeds": embeds}
            )
```
