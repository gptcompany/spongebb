## Objective

Implement SHIBOR and DR007 collectors using akshare library to provide daily proxies for PBoC policy stance, addressing the 1-month lag in official PBoC balance sheet data.

## Background

The Global Liquidity Index requires PBoC data, but official balance sheet releases have ~1 month lag. SHIBOR (Shanghai Interbank Offered Rate) and DR007 (7-day repo rate) are daily indicators that reflect PBoC policy stance and can be used for nowcasting.

```python
# src/liquidity/collectors/china_rates.py
"""China interbank rates collector using akshare.

Provides SHIBOR and DR007 as daily proxies for PBoC policy stance.
"""

import asyncio
import logging
from datetime import UTC, datetime, timedelta
from typing import Any

import pandas as pd

from liquidity.collectors.base import BaseCollector, CollectorFetchError
from liquidity.collectors.registry import registry
from liquidity.config import Settings, get_settings

logger = logging.getLogger(__name__)

# Tenor mapping for SHIBOR
SHIBOR_TENORS = {
    "O/N": "overnight",
    "1W": "1_week",
    "2W": "2_weeks",
    "1M": "1_month",
    "3M": "3_months",
    "6M": "6_months",
    "9M": "9_months",
    "1Y": "1_year",
}


class ChinaRatesCollector(BaseCollector[pd.DataFrame]):
    """Collector for China interbank rates using akshare.

    Provides:
    - SHIBOR (Shanghai Interbank Offered Rate) - multiple tenors
    - DR007 (7-day repo rate) - PBoC's de facto policy target

    These rates serve as high-frequency proxies for PBoC policy stance,
    useful for nowcasting when official balance sheet data lags by ~1 month.

    Requires: pip install akshare>=1.18.0
    """

    def __init__(
        self,
        name: str = "china_rates",
        settings: Settings | None = None,
        **kwargs: Any,
    ) -> None:
        super().__init__(name=name, settings=settings, **kwargs)
        self._settings = settings or get_settings()

    async def collect_shibor(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect SHIBOR rates for all tenors.

        Args:
            start_date: Start date (default: 90 days ago)
            end_date: End date (default: today)

        Returns:
            DataFrame with columns: timestamp, series_id, source, value, unit
        """
        if start_date is None:
            start_date = datetime.now(UTC) - timedelta(days=90)
        if end_date is None:
            end_date = datetime.now(UTC)

        async def _fetch() -> pd.DataFrame:
            return await asyncio.to_thread(
                self._fetch_shibor_sync, start_date, end_date
            )

        try:
            return await self.fetch_with_retry(_fetch, breaker_name="china_shibor")
        except Exception as e:
            logger.error("SHIBOR fetch failed: %s", e)
            raise CollectorFetchError(f"SHIBOR fetch failed: {e}") from e

    def _fetch_shibor_sync(
        self,
        start_date: datetime,
        end_date: datetime,
    ) -> pd.DataFrame:
        """Synchronous SHIBOR fetch using akshare."""
        try:
            import akshare as ak
        except ImportError as e:
            raise CollectorFetchError(
                "akshare not installed. Run: pip install akshare>=1.18.0"
            ) from e

        logger.info("Fetching SHIBOR rates")

        # Get SHIBOR all tenors
        df_raw = ak.macro_china_shibor_all()

        if df_raw.empty:
            logger.warning("No SHIBOR data returned")
            return pd.DataFrame(
                columns=["timestamp", "series_id", "source", "value", "unit"]
            )

        # Parse the data
        # Columns: 日期, O/N-定价, O/N-涨跌幅, 1W-定价, 1W-涨跌幅, ...
        rows = []

        for _, row in df_raw.iterrows():
            date = pd.to_datetime(row["日期"])

            # Filter by date range
            if date < start_date or date > end_date:
                continue

            # Extract each tenor's rate
            for tenor, tenor_name in SHIBOR_TENORS.items():
                col_name = f"{tenor}-定价"
                if col_name in row.index and pd.notna(row[col_name]):
                    rows.append({
                        "timestamp": date,
                        "series_id": f"SHIBOR_{tenor_name.upper()}",
                        "source": "akshare",
                        "value": float(row[col_name]),
                        "unit": "percent",
                    })

        df = pd.DataFrame(rows)
        df = df.sort_values("timestamp").reset_index(drop=True)

        logger.info("Fetched %d SHIBOR records", len(df))
        return df

    async def collect_dr007(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect DR007 (7-day repo rate).

        DR007 is the PBoC's de-facto policy rate target.
        It reflects interbank funding conditions more accurately than SHIBOR.

        Args:
            start_date: Start date (default: 90 days ago)
            end_date: End date (default: today)

        Returns:
            DataFrame with DR007 rate data.
        """
        if start_date is None:
            start_date = datetime.now(UTC) - timedelta(days=90)
        if end_date is None:
            end_date = datetime.now(UTC)

        async def _fetch() -> pd.DataFrame:
            return await asyncio.to_thread(
                self._fetch_dr007_sync, start_date, end_date
            )

        try:
            return await self.fetch_with_retry(_fetch, breaker_name="china_dr007")
        except Exception as e:
            logger.error("DR007 fetch failed: %s", e)
            raise CollectorFetchError(f"DR007 fetch failed: {e}") from e

    def _fetch_dr007_sync(
        self,
        start_date: datetime,
        end_date: datetime,
    ) -> pd.DataFrame:
        """Synchronous DR007 fetch using akshare."""
        try:
            import akshare as ak
        except ImportError as e:
            raise CollectorFetchError(
                "akshare not installed. Run: pip install akshare>=1.18.0"
            ) from e

        logger.info("Fetching DR007 rates")

        # Try rate_interbank function
        try:
            df_raw = ak.rate_interbank()
        except Exception:
            # Fallback: try to extract from SHIBOR
            logger.warning("rate_interbank failed, trying SHIBOR data")
            return self._dr007_from_shibor(start_date, end_date)

        if df_raw.empty:
            logger.warning("No DR007 data returned")
            return pd.DataFrame(
                columns=["timestamp", "series_id", "source", "value", "unit"]
            )

        rows = []
        for _, row in df_raw.iterrows():
            # Look for DR007 column
            date = pd.to_datetime(row.get("日期", row.get("date")))

            if date < start_date or date > end_date:
                continue

            # Column might be DR007, DR007加权, or similar
            dr007_value = None
            for col in df_raw.columns:
                if "DR007" in col.upper() and "加权" in col:
                    dr007_value = row.get(col)
                    break
                elif "DR007" in col.upper():
                    dr007_value = row.get(col)

            if dr007_value is not None and pd.notna(dr007_value):
                rows.append({
                    "timestamp": date,
                    "series_id": "DR007",
                    "source": "akshare",
                    "value": float(dr007_value),
                    "unit": "percent",
                })

        df = pd.DataFrame(rows)
        df = df.sort_values("timestamp").reset_index(drop=True)

        logger.info("Fetched %d DR007 records", len(df))
        return df

    def _dr007_from_shibor(
        self,
        start_date: datetime,
        end_date: datetime,
    ) -> pd.DataFrame:
        """Extract DR007 proxy from SHIBOR 1W rate."""
        # SHIBOR 1W is close to DR007
        shibor_df = self._fetch_shibor_sync(start_date, end_date)

        # Filter for 1W rate
        df = shibor_df[shibor_df["series_id"] == "SHIBOR_1_WEEK"].copy()
        df["series_id"] = "DR007_PROXY"

        logger.info("Using SHIBOR 1W as DR007 proxy")
        return df

    async def collect(
        self,
        data_type: str = "shibor",
        **kwargs: Any,
    ) -> pd.DataFrame:
        """Generic collect method.

        Args:
            data_type: "shibor" or "dr007"
            **kwargs: Passed to specific collector
        """
        if data_type == "shibor":
            return await self.collect_shibor(**kwargs)
        elif data_type == "dr007":
            return await self.collect_dr007(**kwargs)
        else:
            raise ValueError(f"Unknown data_type: {data_type}")

    async def collect_all(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect all China rate data (SHIBOR + DR007).

        Returns:
            Combined DataFrame with all rate series.
        """
        shibor_df = await self.collect_shibor(start_date, end_date)
        dr007_df = await self.collect_dr007(start_date, end_date)

        return pd.concat([shibor_df, dr007_df], ignore_index=True)


# Register collector
registry.register("china_rates", ChinaRatesCollector)
```

```python
"""Unit tests for China rates collector."""

import pandas as pd
import pytest
from unittest.mock import patch, MagicMock

from liquidity.collectors.china_rates import ChinaRatesCollector


@pytest.fixture
def mock_shibor_data():
    """Sample akshare SHIBOR response."""
    return pd.DataFrame({
        "日期": ["2026-02-04", "2026-02-03"],
        "O/N-定价": [1.325, 1.320],
        "O/N-涨跌幅": [0.005, -0.002],
        "1W-定价": [1.650, 1.645],
        "1W-涨跌幅": [0.005, 0.003],
    })


@pytest.mark.unit
async def test_collect_shibor_returns_dataframe(mock_shibor_data):
    """Test SHIBOR collection returns properly formatted DataFrame."""
    collector = ChinaRatesCollector()

    with patch("akshare.macro_china_shibor_all", return_value=mock_shibor_data):
        result = await collector.collect_shibor()

        assert isinstance(result, pd.DataFrame)
        assert "timestamp" in result.columns
        assert "value" in result.columns
        assert any("SHIBOR" in s for s in result["series_id"].values)


@pytest.mark.unit
async def test_collect_dr007_fallback():
    """Test DR007 falls back to SHIBOR proxy if rate_interbank fails."""
    collector = ChinaRatesCollector()

    mock_shibor = pd.DataFrame({
        "日期": ["2026-02-04"],
        "O/N-定价": [1.325],
        "1W-定价": [1.650],
    })

    with patch("akshare.rate_interbank", side_effect=Exception("API error")):
        with patch("akshare.macro_china_shibor_all", return_value=mock_shibor):
            result = await collector.collect_dr007()

            # Should fallback to SHIBOR proxy
            assert len(result) >= 0  # May be empty if filtered


@pytest.mark.unit
async def test_collect_handles_empty_response():
    """Test handling of empty API response."""
    collector = ChinaRatesCollector()

    with patch("akshare.macro_china_shibor_all", return_value=pd.DataFrame()):
        result = await collector.collect_shibor()

        assert len(result) == 0
```
