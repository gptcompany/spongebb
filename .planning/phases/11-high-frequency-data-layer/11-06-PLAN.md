### Task 2: Create ConsumerCreditCollector wrapper

```python
# src/liquidity/collectors/consumer_credit.py
"""Consumer credit and spending collector.

Wraps FREDCollector for consumer-focused series.
"""

import logging
from datetime import UTC, datetime, timedelta
from typing import Any

import pandas as pd

from liquidity.collectors.fred import FredCollector
from liquidity.collectors.registry import registry
from liquidity.config import Settings, get_settings

logger = logging.getLogger(__name__)

# Consumer series groups
CONSUMER_SERIES = {
    "spending": ["RSAFS", "RRSFS", "PCE"],
    "credit": ["TOTALSL", "CCLACBW027SBOG", "CLSACBW027SBOG"],
    "sentiment": ["UMCSENT"],
}

# All consumer series
ALL_CONSUMER_SERIES = [
    s for group in CONSUMER_SERIES.values() for s in group
]


class ConsumerCreditCollector:
    """Convenience wrapper for consumer credit/spending data.

    Uses FREDCollector under the hood but provides consumer-focused
    methods and data interpretation.
    """

    def __init__(
        self,
        settings: Settings | None = None,
    ) -> None:
        self._settings = settings or get_settings()
        self._fred = FredCollector(name="consumer_credit", settings=settings)

    async def collect_spending(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect consumer spending data.

        Series:
        - RSAFS: Retail Sales Total
        - RRSFS: Retail Sales ex Autos
        - PCE: Personal Consumption Expenditures

        Returns:
            DataFrame with spending series.
        """
        if start_date is None:
            start_date = datetime.now(UTC) - timedelta(days=365)

        return await self._fred.collect(
            CONSUMER_SERIES["spending"],
            start_date,
            end_date,
        )

    async def collect_credit(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect consumer credit data.

        Series:
        - TOTALSL: Total Consumer Credit (monthly)
        - CCLACBW027SBOG: Credit Card Loans at Banks (weekly)
        - CLSACBW027SBOG: Consumer Loans at Banks (weekly)

        Returns:
            DataFrame with credit series.
        """
        if start_date is None:
            start_date = datetime.now(UTC) - timedelta(days=365)

        return await self._fred.collect(
            CONSUMER_SERIES["credit"],
            start_date,
            end_date,
        )

    async def collect_sentiment(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect consumer sentiment data.

        Series:
        - UMCSENT: U of Michigan Consumer Sentiment Index

        Returns:
            DataFrame with sentiment series.
        """
        if start_date is None:
            start_date = datetime.now(UTC) - timedelta(days=365)

        return await self._fred.collect(
            CONSUMER_SERIES["sentiment"],
            start_date,
            end_date,
        )

    async def collect_all(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect all consumer series.

        Returns:
            DataFrame with all consumer series.
        """
        if start_date is None:
            start_date = datetime.now(UTC) - timedelta(days=365)

        return await self._fred.collect(
            ALL_CONSUMER_SERIES,
            start_date,
            end_date,
        )

    async def collect_weekly_hf(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect weekly high-frequency credit proxies.

        Provides more timely data than monthly series.

        Returns:
            DataFrame with weekly bank credit data.
        """
        if start_date is None:
            start_date = datetime.now(UTC) - timedelta(days=90)

        return await self._fred.collect(
            ["CCLACBW027SBOG", "CLSACBW027SBOG"],
            start_date,
            end_date,
        )

    @staticmethod
    def calculate_yoy_growth(df: pd.DataFrame) -> pd.DataFrame:
        """Calculate year-over-year growth rates.

        Args:
            df: DataFrame with timestamp, series_id, value columns.

        Returns:
            DataFrame with YoY growth rates added.
        """
        result_rows = []

        for series_id in df["series_id"].unique():
            series_df = df[df["series_id"] == series_id].copy()
            series_df = series_df.sort_values("timestamp")

            # Calculate YoY growth
            series_df["value_1y_ago"] = series_df["value"].shift(12)  # 12 months
            series_df["yoy_growth"] = (
                (series_df["value"] - series_df["value_1y_ago"])
                / series_df["value_1y_ago"] * 100
            )

            result_rows.append(series_df[["timestamp", "series_id", "value", "yoy_growth"]])

        return pd.concat(result_rows, ignore_index=True)

    @staticmethod
    def interpret_sentiment(value: float) -> str:
        """Interpret consumer sentiment level.

        Historical context:
        - > 100: Very optimistic
        - 80-100: Optimistic
        - 60-80: Neutral/cautious
        - < 60: Pessimistic (recession territory)

        Args:
            value: UMCSENT index value.

        Returns:
            Interpretation string.
        """
        if value > 100:
            return "very_optimistic"
        elif value > 80:
            return "optimistic"
        elif value > 60:
            return "neutral"
        else:
            return "pessimistic"


# Register convenience function to create collector
def create_consumer_credit_collector(**kwargs: Any) -> ConsumerCreditCollector:
    return ConsumerCreditCollector(**kwargs)


registry.register("consumer_credit", create_consumer_credit_collector)
```

# tests/integration/collectors/test_consumer_credit_integration.py
"""Integration tests for consumer credit collector with real FRED API."""

import pytest
from datetime import UTC, datetime, timedelta

from liquidity.collectors.consumer_credit import ConsumerCreditCollector


@pytest.mark.integration
async def test_collect_spending_real_api():
    """Test fetching real spending data from FRED."""
    collector = ConsumerCreditCollector()

    start = datetime.now(UTC) - timedelta(days=365)
    df = await collector.collect_spending(start_date=start)

    # Should have data for all spending series
    series_ids = df["series_id"].unique()
    assert "RSAFS" in series_ids or len(series_ids) > 0

    # Retail sales should be in billions range
    rsafs = df[df["series_id"] == "RSAFS"]
    if len(rsafs) > 0:
        # RSAFS in millions, should be 500,000-1,000,000 (=$500B-$1T)
        latest = rsafs["value"].iloc[-1]
        assert 100000 < latest < 2000000


@pytest.mark.integration
async def test_collect_sentiment_real_api():
    """Test fetching real sentiment data from FRED."""
    collector = ConsumerCreditCollector()

    start = datetime.now(UTC) - timedelta(days=365)
    df = await collector.collect_sentiment(start_date=start)

    assert len(df) > 0

    # UMCSENT typically 50-110
    umcsent = df[df["series_id"] == "UMCSENT"]
    if len(umcsent) > 0:
        latest = umcsent["value"].iloc[-1]
        assert 30 < latest < 130

        # Test interpretation
        interpretation = ConsumerCreditCollector.interpret_sentiment(latest)
        assert interpretation in ["very_optimistic", "optimistic", "neutral", "pessimistic"]


@pytest.mark.integration
async def test_collect_weekly_hf_real_api():
    """Test fetching weekly high-frequency data."""
    collector = ConsumerCreditCollector()

    start = datetime.now(UTC) - timedelta(days=90)
    df = await collector.collect_weekly_hf(start_date=start)

    # Should have weekly data (more frequent than monthly)
    if len(df) > 0:
        series_ids = df["series_id"].unique()
        # At least one of the weekly series
        assert "CCLACBW027SBOG" in series_ids or "CLSACBW027SBOG" in series_ids
