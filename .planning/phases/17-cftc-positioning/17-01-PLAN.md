class CFTCCOTCollector(BaseCollector[pd.DataFrame]):

### 2. Implement Async Fetch with SoQL

- Use `httpx.AsyncClient` for async requests with 30s timeout
- No authentication required (public API)
- SoQL query parameters for filtering (parameterized, no f-string injection)
- Retry logic via `BaseCollector.fetch_with_retry()`
- Validate commodity exists in COMMODITY_MAP
- Handle pagination for large requests

```python
async def _get_client(self) -> httpx.AsyncClient:
    """Create HTTP client with timeout."""
    return httpx.AsyncClient(timeout=httpx.Timeout(30.0))

async def _fetch_commodity(
    self,
    client: httpx.AsyncClient,
    commodity: str,
    start_date: date | None,
    end_date: date | None,
    limit: int = 1000,
) -> pd.DataFrame:
    """Fetch COT data for a single commodity."""
    # Validate commodity exists
    if commodity not in self.COMMODITY_MAP:
        raise ValueError(f"Unknown commodity: {commodity}. Valid: {list(self.COMMODITY_MAP.keys())}")

    info = self.COMMODITY_MAP[commodity]

    # Build SoQL WHERE clause safely (values are from our COMMODITY_MAP, not user input)
    where_clauses = [
        f"cftc_commodity_code='{info['code']}'",
        f"contract_market_name='{info['contract']}'",
    ]

    # Date filtering with validation
    if start_date:
        # Validate date format before including
        validated_start = start_date.isoformat() if isinstance(start_date, date) else str(start_date)[:10]
        where_clauses.append(f"report_date_as_yyyy_mm_dd >= '{validated_start}'")

    if end_date:
        validated_end = end_date.isoformat() if isinstance(end_date, date) else str(end_date)[:10]
        where_clauses.append(f"report_date_as_yyyy_mm_dd <= '{validated_end}'")

    params = {
        "$where": " AND ".join(where_clauses),
        "$order": "report_date_as_yyyy_mm_dd DESC",
        "$limit": str(min(limit, 50000)),  # API max is 50k
    }

    response = await client.get(self.BASE_URL, params=params)
    response.raise_for_status()
    return self._parse_response(response.json(), commodity)
```

### 3. Data Normalization

Output format matching existing collectors. Robust parsing with error handling:

```python
def _safe_int(self, value: str | int | None, default: int = 0) -> int:
    """Safely convert value to int, handling None and malformed data."""
    if value is None:
        return default
    try:
        return int(value)
    except (ValueError, TypeError):
        logger.warning(f"Could not parse value to int: {value}")
        return default

def _parse_response(self, data: list[dict], commodity: str) -> pd.DataFrame:
    """Parse API response to standard format."""
    if not data:
        return self._empty_dataframe()

    records = []
    for row in data:
        try:
            timestamp = pd.to_datetime(row.get("report_date_as_yyyy_mm_dd"))
        except Exception as e:
            logger.warning(f"Skipping row with invalid timestamp: {e}")
            continue

        # Net positions with safe integer parsing
        comm_long = self._safe_int(row.get("prod_merc_positions_long"))
        comm_short = self._safe_int(row.get("prod_merc_positions_short"))
        spec_long = self._safe_int(row.get("m_money_positions_long_all"))
        spec_short = self._safe_int(row.get("m_money_positions_short_all"))
        swap_long = self._safe_int(row.get("swap_positions_long_all"))
        swap_short = self._safe_int(row.get("swap__positions_short_all"))
        oi = self._safe_int(row.get("open_interest_all"))

        comm_net = comm_long - comm_short
        spec_net = spec_long - spec_short
        swap_net = swap_long - swap_short

        commodity_lower = commodity.lower()
        records.extend([
            {"timestamp": timestamp, "series_id": f"cot_{commodity_lower}_comm_net", "source": "cftc", "value": comm_net, "unit": "contracts"},
            {"timestamp": timestamp, "series_id": f"cot_{commodity_lower}_spec_net", "source": "cftc", "value": spec_net, "unit": "contracts"},
            {"timestamp": timestamp, "series_id": f"cot_{commodity_lower}_swap_net", "source": "cftc", "value": swap_net, "unit": "contracts"},
            {"timestamp": timestamp, "series_id": f"cot_{commodity_lower}_oi", "source": "cftc", "value": oi, "unit": "contracts"},
            # Also store raw long/short for ratio calculations
            {"timestamp": timestamp, "series_id": f"cot_{commodity_lower}_comm_long", "source": "cftc", "value": comm_long, "unit": "contracts"},
            {"timestamp": timestamp, "series_id": f"cot_{commodity_lower}_comm_short", "source": "cftc", "value": comm_short, "unit": "contracts"},
            {"timestamp": timestamp, "series_id": f"cot_{commodity_lower}_spec_long", "source": "cftc", "value": spec_long, "unit": "contracts"},
            {"timestamp": timestamp, "series_id": f"cot_{commodity_lower}_spec_short", "source": "cftc", "value": spec_short, "unit": "contracts"},
        ])

    return pd.DataFrame(records)

def _empty_dataframe(self) -> pd.DataFrame:
    """Return empty DataFrame with correct schema."""
    return pd.DataFrame(columns=["timestamp", "series_id", "source", "value", "unit"])
```
