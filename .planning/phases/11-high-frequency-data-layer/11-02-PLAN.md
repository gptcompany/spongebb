```python
# src/liquidity/collectors/nyfed.py
"""NY Fed Markets API collector for RRP and SOMA data."""

import logging
from datetime import UTC, datetime, timedelta
from typing import Any

import httpx
import pandas as pd

from liquidity.collectors.base import BaseCollector, CollectorFetchError
from liquidity.collectors.registry import registry
from liquidity.config import Settings, get_settings

logger = logging.getLogger(__name__)

NYFED_BASE_URL = "https://markets.newyorkfed.org/api"


class NYFedCollector(BaseCollector[pd.DataFrame]):
    """Collector for NY Fed Markets API data.

    Provides access to:
    - RRP (Reverse Repo) daily operations
    - SOMA (System Open Market Account) holdings

    API Docs: https://markets.newyorkfed.org/static/docs/markets-api.html
    """

    def __init__(
        self,
        name: str = "nyfed",
        settings: Settings | None = None,
        **kwargs: Any,
    ) -> None:
        super().__init__(name=name, settings=settings, **kwargs)
        self._settings = settings or get_settings()
        self._client: httpx.AsyncClient | None = None

    async def _get_client(self) -> httpx.AsyncClient:
        if self._client is None or self._client.is_closed:
            self._client = httpx.AsyncClient(
                timeout=30.0,
                headers={"Accept": "application/json"},
            )
        return self._client

    async def collect_rrp(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect RRP (Reverse Repo) operations data.

        Args:
            start_date: Start date (default: 30 days ago)
            end_date: End date (default: today)

        Returns:
            DataFrame with columns: timestamp, series_id, source, value, unit
        """
        if start_date is None:
            start_date = datetime.now(UTC) - timedelta(days=30)
        if end_date is None:
            end_date = datetime.now(UTC)

        async def _fetch() -> pd.DataFrame:
            return await self._fetch_rrp(start_date, end_date)

        try:
            return await self.fetch_with_retry(_fetch, breaker_name="nyfed_rrp")
        except Exception as e:
            logger.error("NY Fed RRP fetch failed: %s", e)
            raise CollectorFetchError(f"NY Fed RRP fetch failed: {e}") from e

    async def _fetch_rrp(
        self,
        start_date: datetime,
        end_date: datetime,
    ) -> pd.DataFrame:
        """Fetch RRP operations from NY Fed API."""
        client = await self._get_client()

        # Use search endpoint for date range
        url = f"{NYFED_BASE_URL}/rp/all/all/results/search.json"
        params = {
            "startDate": start_date.strftime("%Y-%m-%d"),
            "endDate": end_date.strftime("%Y-%m-%d"),
        }

        response = await client.get(url, params=params)
        response.raise_for_status()

        data = response.json()
        operations = data.get("repo", {}).get("operations", [])

        if not operations:
            logger.warning("No RRP operations returned")
            return pd.DataFrame(
                columns=["timestamp", "series_id", "source", "value", "unit"]
            )

        rows = []
        for op in operations:
            # Filter for Reverse Repo (not regular repo)
            if "reverse" in op.get("operationType", "").lower():
                rows.append({
                    "timestamp": pd.to_datetime(op["operationDate"]),
                    "series_id": "RRP_DAILY",
                    "source": "nyfed",
                    "value": float(op.get("totalAmtAccepted", 0)) / 1e9,  # Convert to billions
                    "unit": "billions_usd",
                })

        df = pd.DataFrame(rows)
        if len(df) > 0:
            # Aggregate by date (multiple operations per day)
            df = df.groupby("timestamp").agg({
                "series_id": "first",
                "source": "first",
                "value": "sum",
                "unit": "first",
            }).reset_index()

        df = df.sort_values("timestamp").reset_index(drop=True)

        logger.info("Fetched %d RRP daily records", len(df))
        return df

    async def collect_soma(self) -> pd.DataFrame:
        """Collect SOMA holdings summary.

        Returns:
            DataFrame with SOMA holdings by security type.
        """
        async def _fetch() -> pd.DataFrame:
            return await self._fetch_soma()

        try:
            return await self.fetch_with_retry(_fetch, breaker_name="nyfed_soma")
        except Exception as e:
            logger.error("NY Fed SOMA fetch failed: %s", e)
            raise CollectorFetchError(f"NY Fed SOMA fetch failed: {e}") from e

    async def _fetch_soma(self) -> pd.DataFrame:
        """Fetch SOMA holdings from NY Fed API."""
        client = await self._get_client()

        url = f"{NYFED_BASE_URL}/soma/summary.json"
        response = await client.get(url)
        response.raise_for_status()

        data = response.json()
        soma_data = data.get("soma", {})

        as_of_date = soma_data.get("asOfDate")
        holdings = soma_data.get("holdings", [])

        rows = []
        for holding in holdings:
            rows.append({
                "timestamp": pd.to_datetime(as_of_date),
                "series_id": f"SOMA_{holding.get('securityType', 'UNKNOWN')}",
                "source": "nyfed",
                "value": float(holding.get("parValue", 0)) / 1e9,  # billions
                "unit": "billions_usd",
            })

        df = pd.DataFrame(rows)
        logger.info("Fetched SOMA holdings as of %s", as_of_date)
        return df

    async def collect(
        self,
        data_type: str = "rrp",
        **kwargs: Any,
    ) -> pd.DataFrame:
        """Generic collect method.

        Args:
            data_type: "rrp" or "soma"
            **kwargs: Passed to specific collector
        """
        if data_type == "rrp":
            return await self.collect_rrp(**kwargs)
        elif data_type == "soma":
            return await self.collect_soma()
        else:
            raise ValueError(f"Unknown data_type: {data_type}")

    async def close(self) -> None:
        if self._client is not None and not self._client.is_closed:
            await self._client.aclose()


# Register collector
registry.register("nyfed", NYFedCollector)
```

```python
# src/liquidity/collectors/swap_lines.py
"""Central Bank Swap Lines collector from NY Fed."""

import logging
from datetime import UTC, datetime, timedelta
from typing import Any

import httpx
import pandas as pd

from liquidity.collectors.base import BaseCollector, CollectorFetchError
from liquidity.collectors.registry import registry
from liquidity.config import Settings, get_settings

logger = logging.getLogger(__name__)

NYFED_BASE_URL = "https://markets.newyorkfed.org/api"

# Central bank partners with standing swap lines
SWAP_PARTNERS = {
    "ECB": "European Central Bank",
    "BOJ": "Bank of Japan",
    "BOE": "Bank of England",
    "SNB": "Swiss National Bank",
    "BOC": "Bank of Canada",
}


class SwapLinesCollector(BaseCollector[pd.DataFrame]):
    """Collector for central bank liquidity swap line data.

    Tracks USD swap line usage between Fed and major central banks.
    High usage indicates global USD funding stress.
    """

    def __init__(
        self,
        name: str = "swap_lines",
        settings: Settings | None = None,
        **kwargs: Any,
    ) -> None:
        super().__init__(name=name, settings=settings, **kwargs)
        self._settings = settings or get_settings()
        self._client: httpx.AsyncClient | None = None

    async def _get_client(self) -> httpx.AsyncClient:
        if self._client is None or self._client.is_closed:
            self._client = httpx.AsyncClient(
                timeout=30.0,
                headers={"Accept": "application/json"},
            )
        return self._client

    async def collect(
        self,
        start_date: datetime | None = None,
        end_date: datetime | None = None,
    ) -> pd.DataFrame:
        """Collect swap line operations data.

        Args:
            start_date: Start date (default: 90 days ago)
            end_date: End date (default: today)

        Returns:
            DataFrame with swap line usage by partner.
        """
        if start_date is None:
            start_date = datetime.now(UTC) - timedelta(days=90)
        if end_date is None:
            end_date = datetime.now(UTC)

        async def _fetch() -> pd.DataFrame:
            return await self._fetch_swap_lines(start_date, end_date)

        try:
            return await self.fetch_with_retry(_fetch)
        except Exception as e:
            logger.error("Swap lines fetch failed: %s", e)
            raise CollectorFetchError(f"Swap lines fetch failed: {e}") from e

    async def _fetch_swap_lines(
        self,
        start_date: datetime,
        end_date: datetime,
    ) -> pd.DataFrame:
        """Fetch swap line operations from NY Fed API."""
        client = await self._get_client()

        # Try FX swaps endpoint
        url = f"{NYFED_BASE_URL}/fxs/all/results/latest.json"

        try:
            response = await client.get(url)
            response.raise_for_status()
            data = response.json()
        except httpx.HTTPStatusError as e:
            if e.response.status_code == 404:
                # No swap operations (normal in calm markets)
                logger.info("No swap line operations (market calm)")
                return self._empty_df()
            raise

        operations = data.get("fxSwaps", {}).get("operations", [])

        if not operations:
            logger.info("No swap line operations in period")
            return self._empty_df()

        rows = []
        for op in operations:
            counterparty = op.get("counterparty", "UNKNOWN")
            rows.append({
                "timestamp": pd.to_datetime(op.get("settlementDate")),
                "series_id": f"SWAP_{counterparty}",
                "source": "nyfed",
                "value": float(op.get("amount", 0)) / 1e9,  # billions
                "unit": "billions_usd",
                "counterparty": counterparty,
            })

        df = pd.DataFrame(rows)
        df = df.sort_values("timestamp").reset_index(drop=True)

        logger.info("Fetched %d swap line operations", len(df))
        return df

    def _empty_df(self) -> pd.DataFrame:
        return pd.DataFrame(
            columns=["timestamp", "series_id", "source", "value", "unit", "counterparty"]
        )

    async def close(self) -> None:
        if self._client is not None and not self._client.is_closed:
            await self._client.aclose()


# Register collector
registry.register("swap_lines", SwapLinesCollector)
```

```python
import pytest
from datetime import UTC, datetime, timedelta

from liquidity.collectors.nyfed import NYFedCollector
from liquidity.collectors.swap_lines import SwapLinesCollector


@pytest.mark.integration
async def test_collect_rrp_real_api():
    """Test fetching real RRP data from NY Fed API."""
    collector = NYFedCollector()

    try:
        start = datetime.now(UTC) - timedelta(days=14)
        df = await collector.collect_rrp(start_date=start)

        # Should have data (daily operations)
        assert len(df) > 5, "Expected at least 5 records for 14 days"

        # Check data structure
        assert "timestamp" in df.columns
        assert "value" in df.columns
        assert df["series_id"].iloc[0] == "RRP_DAILY"

        # RRP should be reasonable (0 - 2.5 trillion = 0 - 2500 billion)
        latest_value = df["value"].iloc[-1]
        assert 0 <= latest_value < 3000, f"RRP value {latest_value} out of range"

    finally:
        await collector.close()


@pytest.mark.integration
async def test_collect_soma_real_api():
    """Test fetching SOMA holdings from NY Fed API."""
    collector = NYFedCollector()

    try:
        df = await collector.collect_soma()

        # Should have holdings data
        assert len(df) > 0

        # Should include Treasury securities
        assert any("TREASURY" in s or "Bills" in s or "Notes" in s

        await collector.close()
```
