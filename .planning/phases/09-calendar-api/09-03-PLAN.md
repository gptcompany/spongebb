@router.get("/events")

    start: date = None,
    end: date = None,
    event_type: str = None,
    impact: str = None,
):
    """Calendar events affecting liquidity."""
    if start is None:
        start = date.today()
    if end is None:
        end = start + timedelta(days=30)

    events = calendar_registry.get_events(start, end)

    if event_type:
        events = [e for e in events if e.event_type.value == event_type]
    if impact:
        events = [e for e in events if e.impact == impact]

    return {
        "start": start,
        "end": end,
        "count": len(events),
        "events": [event.to_dict() for event in events]

### 6. NautilusTrader Integration (API-05)
- Document the /regime/current endpoint format
- Add example Python client code in docs/nautilus_integration.md

```python
# Example NautilusTrader macro filter
import httpx

class LiquidityMacroFilter:
    def __init__(self, api_url: str = "http://localhost:8000"):
        self.api_url = api_url

    async def should_trade(self) -> bool:
        async with httpx.AsyncClient() as client:
            resp = await client.get(f"{self.api_url}/regime/current")
            data = resp.json()
            # Only trade in EXPANSION regime with high confidence
            return (
                data["regime"] == "EXPANSION" and
                data["confidence"] == "HIGH" and
                data["intensity"] > 50
            )
```
