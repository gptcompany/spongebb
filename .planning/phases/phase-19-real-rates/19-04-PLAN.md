# Plan 19-04: Inflation Expectations Dashboard

## Goal

Creare panel dashboard per visualizzare real rates, breakeven inflation, e oil-rates correlation.

## Components

### 1. Real Rates Chart
- Line chart con TIPS 10Y e 5Y
- Shading per zone (restrictive vs accommodative)

### 2. Breakeven Inflation Chart
- Line chart con BEI 10Y, 5Y, 5Y5Y forward
- Reference line a 2% (Fed target)
- Shading se > 2.5% (inflation concern) o < 1.5% (deflation risk)

### 3. Oil vs Real Rates Scatter
- Scatter plot oil returns vs real rates changes
- Regression line con R² display
- Color-coded per regime (normal/breakdown/surge)

### 4. Correlation Heatmap
- Matrice correlazione: oil, real rates 10Y, BEI 10Y, 5Y5Y forward
- Rolling window selezionabile (30d/90d/YTD)

```python
def create_real_rates_chart(df: pd.DataFrame) -> go.Figure:
    """Create real rates (TIPS yields) chart."""
    fig = go.Figure()

    fig.add_trace(go.Scatter(
        x=df.index, y=df["tips_10y"],
        name="10Y TIPS", line=dict(color="#1f77b4", width=2)
    ))
    fig.add_trace(go.Scatter(
        x=df.index, y=df["tips_5y"],
        name="5Y TIPS", line=dict(color="#ff7f0e", width=2)
    ))

    # Zero line
    fig.add_hline(y=0, line_dash="dash", line_color="gray", opacity=0.5)

    # Shading: restrictive (> 2%) vs accommodative (< 0%)
    fig.add_hrect(y0=2, y1=5, fillcolor="red", opacity=0.1,
                  annotation_text="Restrictive", annotation_position="top left")
    fig.add_hrect(y0=-2, y1=0, fillcolor="green", opacity=0.1,
                  annotation_text="Accommodative", annotation_position="bottom left")

    fig.update_layout(
        title="Real Rates (TIPS Yields)",
        yaxis_title="Yield (%)",
        hovermode="x unified",
        template="plotly_white",
    )

    return fig


def create_breakeven_chart(df: pd.DataFrame) -> go.Figure:
    """Create breakeven inflation chart."""
    fig = go.Figure()

    fig.add_trace(go.Scatter(
        x=df.index, y=df["bei_10y"],
        name="10Y BEI", line=dict(color="#2ca02c", width=2)
    ))
    fig.add_trace(go.Scatter(
        x=df.index, y=df["bei_5y"],
        name="5Y BEI", line=dict(color="#9467bd", width=2)
    ))
    fig.add_trace(go.Scatter(
        x=df.index, y=df["forward_5y5y"],
        name="5Y5Y Forward", line=dict(color="#d62728", width=2, dash="dash")
    ))

    # Fed target reference
    fig.add_hline(y=2.0, line_dash="dot", line_color="black",
                  annotation_text="Fed Target (2%)")

    # Warning zones
    fig.add_hrect(y0=2.5, y1=5, fillcolor="orange", opacity=0.1,
                  annotation_text="Inflation Concern")
    fig.add_hrect(y0=0, y1=1.5, fillcolor="blue", opacity=0.1,
                  annotation_text="Deflation Risk")

    fig.update_layout(
        title="Breakeven Inflation Expectations",
        yaxis_title="Inflation (%)",
        hovermode="x unified",
        template="plotly_white",
    )

    return fig


def create_oil_rates_scatter(df: pd.DataFrame) -> go.Figure:
```

### Task 2: Add callbacks

```python
# In callbacks_main.py or new callbacks/inflation_callbacks.py

@app.callback(
    [
        Output("real-rates-chart", "figure"),
        Output("breakeven-chart", "figure"),
        Output("oil-rates-scatter", "figure"),
        Output("inflation-indicators", "children"),
    ],
    [Input("interval-component", "n_intervals")],
)
async def update_inflation_panel(_):
    """Update inflation panel charts."""
    rates_analyzer = RealRatesAnalyzer()
    oil_analyzer = OilRealRatesAnalyzer()

    # Fetch data
    rates_df = await rates_analyzer.calculate_breakeven()
    oil_df = await oil_analyzer.compute_correlation()
    state = await rates_analyzer.get_current_state()

    # Create charts
    real_rates_fig = create_real_rates_chart(rates_df)
    bei_fig = create_breakeven_chart(rates_df)
    scatter_fig = create_oil_rates_scatter(oil_df)

    # Create indicators
    indicators = create_inflation_indicators(state)

    return real_rates_fig, bei_fig, scatter_fig, indicators


def create_inflation_indicators(state: RealRatesState) -> html.Div:
    """Create inflation indicator badges."""
    return html.Div([
        dbc.Badge(f"10Y TIPS: {state.tips_10y:.2f}%", color="info", className="me-2"),
        dbc.Badge(f"10Y BEI: {state.bei_10y:.2f}%", color="success", className="me-2"),
        dbc.Badge(f"5Y5Y Fwd: {state.forward_5y5y:.2f}%",
                  color="warning" if state.forward_5y5y > 2.5 else "success"),
    ])
```
