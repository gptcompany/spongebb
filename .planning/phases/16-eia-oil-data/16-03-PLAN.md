### 2. Create Refinery-Specific Methods

```python
async def collect_refinery_utilization(
    self,
    regions: list[str] | None = None,
    lookback_weeks: int = 52
) -> pd.DataFrame:
    """Collect refinery utilization by region."""
    if regions is None:
        regions = ["us", "padd1", "padd3", "padd5"]

    series_ids = [f"refinery_utilization_{r}" for r in regions]
    return await self.collect(series_ids=series_ids, lookback_weeks=lookback_weeks)

def calculate_utilization_signal(self, df: pd.DataFrame) -> str:
    """Classify utilization level."""
    latest = df[df["series_id"] == "refinery_utilization_us"].iloc[-1]["value"]
    if latest > 95:
        return "TIGHT"  # Supply constraint risk
    elif latest > 90:
        return "NORMAL"
    elif latest > 85:
        return "SOFT"
    else:
        return "WEAK"  # Demand concerns
```
